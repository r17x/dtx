### 5.3.1 MCP Server Binary (3h)

```rust
// crates/dtx/src/cmd/mcp.rs

pub async fn mcp_command(args: McpArgs) -> Result<()> {
    // Initialize orchestrator
    let event_bus = Arc::new(EventBus::new());
    let orchestrator = create_orchestrator(event_bus.clone()).await?;

    // Create protocol handler
    let protocol_handler = DxtProtocolHandler::new(orchestrator);
    let mcp_handler = DefaultMcpHandler::new(protocol_handler);

    // Serve over stdio
    serve_stdio(|request| async {
        dispatch_mcp(&mcp_handler, request).await
    }).await
}

async fn dispatch_mcp<H: McpHandler>(handler: &H, request: Request) -> Response {
    match request.method.as_str() {
        "initialize" => {
            let params: InitializeParams = parse_params(&request).unwrap();
            match handler.initialize(params).await {
                Ok(result) => Response::success(request.id, serde_json::to_value(result).unwrap()),
                Err(e) => Response::error(request.id, e),
            }
        }
        "resources/list" => {
            match handler.list_resources().await {
                Ok(result) => Response::success(request.id, serde_json::to_value(result).unwrap()),
                Err(e) => Response::error(request.id, e),
            }
        }
        "tools/list" => {
            match handler.list_tools().await {
                Ok(result) => Response::success(request.id, serde_json::to_value(result).unwrap()),
                Err(e) => Response::error(request.id, e),
            }
        }
        "tools/call" => {
            let params: CallToolParams = parse_params(&request).unwrap();
            match handler.call_tool(params).await {
                Ok(result) => Response::success(request.id, serde_json::to_value(result).unwrap()),
                Err(e) => Response::error(request.id, e),
            }
        }
        _ => {
            // Fall through to regular protocol handler
            dispatch(&protocol_handler, request).await
        }
    }
}
```

### 5.3.3 Resource Subscriptions (2h)

```rust
// Notify MCP client of resource changes
impl MCP notifications for resource updates

pub async fn send_resource_changed_notification(
    transport: &mut impl ServerTransport,
    uri: &str,
) -> Result<(), TransportError> {
    let notification = Request::notification(
        "notifications/resources/updated",
        Some(serde_json::json!({ "uri": uri })),
    );
    transport.notify(notification).await
}
```
