### 4.3.1 Update CLI Commands (4h)

```rust
// crates/dtx/src/cmd/start.rs

pub async fn start_command(args: StartArgs) -> Result<()> {
    let ctx = Context::with_timeout(Duration::from_secs(300));

    // Load services from database
    let services = load_services(&args.project_id).await?;

    // Create event bus
    let event_bus = Arc::new(EventBus::new());

    // Create middleware stack
    let middleware = MiddlewareStack::new()
        .layer(LoggingMiddleware::new())
        .layer(MetricsMiddleware::new(metrics_registry.clone()))
        .layer(TimeoutMiddleware::new(Duration::from_secs(60)));

    // Create orchestrator
    let mut orchestrator = Orchestrator::new(event_bus.clone(), middleware);

    // Register processes
    for service in services {
        let process = Process::new(service.into(), event_bus.clone());
        orchestrator.register(process);
    }

    // Start based on mode
    if args.foreground {
        run_foreground(orchestrator, event_bus, ctx).await
    } else {
        run_tui(orchestrator, event_bus, ctx).await
    }
}
```

**Acceptance:**
- [ ] start command uses Orchestrator v2
- [ ] stop command uses Orchestrator v2
- [ ] status command uses new types
- [ ] logs command uses EventBus

### 4.3.2 Update TUI (3h)

// crates/dtx/src/tui/app.rs

pub struct App {
    resources: HashMap<ResourceId, ResourceView>,
    logs: VecDeque<LogEntry>,
    event_subscriber: EventSubscriber,
}

impl App {
    pub fn poll_events(&mut self) {
        while let Some(event) = self.event_subscriber.try_recv() {
            match event {
                LifecycleEvent::Starting { id, .. } => {
                    self.resources.get_mut(&id).map(|r| r.state = "starting");
                }
                LifecycleEvent::Running { id, pid, .. } => {
                    self.resources.get_mut(&id).map(|r| {
                        r.state = "running";
                        r.pid = pid;
                    });
                }
                LifecycleEvent::Log { id, line, stream, .. } => {
                    self.logs.push_back(LogEntry { id, line, stream, .. });
                }
                // ... other events
            }
        }
    }
}
```

**Acceptance:**
- [ ] TUI uses EventSubscriber
- [ ] State updates from events

### 4.3.3 Update Web Handlers (3h)

// crates/dtx-web/src/handlers/api.rs

pub async fn start_resource(
    State(state): State<AppState>,
    Path(id): Path<String>,
) -> impl IntoResponse {
    let ctx = Context::with_timeout(Duration::from_secs(60));

    match state.orchestrator.start(&ResourceId::new(id), &ctx).await {
        Ok(_) => Json(json!({"status": "started"})),
        Err(e) => (StatusCode::INTERNAL_SERVER_ERROR, Json(e.to_json())),
    }
}

pub async fn events_stream(
    State(state): State<AppState>,
) -> Sse<impl Stream<Item = ...>> {
    let subscriber = state.event_bus.subscribe();

    let stream = async_stream::stream! {
        let mut subscriber = subscriber;
        while let Some(event) = subscriber.recv().await {
            let data = serde_json::to_string(&event).unwrap();
            yield Ok(Event::default().data(data));
        }
    };

    Sse::new(stream)
}
```

#[tokio::test]
async fn test_full_lifecycle() {
    let event_bus = Arc::new(EventBus::new());
    let middleware = MiddlewareStack::new()
        .layer(LoggingMiddleware::new());

    let mut orchestrator = Orchestrator::new(event_bus.clone(), middleware);

    // Register test processes
    orchestrator.register(Process::new(ProcessConfig {
        id: ResourceId::new("db"),
        command: "echo 'db started'".into(),
        ..Default::default()
    }, event_bus.clone()));

    orchestrator.register(Process::new(ProcessConfig {
        id: ResourceId::new("api"),
        command: "echo 'api started'".into(),
        depends_on: vec![Dependency {
            target: ResourceId::new("db"),
            condition: DependencyCondition::Completed,
        }],
        ..Default::default()
    }, event_bus.clone()));

    // Start all
    let ctx = Context::with_timeout(Duration::from_secs(10));
    orchestrator.start_all(&ctx).await.unwrap();

    // Verify order via events
    let events = event_bus.replay(&EventFilter::all());
    // db should start before api
}
