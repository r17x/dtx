> Metrics, tracing, and telemetry for full observability.

---

## Goal

Implement comprehensive observability middleware including metrics collection and distributed tracing.

---

## User Value

```
- Monitor operation latency and throughput
- Track success/failure rates
- Trace requests across components
- Debug complex issues with spans
- Export to standard observability backends
```

```
IN SCOPE:
  - MetricsMiddleware implementation
  - TracingMiddleware implementation
  - Counter, histogram, gauge metrics
  - Prometheus format export
  - OpenTelemetry span creation

  - Labels/attributes for resource/operation

OUT OF SCOPE:
  - Full OTLP exporter implementation (use opentelemetry-otlp crate)
  - Metrics aggregation service
```

### 2.3.1 Metrics Types (2h)

**File:** `crates/dtx-middleware/src/metrics/types.rs`

```rust
use std::collections::HashMap;
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;
use parking_lot::RwLock;

/// A counter metric
#[derive(Default)]
pub struct Counter {
    value: AtomicU64,
}

impl Counter {
    pub fn inc(&self) {
        self.value.fetch_add(1, Ordering::Relaxed);
    }

    pub fn inc_by(&self, n: u64) {
        self.value.fetch_add(n, Ordering::Relaxed);
    }

    pub fn get(&self) -> u64 {
        self.value.load(Ordering::Relaxed)
    }
}

/// A gauge metric
pub struct Gauge {
    value: AtomicU64,
}

impl Default for Gauge {
    fn default() -> Self {
        Self {
            value: AtomicU64::new(0),
        }
    }
}

impl Gauge {
    pub fn set(&self, value: u64) {
        self.value.store(value, Ordering::Relaxed);
    }

    pub fn inc(&self) {
        self.value.fetch_add(1, Ordering::Relaxed);
    }

    pub fn dec(&self) {
        self.value.fetch_sub(1, Ordering::Relaxed);
    }

    pub fn get(&self) -> u64 {
        self.value.load(Ordering::Relaxed)
    }
}

/// A histogram metric with buckets
pub struct Histogram {
    buckets: Vec<f64>,
    counts: Vec<AtomicU64>,
    sum: AtomicU64,
    count: AtomicU64,
}

impl Histogram {
    pub fn new(buckets: Vec<f64>) -> Self {
        let counts = buckets.iter().map(|_| AtomicU64::new(0)).collect();
        Self {
            buckets,
            counts,
            sum: AtomicU64::new(0),
            count: AtomicU64::new(0),
        }
    }

    pub fn default_buckets() -> Self {
        Self::new(vec![
            0.001, 0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1.0, 2.5, 5.0, 10.0,
        ])
    }

    pub fn observe(&self, value: f64) {
        for (i, bucket) in self.buckets.iter().enumerate() {
            if value <= *bucket {
                self.counts[i].fetch_add(1, Ordering::Relaxed);
            }
        }
        self.sum.fetch_add((value * 1000.0) as u64, Ordering::Relaxed);
        self.count.fetch_add(1, Ordering::Relaxed);
    }

    pub fn get_count(&self) -> u64 {
        self.count.load(Ordering::Relaxed)
    }

    pub fn get_sum(&self) -> f64 {
        self.sum.load(Ordering::Relaxed) as f64 / 1000.0
    }
}

/// Label set for metrics
#[derive(Clone, Debug, Default, Eq, PartialEq, Hash)]
pub struct Labels {
    labels: Vec<(String, String)>,
}

impl Labels {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn add(mut self, key: impl Into<String>, value: impl Into<String>) -> Self {
        self.labels.push((key.into(), value.into()));
        self
    }

    pub fn to_prometheus(&self) -> String {
        if self.labels.is_empty() {
            return String::new();
        }
        let parts: Vec<String> = self
            .labels
            .iter()
            .map(|(k, v)| format!("{}=\"{}\"", k, v))
            .collect();
        format!("{{{}}}", parts.join(","))
    }
}
```

**Acceptance:**
- [ ] Counter, Gauge, Histogram types
- [ ] Thread-safe operations
- [ ] Labels support

### 2.3.2 MetricsRegistry (2h)

**File:** `crates/dtx-middleware/src/metrics/registry.rs`

```rust
use std::collections::HashMap;
use std::sync::Arc;
use parking_lot::RwLock;

use super::types::{Counter, Gauge, Histogram, Labels};

/// Central registry for all metrics
#[derive(Default)]
pub struct MetricsRegistry {
    counters: RwLock<HashMap<(String, Labels), Arc<Counter>>>,
    gauges: RwLock<HashMap<(String, Labels), Arc<Gauge>>>,
    histograms: RwLock<HashMap<(String, Labels), Arc<Histogram>>>,
}

impl MetricsRegistry {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn counter(&self, name: &str, labels: Labels) -> Arc<Counter> {
        let key = (name.to_string(), labels);
        let mut counters = self.counters.write();
        counters
            .entry(key)
            .or_insert_with(|| Arc::new(Counter::default()))
            .clone()
    }

    pub fn gauge(&self, name: &str, labels: Labels) -> Arc<Gauge> {
        let key = (name.to_string(), labels);
        let mut gauges = self.gauges.write();
        gauges
            .entry(key)
            .or_insert_with(|| Arc::new(Gauge::default()))
            .clone()
    }

    pub fn histogram(&self, name: &str, labels: Labels) -> Arc<Histogram> {
        let key = (name.to_string(), labels);
        let mut histograms = self.histograms.write();
        histograms
            .entry(key)
            .or_insert_with(|| Arc::new(Histogram::default_buckets()))
            .clone()
    }

    /// Export metrics in Prometheus format
    pub fn export_prometheus(&self) -> String {
        let mut output = String::new();

        // Export counters
        for ((name, labels), counter) in self.counters.read().iter() {
            output.push_str(&format!(
                "{}{} {}\n",
                name,
                labels.to_prometheus(),
                counter.get()
            ));
        }

        // Export gauges
        for ((name, labels), gauge) in self.gauges.read().iter() {
            output.push_str(&format!(
                "{}{} {}\n",
                name,
                labels.to_prometheus(),
                gauge.get()
            ));
        }

        // Export histograms
        for ((name, labels), histogram) in self.histograms.read().iter() {
            output.push_str(&format!(
                "{}_count{} {}\n",
                name,
                labels.to_prometheus(),
                histogram.get_count()
            ));
            output.push_str(&format!(
                "{}_sum{} {}\n",
                name,
                labels.to_prometheus(),
                histogram.get_sum()
            ));
        }

        output
    }
}
```

**Acceptance:**
- [ ] Metric registration
- [ ] Prometheus export
- [ ] Thread-safe access

### 2.3.3 MetricsMiddleware (3h)

**File:** `crates/dtx-middleware/src/metrics/middleware.rs`

```rust
use async_trait::async_trait;
use std::sync::Arc;
use std::time::Instant;

use dtx_core::{
    Context, Result,
    middleware::{Middleware, Next, Operation, Response},
};

use super::registry::MetricsRegistry;
use super::types::Labels;

/// Middleware that collects metrics for all operations
pub struct MetricsMiddleware {
    registry: Arc<MetricsRegistry>,
}

impl MetricsMiddleware {
    pub fn new(registry: Arc<MetricsRegistry>) -> Self {
        Self { registry }
    }

    fn labels_for_op(&self, op: &Operation, success: bool) -> Labels {
        Labels::new()
            .add("operation", op.name())
            .add("success", if success { "true" } else { "false" })
    }

    fn record_operation(&self, op: &Operation, elapsed: std::time::Duration, success: bool) {
        let labels = self.labels_for_op(op, success);

        // Increment operation counter
        self.registry
            .counter("dtx_operations_total", labels.clone())
            .inc();

        // Record duration
        self.registry
            .histogram("dtx_operation_duration_seconds", labels)
            .observe(elapsed.as_secs_f64());
    }
}

#[async_trait]
impl Middleware for MetricsMiddleware {
    fn name(&self) -> &'static str {
        "metrics"
    }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        // Track in-flight operations
        let in_flight = self.registry.gauge(
            "dtx_operations_in_flight",
            Labels::new().add("operation", op.name()),
        );
        in_flight.inc();

        let start = Instant::now();
        let result = next.run(op.clone(), ctx).await;
        let elapsed = start.elapsed();

        in_flight.dec();

        self.record_operation(&op, elapsed, result.is_ok());

        result
    }
}
```

**Acceptance:**
- [ ] Records operation count
- [ ] Records duration histogram
- [ ] Tracks in-flight operations
- [ ] Labels by operation type

### 2.3.4 Tracing Types (2h)

**File:** `crates/dtx-middleware/src/tracing/types.rs`

```rust
use std::collections::HashMap;
use std::fmt;

/// Unique trace identifier
#[derive(Clone, Debug, Eq, PartialEq, Hash)]
pub struct TraceId(pub [u8; 16]);

impl TraceId {
    pub fn new() -> Self {
        let mut bytes = [0u8; 16];
        getrandom::getrandom(&mut bytes).expect("failed to generate trace id");
        Self(bytes)
    }

    pub fn from_hex(s: &str) -> Option<Self> {
        let bytes = hex::decode(s).ok()?;
        if bytes.len() != 16 {
            return None;
        }
        let mut arr = [0u8; 16];
        arr.copy_from_slice(&bytes);
        Some(Self(arr))
    }

    pub fn to_hex(&self) -> String {
        hex::encode(self.0)
    }
}

impl fmt::Display for TraceId {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{}", self.to_hex())
    }
}

/// Unique span identifier
#[derive(Clone, Debug, Eq, PartialEq, Hash)]
pub struct SpanId(pub [u8; 8]);

impl SpanId {
    pub fn new() -> Self {
        let mut bytes = [0u8; 8];
        getrandom::getrandom(&mut bytes).expect("failed to generate span id");
        Self(bytes)
    }

    pub fn to_hex(&self) -> String {
        hex::encode(self.0)
    }
}

impl fmt::Display for SpanId {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{}", self.to_hex())
    }
}

/// Trace context for propagation
#[derive(Clone, Debug)]
pub struct TraceContext {
    pub trace_id: TraceId,
    pub span_id: SpanId,
    pub parent_span_id: Option<SpanId>,
    pub sampled: bool,
}

impl TraceContext {
    pub fn new() -> Self {
        Self {
            trace_id: TraceId::new(),
            span_id: SpanId::new(),
            parent_span_id: None,
            sampled: true,
        }
    }

    pub fn child(&self) -> Self {
        Self {
            trace_id: self.trace_id.clone(),
            span_id: SpanId::new(),
            parent_span_id: Some(self.span_id.clone()),
            sampled: self.sampled,
        }
    }

    /// W3C Trace Context traceparent header
    pub fn to_traceparent(&self) -> String {
        let sampled = if self.sampled { "01" } else { "00" };
        format!(
            "00-{}-{}-{}",
            self.trace_id.to_hex(),
            self.span_id.to_hex(),
            sampled
        )
    }

    /// Parse W3C Trace Context traceparent header
    pub fn from_traceparent(s: &str) -> Option<Self> {
        let parts: Vec<&str> = s.split('-').collect();
        if parts.len() != 4 || parts[0] != "00" {
            return None;
        }

        let trace_id = TraceId::from_hex(parts[1])?;
        let span_id_bytes = hex::decode(parts[2]).ok()?;
        if span_id_bytes.len() != 8 {
            return None;
        }
        let mut span_arr = [0u8; 8];
        span_arr.copy_from_slice(&span_id_bytes);
        let span_id = SpanId(span_arr);
        let sampled = parts[3] == "01";

        Some(Self {
            trace_id,
            span_id,
            parent_span_id: None,
            sampled,
        })
    }
}

/// Span status
#[derive(Clone, Debug, Default)]
pub enum SpanStatus {
    #[default]
    Unset,
    Ok,
    Error { message: String },
}

/// Span attributes
pub type SpanAttributes = HashMap<String, SpanValue>;

/// Attribute value types
#[derive(Clone, Debug)]
pub enum SpanValue {
    String(String),
    Int(i64),
    Float(f64),
    Bool(bool),
}

impl From<&str> for SpanValue {
    fn from(s: &str) -> Self {
        Self::String(s.to_string())
    }
}

impl From<String> for SpanValue {
    fn from(s: String) -> Self {
        Self::String(s)
    }
}

impl From<i64> for SpanValue {
    fn from(n: i64) -> Self {
        Self::Int(n)
    }
}

impl From<f64> for SpanValue {
    fn from(n: f64) -> Self {
        Self::Float(n)
    }
}

impl From<bool> for SpanValue {
    fn from(b: bool) -> Self {
        Self::Bool(b)
    }
}
```

**Acceptance:**
- [ ] TraceId and SpanId types
- [ ] TraceContext with propagation
- [ ] W3C Trace Context support

### 2.3.5 Span and SpanBuilder (3h)

**File:** `crates/dtx-middleware/src/tracing/span.rs`

```rust
use std::collections::HashMap;
use std::time::{Instant, SystemTime};

use super::types::{TraceContext, SpanId, SpanStatus, SpanAttributes, SpanValue};

/// A completed span
#[derive(Clone, Debug)]
pub struct Span {
    pub name: String,
    pub trace_context: TraceContext,
    pub start_time: SystemTime,
    pub end_time: SystemTime,
    pub duration_ms: u64,
    pub status: SpanStatus,
    pub attributes: SpanAttributes,
    pub events: Vec<SpanEvent>,
}

/// Span event (log entry within a span)
#[derive(Clone, Debug)]
pub struct SpanEvent {
    pub name: String,
    pub timestamp: SystemTime,
    pub attributes: SpanAttributes,
}

/// Active span that can be modified
pub struct ActiveSpan {
    name: String,
    trace_context: TraceContext,
    start_time: SystemTime,
    start_instant: Instant,
    status: SpanStatus,
    attributes: SpanAttributes,
    events: Vec<SpanEvent>,
}

impl ActiveSpan {
    pub fn new(name: impl Into<String>, trace_context: TraceContext) -> Self {
        Self {
            name: name.into(),
            trace_context,
            start_time: SystemTime::now(),
            start_instant: Instant::now(),
            status: SpanStatus::Unset,
            attributes: HashMap::new(),
            events: Vec::new(),
        }
    }

    pub fn trace_context(&self) -> &TraceContext {
        &self.trace_context
    }

    pub fn set_attribute(&mut self, key: impl Into<String>, value: impl Into<SpanValue>) {
        self.attributes.insert(key.into(), value.into());
    }

    pub fn set_status(&mut self, status: SpanStatus) {
        self.status = status;
    }

    pub fn set_ok(&mut self) {
        self.status = SpanStatus::Ok;
    }

    pub fn set_error(&mut self, message: impl Into<String>) {
        self.status = SpanStatus::Error {
            message: message.into(),
        };
    }

    pub fn add_event(&mut self, name: impl Into<String>, attributes: SpanAttributes) {
        self.events.push(SpanEvent {
            name: name.into(),
            timestamp: SystemTime::now(),
            attributes,
        });
    }

    pub fn end(self) -> Span {
        let end_time = SystemTime::now();
        let duration_ms = self.start_instant.elapsed().as_millis() as u64;

        Span {
            name: self.name,
            trace_context: self.trace_context,
            start_time: self.start_time,
            end_time,
            duration_ms,
            status: self.status,
            attributes: self.attributes,
            events: self.events,
        }
    }
}

/// Builder for creating spans
pub struct SpanBuilder {
    name: String,
    parent_context: Option<TraceContext>,
    attributes: SpanAttributes,
}

impl SpanBuilder {
    pub fn new(name: impl Into<String>) -> Self {
        Self {
            name: name.into(),
            parent_context: None,
            attributes: HashMap::new(),
        }
    }

    pub fn with_parent(mut self, context: TraceContext) -> Self {
        self.parent_context = Some(context);
        self
    }

    pub fn with_attribute(mut self, key: impl Into<String>, value: impl Into<SpanValue>) -> Self {
        self.attributes.insert(key.into(), value.into());
        self
    }

    pub fn start(self) -> ActiveSpan {
        let trace_context = match self.parent_context {
            Some(parent) => parent.child(),
            None => TraceContext::new(),
        };

        let mut span = ActiveSpan::new(self.name, trace_context);
        span.attributes = self.attributes;
        span
    }
}
```

**Acceptance:**
- [ ] ActiveSpan for in-progress operations
- [ ] Span for completed operations
- [ ] SpanBuilder for ergonomic creation
- [ ] Event recording within spans
- [ ] Status tracking

### 2.3.6 TracingMiddleware (3h)

**File:** `crates/dtx-middleware/src/tracing/middleware.rs`

```rust
use async_trait::async_trait;
use std::sync::Arc;
use parking_lot::RwLock;

use dtx_core::{
    Context, Result,
    middleware::{Middleware, Next, Operation, Response},
};

use super::span::{Span, SpanBuilder, ActiveSpan};
use super::types::{TraceContext, SpanStatus};

/// Trait for span exporters
pub trait SpanExporter: Send + Sync {
    fn export(&self, span: Span);
}

/// Simple in-memory exporter for testing
#[derive(Default)]
pub struct InMemoryExporter {
    spans: RwLock<Vec<Span>>,
}

impl InMemoryExporter {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn spans(&self) -> Vec<Span> {
        self.spans.read().clone()
    }

    pub fn clear(&self) {
        self.spans.write().clear();
    }
}

impl SpanExporter for InMemoryExporter {
    fn export(&self, span: Span) {
        self.spans.write().push(span);
    }
}

/// Middleware that creates spans for all operations
pub struct TracingMiddleware {
    service_name: String,
    exporter: Arc<dyn SpanExporter>,
}

impl TracingMiddleware {
    pub fn new(service_name: impl Into<String>, exporter: Arc<dyn SpanExporter>) -> Self {
        Self {
            service_name: service_name.into(),
            exporter,
        }
    }

    fn create_span(&self, op: &Operation, ctx: &Context) -> ActiveSpan {
        let span_name = format!("dtx.{}", op.name());

        let mut builder = SpanBuilder::new(&span_name)
            .with_attribute("dtx.service", self.service_name.clone())
            .with_attribute("dtx.operation", op.name());

        // Extract parent context from Context if present
        if let Some(trace_ctx) = ctx.get::<TraceContext>() {
            builder = builder.with_parent(trace_ctx.clone());
        }

        // Add resource ID if present
        if let Some(resource_id) = op.resource_id() {
            builder = builder.with_attribute("dtx.resource_id", resource_id.to_string());
        }

        builder.start()
    }
}

#[async_trait]
impl Middleware for TracingMiddleware {
    fn name(&self) -> &'static str {
        "tracing"
    }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        let mut span = self.create_span(&op, &ctx);

        // Propagate trace context to child operations
        let ctx = ctx.with_value(span.trace_context().clone());

        let result = next.run(op, ctx).await;

        // Set span status based on result
        match &result {
            Ok(_) => span.set_ok(),
            Err(e) => span.set_error(e.to_string()),
        }

        // Export the completed span
        self.exporter.export(span.end());

        result
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use dtx_core::middleware::{Handler, MiddlewareStackBuilder};

    struct TestHandler;

    #[async_trait]
    impl Handler for TestHandler {
        async fn handle(&self, _op: Operation, _ctx: Context) -> Result<Response> {
            Ok(Response::ok())
        }
    }

    #[tokio::test]
    async fn tracing_middleware_creates_spans() {
        let exporter = Arc::new(InMemoryExporter::new());
        let chain = MiddlewareStackBuilder::new()
            .layer(TracingMiddleware::new("test-service", exporter.clone()))
            .build(TestHandler);

        chain.execute(Operation::StartAll, Context::new()).await.unwrap();

        let spans = exporter.spans();
        assert_eq!(spans.len(), 1);
        assert!(spans[0].name.contains("start_all"));
    }

    #[tokio::test]
    async fn tracing_propagates_context() {
        let exporter = Arc::new(InMemoryExporter::new());
        let chain = MiddlewareStackBuilder::new()
            .layer(TracingMiddleware::new("test-service", exporter.clone()))
            .build(TestHandler);

        // Create parent context
        let parent = TraceContext::new();
        let ctx = Context::new().with_value(parent.clone());

        chain.execute(Operation::StartAll, ctx).await.unwrap();

        let spans = exporter.spans();
        assert_eq!(spans[0].trace_context.trace_id, parent.trace_id);
    }
}
```
