```rust
        Self::new(s)
```
```rust
    }
}
```

### 1.1.2 ResourceKind Enum (2h)

**File:** `crates/dtx-core/src/resource/kind.rs`

```rust
use serde::{Deserialize, Serialize};

/// The type of resource being orchestrated
#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum ResourceKind {
    /// Native OS process
    Process,
    /// Container (Docker, Podman)
    Container,
    /// Virtual machine
    VM,
    /// AI agent
    Agent,
    /// Plugin-defined resource type
    #[serde(rename = "custom")]
    Custom(u16),
}

impl ResourceKind {
    pub fn as_str(&self) -> &'static str {
        match self {
            Self::Process => "process",
            Self::Container => "container",
            Self::VM => "vm",
            Self::Agent => "agent",
            Self::Custom(_) => "custom",
        }
    }
}

impl std::fmt::Display for ResourceKind {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::Custom(id) => write!(f, "custom:{}", id),
            _ => write!(f, "{}", self.as_str()),
        }
    }
}
```

**Acceptance:**
- [ ] All resource kinds defined
- [ ] Custom kind supports plugin ID
- [ ] Serialization works
- [ ] Display trait implemented

### 1.1.3 ResourceState Enum (3h)

**File:** `crates/dtx-core/src/resource/state.rs`

```rust
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

/// The current state of a resource
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(tag = "state", rename_all = "lowercase")]
pub enum ResourceState {
    /// Resource created but not started
    Pending,

    /// Resource is starting up
    Starting {
        started_at: DateTime<Utc>,
    },

    /// Resource is running and ready
    Running {
        pid: Option<u32>,
        started_at: DateTime<Utc>,
    },

    /// Resource is stopping
    Stopping {
        started_at: DateTime<Utc>,
        stopping_at: DateTime<Utc>,
    },

    /// Resource has stopped normally
    Stopped {
        exit_code: Option<i32>,
        started_at: DateTime<Utc>,
        stopped_at: DateTime<Utc>,
    },

    /// Resource has failed
    Failed {
        error: String,
        exit_code: Option<i32>,
        started_at: Option<DateTime<Utc>>,
        failed_at: DateTime<Utc>,
    },
}

impl ResourceState {
    pub fn is_running(&self) -> bool {
        matches!(self, Self::Running { .. })
    }

    pub fn is_stopped(&self) -> bool {
        matches!(self, Self::Stopped { .. } | Self::Failed { .. })
    }

    pub fn is_transitioning(&self) -> bool {
        matches!(self, Self::Starting { .. } | Self::Stopping { .. })
    }

    pub fn exit_code(&self) -> Option<i32> {
        match self {
            Self::Stopped { exit_code, .. } => *exit_code,
            Self::Failed { exit_code, .. } => *exit_code,
            _ => None,
        }
    }
}

impl Default for ResourceState {
    fn default() -> Self {
        Self::Pending
    }
}
```

**Acceptance:**
- [ ] All states defined with timestamps
- [ ] State queries (is_running, etc.)
- [ ] Default is Pending
- [ ] Serializes with state tag

**Tests:**
```rust
#[test]
fn state_transitions() {
    let state = ResourceState::Pending;
    assert!(!state.is_running());
    assert!(!state.is_stopped());
}

#[test]

    let state = ResourceState::Running {
        pid: Some(1234),
        started_at: Utc::now(),
    };
    let json = serde_json::to_value(&state).unwrap();
    assert_eq!(json["state"], "running");
    assert_eq!(json["pid"], 1234);
}
```

### 1.1.4 Resource Trait (4h)

**File:** `crates/dtx-core/src/resource/trait.rs`

```rust
use async_trait::async_trait;
use std::any::Any;

use super::{ResourceId, ResourceKind, ResourceState};
use crate::context::Context;
use crate::error::Result;

/// A resource that can be orchestrated by dtx
#[async_trait]
pub trait Resource: Send + Sync {
    /// Unique identifier for this resource
    fn id(&self) -> &ResourceId;

    /// The kind of resource
    fn kind(&self) -> ResourceKind;

    /// Current state
    fn state(&self) -> &ResourceState;

    /// Start the resource
    async fn start(&mut self, ctx: &Context) -> Result<()>;

    /// Stop the resource gracefully
    async fn stop(&mut self, ctx: &Context) -> Result<()>;

    /// Force stop the resource
    async fn kill(&mut self, ctx: &Context) -> Result<()> {
        self.stop(ctx).await
    }

    /// Restart the resource
    async fn restart(&mut self, ctx: &Context) -> Result<()> {
        self.stop(ctx).await?;
        self.start(ctx).await
    }

    /// Check health status
    async fn health(&self) -> HealthStatus {
        HealthStatus::Unknown
    }

    /// Get logs stream
    fn logs(&self) -> Option<Box<dyn LogStream>> {
        None
    }

    /// Downcast to concrete type
    fn as_any(&self) -> &dyn Any;

    /// Downcast to concrete type (mutable)
    fn as_any_mut(&mut self) -> &mut dyn Any;
}

/// Health status of a resource
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum HealthStatus {
    Healthy,
    Unhealthy { reason: String },
    Unknown,
}

impl HealthStatus {
    pub fn is_healthy(&self) -> bool {
        matches!(self, Self::Healthy)
    }
}

/// Log stream trait
pub trait LogStream: Send {
    fn try_recv(&mut self) -> Option<LogEntry>;
}

/// A log entry
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct LogEntry {
    pub timestamp: DateTime<Utc>,
    pub stream: LogStreamKind,
    pub line: String,
}

#[derive(Clone, Copy, Debug, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum LogStreamKind {
    Stdout,
    Stderr,
}
```

**Acceptance:**
- [ ] Resource trait with all methods
- [ ] Async methods use async_trait
- [ ] Default implementations where sensible
- [ ] Health status enum
- [ ] Log stream abstraction
- [ ] as_any for downcasting

**Tests:**
```rust
// Integration test with mock resource
struct MockResource {
    id: ResourceId,
    state: ResourceState,
}

#[async_trait]
impl Resource for MockResource {
    fn id(&self) -> &ResourceId { &self.id }
    fn kind(&self) -> ResourceKind { ResourceKind::Process }
    fn state(&self) -> &ResourceState { &self.state }

    async fn start(&mut self, _ctx: &Context) -> Result<()> {
        self.state = ResourceState::Running {
            pid: Some(1234),
            started_at: Utc::now(),
        };
        Ok(())
    }

    async fn stop(&mut self, _ctx: &Context) -> Result<()> {
        self.state = ResourceState::Stopped {
            exit_code: Some(0),
            started_at: Utc::now(),
            stopped_at: Utc::now(),
        };
        Ok(())
    }

    fn as_any(&self) -> &dyn Any { self }
    fn as_any_mut(&mut self) -> &mut dyn Any { self }
}

#[tokio::test]
async fn mock_resource_lifecycle() {
    let mut resource = MockResource {
        id: ResourceId::new("test"),
        state: ResourceState::Pending,
    };
    let ctx = Context::default();

    resource.start(&ctx).await.unwrap();
    assert!(resource.state().is_running());

    resource.stop(&ctx).await.unwrap();
    assert!(resource.state().is_stopped());
}
```
