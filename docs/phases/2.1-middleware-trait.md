### 2.1.1 Operation Types (2h)

**File:** `crates/dtx-core/src/middleware/operation.rs`

```rust
use serde::{Deserialize, Serialize};
use crate::resource::ResourceId;

/// Operations that can be performed on resources
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(tag = "op", rename_all = "snake_case")]
pub enum Operation {
    // Lifecycle operations
    Start { id: ResourceId },
    Stop { id: ResourceId },
    Restart { id: ResourceId },
    Kill { id: ResourceId },

    // Query operations
    Status { id: ResourceId },
    Health { id: ResourceId },
    Logs { id: ResourceId, follow: bool },

    // Batch operations
    StartAll,
    StopAll,

    // Configuration
    Configure { id: ResourceId, config: serde_json::Value },

    // Custom operation
    Custom { name: String, params: serde_json::Value },
}

impl Operation {
    pub fn resource_id(&self) -> Option<&ResourceId> {
        match self {
            Self::Start { id }
            | Self::Stop { id }
            | Self::Restart { id }
            | Self::Kill { id }
            | Self::Status { id }
            | Self::Health { id }
            | Self::Logs { id, .. }
            | Self::Configure { id, .. } => Some(id),
            _ => None,
        }
    }

    pub fn name(&self) -> &'static str {
        match self {
            Self::Start { .. } => "start",
            Self::Stop { .. } => "stop",
            Self::Restart { .. } => "restart",
            Self::Kill { .. } => "kill",
            Self::Status { .. } => "status",
            Self::Health { .. } => "health",
            Self::Logs { .. } => "logs",
            Self::StartAll => "start_all",
            Self::StopAll => "stop_all",
            Self::Configure { .. } => "configure",
            Self::Custom { name, .. } => "custom",
        }
    }

    pub fn is_mutating(&self) -> bool {
        matches!(
            self,
            Self::Start { .. }
                | Self::Stop { .. }
                | Self::Restart { .. }
                | Self::Kill { .. }
                | Self::StartAll
                | Self::StopAll
                | Self::Configure { .. }
        )
    }
}

/// Response from an operation
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "snake_case")]
pub enum Response {
    /// Operation completed successfully
    Ok,

    /// Operation completed with data
    Data { data: serde_json::Value },

    /// Resource status
    Status {
        id: ResourceId,
        state: String,
        pid: Option<u32>,
        healthy: Option<bool>,
    },

    /// Stream of events (for logs)
    Stream { stream_id: String },

    /// Error response
    Error { message: String },
}

impl Response {
    pub fn ok() -> Self {
        Self::Ok
    }

    pub fn data(data: impl Serialize) -> Self {
        Self::Data {
            data: serde_json::to_value(data).unwrap_or(serde_json::Value::Null),
        }
    }

    pub fn error(message: impl Into<String>) -> Self {
        Self::Error {
            message: message.into(),
        }
    }
}
```

### 2.1.2 Middleware Trait (3h)

**File:** `crates/dtx-core/src/middleware/trait.rs`

```rust
use async_trait::async_trait;
use std::sync::Arc;

use super::{Operation, Response};
use crate::context::Context;
use crate::error::Result;

/// Middleware processes operations before/after core execution
#[async_trait]
pub trait Middleware: Send + Sync {
    /// Unique name for this middleware
    fn name(&self) -> &'static str;

    /// Process an operation
    ///
    /// Call `next.run(op, ctx).await` to continue the chain.
    /// Can modify operation, context, or response.
    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response>;
}

/// The next middleware in the chain
pub struct Next<'a> {
    middleware: &'a [Arc<dyn Middleware>],
    handler: &'a dyn Handler,
}

impl<'a> Next<'a> {
    pub(crate) fn new(middleware: &'a [Arc<dyn Middleware>], handler: &'a dyn Handler) -> Self {
        Self { middleware, handler }
    }

    /// Execute the next middleware or the final handler
    pub async fn run(self, op: Operation, ctx: Context) -> Result<Response> {
        if let Some((first, rest)) = self.middleware.split_first() {
            let next = Next::new(rest, self.handler);
            first.handle(op, ctx, next).await
        } else {
            self.handler.handle(op, ctx).await
        }
    }
}

/// The final handler that processes operations
#[async_trait]
pub trait Handler: Send + Sync {
    async fn handle(&self, op: Operation, ctx: Context) -> Result<Response>;
}

/// A middleware that does nothing (passthrough)
pub struct PassthroughMiddleware;

#[async_trait]
impl Middleware for PassthroughMiddleware {
    fn name(&self) -> &'static str {
        "passthrough"
    }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        next.run(op, ctx).await
    }
}
```

**Acceptance:**
- [ ] Middleware trait with async handle
- [ ] Next type for chain execution
- [ ] Handler trait for final execution
- [ ] Passthrough middleware example

### 2.1.3 MiddlewareStack Builder (3h)

**File:** `crates/dtx-core/src/middleware/stack.rs`

```rust
use std::sync::Arc;
use super::{Middleware, Handler, Next, Operation, Response};
use crate::context::Context;
use crate::error::Result;

/// A composable stack of middleware
pub struct MiddlewareStack {
    middleware: Vec<Arc<dyn Middleware>>,
}

impl MiddlewareStack {
    pub fn new() -> Self {
        Self {
            middleware: Vec::new(),
        }
    }

    /// Add a middleware layer (first added = outermost)
    pub fn layer<M: Middleware + 'static>(mut self, middleware: M) -> Self {
        self.middleware.push(Arc::new(middleware));
        self
    }

    /// Add a middleware layer from Arc
    pub fn layer_arc(mut self, middleware: Arc<dyn Middleware>) -> Self {
        self.middleware.push(middleware);
        self
    }

    /// Build the stack with a handler
    pub fn build<H: Handler + 'static>(self, handler: H) -> MiddlewareChain {
        MiddlewareChain {
            middleware: self.middleware,
            handler: Arc::new(handler),
        }
    }

    /// Get the middleware list
    pub fn middleware(&self) -> &[Arc<dyn Middleware>] {
        &self.middleware
    }

    /// Number of middleware layers
    pub fn len(&self) -> usize {
        self.middleware.len()
    }

    pub fn is_empty(&self) -> bool {
        self.middleware.is_empty()
    }
}

impl Default for MiddlewareStack {
    fn default() -> Self {
        Self::new()
    }
}

/// A complete middleware chain with handler
pub struct MiddlewareChain {
    middleware: Vec<Arc<dyn Middleware>>,
    handler: Arc<dyn Handler>,
}

impl MiddlewareChain {
    /// Execute an operation through the middleware chain
    pub async fn execute(&self, op: Operation, ctx: Context) -> Result<Response> {
        let next = Next::new(&self.middleware, self.handler.as_ref());
        next.run(op, ctx).await
    }
}

/// Builder pattern for creating middleware stacks
pub struct MiddlewareStackBuilder {
    stack: MiddlewareStack,
}

impl MiddlewareStackBuilder {
    pub fn new() -> Self {
        Self {
            stack: MiddlewareStack::new(),
        }
    }

    pub fn layer<M: Middleware + 'static>(mut self, middleware: M) -> Self {
        self.stack = self.stack.layer(middleware);
        self
    }

    pub fn layer_if<M: Middleware + 'static>(self, condition: bool, middleware: M) -> Self {
        if condition {
            self.layer(middleware)
        } else {
            self
        }
    }

    pub fn layer_option<M: Middleware + 'static>(self, middleware: Option<M>) -> Self {
        match middleware {
            Some(m) => self.layer(m),
            None => self,
        }
    }

    pub fn build<H: Handler + 'static>(self, handler: H) -> MiddlewareChain {
        self.stack.build(handler)
    }
}

impl Default for MiddlewareStackBuilder {
    fn default() -> Self {
        Self::new()
    }
}
```

**Acceptance:**
- [ ] Builder pattern
- [ ] Conditional layer addition
- [ ] Execute method
- [ ] Default implementation

async fn middleware_stack_builder() {

#[tokio::test]
async fn middleware_order() {
    use std::sync::atomic::{AtomicUsize, Ordering};

    static COUNTER: AtomicUsize = AtomicUsize::new(0);

    struct OrderedMiddleware(usize);

    #[async_trait]
    impl Middleware for OrderedMiddleware {
        fn name(&self) -> &'static str { "ordered" }

        async fn handle(&self, op: Operation, ctx: Context, next: Next<'_>) -> Result<Response> {
            let before = COUNTER.fetch_add(1, Ordering::SeqCst);
            assert_eq!(before, self.0);
            next.run(op, ctx).await
        }
    }

    COUNTER.store(0, Ordering::SeqCst);

    let chain = MiddlewareStackBuilder::new()
        .layer(OrderedMiddleware(0))
        .layer(OrderedMiddleware(1))
        .layer(OrderedMiddleware(2))
        .build(TestHandler);

    chain.execute(Operation::StartAll, Context::new()).await.unwrap();
}
