## Task 4.4.1: Log Streaming Implementation

### Current State

Process stdout/stderr are piped but never read, leaving TUI logs panel empty.

### Implementation

**Add async readers to ProcessResource:**

```rust
// crates/dtx-process/src/process.rs

use tokio::io::{AsyncBufReadExt, BufReader};
use tokio::process::{ChildStdout, ChildStderr};

pub struct ProcessResource {
    // ... existing fields ...
    stdout_reader: Option<BufReader<ChildStdout>>,
    stderr_reader: Option<BufReader<ChildStderr>>,
}

impl ProcessResource {
    /// Capture stdout/stderr during spawn
    async fn spawn_process(&mut self, env: HashMap<String, String>) -> Result<()> {
        let mut cmd = Command::new(&self.config.shell);
        cmd.args(["-c", &self.config.command])
            .envs(env)
            .stdin(Stdio::null())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .kill_on_drop(true);

        let mut child = cmd.spawn()?;

        // Capture readers before storing child
        self.stdout_reader = child.stdout.take().map(BufReader::new);
        self.stderr_reader = child.stderr.take().map(BufReader::new);
        self.child = Some(child);

        Ok(())
    }

    /// Non-blocking poll for log output
    pub fn poll(&mut self) -> Option<i32> {
        // Poll stdout
        self.poll_stream(LogStreamKind::Stdout);

        // Poll stderr
        self.poll_stream(LogStreamKind::Stderr);

        // Check process exit status
        if let Some(ref mut child) = self.child {
            match child.try_wait() {
                Ok(Some(status)) => {
                    // Drain remaining output
                    self.drain_streams();
                    return status.code();
                }
                Ok(None) => {}
                Err(e) => {
                    tracing::error!("Failed to poll process: {}", e);
                }
            }
        }
        None
    }

    fn poll_stream(&mut self, kind: LogStreamKind) {
        let reader = match kind {
            LogStreamKind::Stdout => &mut self.stdout_reader,
            LogStreamKind::Stderr => &mut self.stderr_reader,
        };

        if let Some(ref mut reader) = reader {
            let mut line = String::new();
            // Non-blocking: read available lines
            while let Ok(n) = reader.try_read_line(&mut line) {
                if n == 0 { break; }
                self.add_log(LogEntry {
                    timestamp: chrono::Utc::now(),
                    stream: kind,
                    line: line.trim_end().to_string(),
                });
                line.clear();
            }
        }
    }

    fn drain_streams(&mut self) {
        // Read any remaining buffered output after process exits
        self.poll_stream(LogStreamKind::Stdout);
        self.poll_stream(LogStreamKind::Stderr);
    }
}
```

## Task 4.4.2: Restart Policy Enforcement

### Current State

Restart counter resets on every `start()` call, causing infinite restart loops.

### Implementation

**Add grace period before counter reset:**

```rust
// crates/dtx-process/src/config.rs

#[derive(Debug, Clone)]
pub struct RestartPolicy {
    pub condition: RestartCondition,
    pub max_retries: u32,
    pub backoff: BackoffConfig,
    /// Duration of healthy operation before resetting restart counter
    pub grace_period: Duration,
}

impl Default for RestartPolicy {
    fn default() -> Self {
        Self {
            condition: RestartCondition::OnFailure,
            max_retries: 3,
            backoff: BackoffConfig::default(),
            grace_period: Duration::from_secs(30),
        }
    }
}
```

```rust
// crates/dtx-process/src/process.rs

pub struct ProcessResource {
    // ... existing fields ...
    healthy_since: Option<Instant>,
}

impl ProcessResource {
    pub fn poll(&mut self) -> Option<i32> {
        // Track healthy operation time
        if self.state.is_running() {
            if self.healthy_since.is_none() {
                self.healthy_since = Some(Instant::now());
            }

            // Reset counter only after grace period
            if let Some(since) = self.healthy_since {
                if since.elapsed() >= self.config.restart_policy.grace_period {
                    if self.restart_state.attempts > 0 {
                        tracing::debug!(
                            service = %self.config.name,
                            "Resetting restart counter after grace period"
                        );
                        self.restart_state.reset();
                    }
                    self.healthy_since = None;
                }
            }
        }

        // ... existing poll logic ...
    }
}

#[async_trait]
impl Resource for ProcessResource {
    async fn start(&mut self, _ctx: &Context) -> ResourceResult<()> {
        // ... spawn process ...

        // DO NOT reset restart_state here
        // Counter reset happens in poll() after grace period
        self.healthy_since = None;

        Ok(())
    }
}
```

## Task 4.4.3: Service Independence

### Goal

Independent services should continue running when unrelated services fail.

### Implementation

**Add dependency-aware failure handling:**

```rust
// crates/dtx-process/src/orchestrator.rs

impl ResourceOrchestrator {
    /// Handle service failure - only stop truly dependent services
    async fn handle_service_failure(&mut self, failed_name: &str) -> Result<()> {
        let dependents = self.get_dependent_services(failed_name);

        for name in dependents {
            tracing::info!(
                service = %name,
                reason = %format!("dependency '{}' failed", failed_name),
                "Stopping dependent service"
            );

            self.event_bus.publish(LifecycleEvent::DependencyFailed {
                service: name.clone(),
                dependency: failed_name.to_string(),
            });

            if let Some(process) = self.processes.get_mut(&name) {
                process.write().await.stop(&self.context).await?;
            }
        }

        Ok(())
    }

    /// Get services that directly depend on the given service
    fn get_dependent_services(&self, service_name: &str) -> Vec<String> {
        self.processes
            .iter()
            .filter(|(_, process)| {
                process.read().blocking_read()
                    .config
                    .depends_on
                    .contains(&service_name.to_string())
            })
            .map(|(name, _)| name.clone())
            .collect()
    }
}
```

## Task 4.4.4: Codebase Inference

### Goal

Automatically detect project type and suggest Nix packages during `dtx init`.

### Implementation

**Add inference module:**

```rust
// crates/dtx-core/src/translation/inference.rs

use std::path::Path;

/// Detected project characteristics
#[derive(Debug, Clone)]
pub struct ProjectInference {
    pub project_type: ProjectType,
    pub detected_packages: Vec<DetectedPackage>,
    pub suggested_services: Vec<SuggestedService>,
}

#[derive(Debug, Clone)]
pub struct DetectedPackage {
    pub name: String,
    pub nixpkg: String,
    pub confidence: Confidence,
    pub source: String,  // e.g., "package.json", "Cargo.toml"
}

#[derive(Debug, Clone, Copy)]
pub enum Confidence {
    High,    // Exact match found
    Medium,  // Inferred from ecosystem
    Low,     // Heuristic guess
}

#[derive(Debug, Clone)]
pub enum ProjectType {
    Rust,
    Node,
    Python,
    Go,
    Mixed(Vec<ProjectType>),
    Unknown,
}

pub struct CodebaseInferrer {
    mappings: NixMappings,
}

impl CodebaseInferrer {
    pub fn new() -> Self {
        Self {
            mappings: NixMappings::load_default(),
        }
    }

    /// Analyze a directory and infer project configuration
    pub async fn infer(&self, project_path: &Path) -> Result<ProjectInference> {
        let mut packages = Vec::new();
        let mut project_types = Vec::new();

        // Check for Cargo.toml (Rust)
        if project_path.join("Cargo.toml").exists() {
            project_types.push(ProjectType::Rust);
            packages.push(DetectedPackage {
                name: "rustc".to_string(),
                nixpkg: "rustc".to_string(),
                confidence: Confidence::High,
                source: "Cargo.toml".to_string(),
            });
            packages.push(DetectedPackage {
                name: "cargo".to_string(),
                nixpkg: "cargo".to_string(),
                confidence: Confidence::High,
                source: "Cargo.toml".to_string(),
            });
        }

        // Check for package.json (Node)
        if let Ok(pkg_json) = self.read_package_json(project_path).await {
            project_types.push(ProjectType::Node);
            packages.extend(self.infer_from_package_json(&pkg_json));
        }

        // Check for pyproject.toml / requirements.txt (Python)
        if project_path.join("pyproject.toml").exists()
            || project_path.join("requirements.txt").exists()
        {
            project_types.push(ProjectType::Python);
            packages.push(DetectedPackage {
                name: "python".to_string(),
                nixpkg: "python3".to_string(),
                confidence: Confidence::High,
                source: "pyproject.toml".to_string(),
            });
        }

        // Check for go.mod (Go)
        if project_path.join("go.mod").exists() {
            project_types.push(ProjectType::Go);
            packages.push(DetectedPackage {
                name: "go".to_string(),
                nixpkg: "go".to_string(),
                confidence: Confidence::High,
                source: "go.mod".to_string(),
            });
        }

        let project_type = match project_types.len() {
            0 => ProjectType::Unknown,
            1 => project_types.remove(0),
            _ => ProjectType::Mixed(project_types),
        };

        Ok(ProjectInference {
            project_type,
            detected_packages: packages,
            suggested_services: self.suggest_services(&project_type),
        })
    }

    fn infer_from_package_json(&self, pkg: &PackageJson) -> Vec<DetectedPackage> {
        let mut packages = vec![
            DetectedPackage {
                name: "node".to_string(),
                nixpkg: "nodejs".to_string(),
                confidence: Confidence::High,
                source: "package.json".to_string(),
            },
        ];

        // Check for common dependencies
        for (dep, _) in pkg.dependencies.iter().chain(pkg.dev_dependencies.iter()) {
            if let Some(nixpkg) = self.mappings.npm_to_nix(dep) {
                packages.push(DetectedPackage {
                    name: dep.clone(),
                    nixpkg,
                    confidence: Confidence::Medium,
                    source: "package.json".to_string(),
                });
            }
        }

        packages
    }

    fn suggest_services(&self, project_type: &ProjectType) -> Vec<SuggestedService> {
        match project_type {
            ProjectType::Rust => vec![
                SuggestedService {
                    name: "app".to_string(),
                    command: "cargo run".to_string(),
                },
            ],
            ProjectType::Node => vec![
                SuggestedService {
                    name: "dev".to_string(),
                    command: "npm run dev".to_string(),
                },
            ],
            // ... other types
            _ => vec![],
        }
    }
}
```

```rust
// crates/dtx/src/cmd/init.rs

#[derive(Parser)]
pub struct InitArgs {
    /// Project name
    name: String,

    /// Auto-detect project type and packages
    #[arg(long)]
    detect: bool,

    /// Skip confirmation prompts (accept all detected packages)
    #[arg(short = 'y', long)]
    yes: bool,
}

pub async fn run(args: InitArgs) -> Result<()> {
    let project_path = std::env::current_dir()?;

    if args.detect {
        let inferrer = CodebaseInferrer::new();
        let inference = inferrer.infer(&project_path).await?;

        println!("Detected project type: {:?}", inference.project_type);
        println!("\nSuggested packages:");

        for pkg in &inference.detected_packages {
            println!(
                "  {} → {} ({:?}, from {})",
                pkg.name, pkg.nixpkg, pkg.confidence, pkg.source
            );

            // ... confirmation logic
        }

        // Add detected packages to project
        for pkg in inference.detected_packages {
            // ... add package logic
        }
    }

    // ... rest of init logic
}
```
