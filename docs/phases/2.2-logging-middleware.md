### 2.2.1 LoggingMiddleware (3h)

**File:** `crates/dtx-middleware/src/logging.rs`

```rust
use async_trait::async_trait;
use std::time::Instant;
use tracing::{info, warn, error, instrument, Span};

use dtx_core::{
    Context, Result,
    middleware::{Middleware, Next, Operation, Response},
};

/// Middleware that logs all operations
pub struct LoggingMiddleware {
    /// Log level for successful operations
    success_level: LogLevel,
    /// Log level for failed operations
    error_level: LogLevel,
    /// Include timing information
    include_timing: bool,
    /// Include context metadata
    include_metadata: bool,
}

#[derive(Clone, Copy, Debug, Default)]
pub enum LogLevel {
    Trace,
    Debug,
    #[default]
    Info,
    Warn,
    Error,
}

impl LoggingMiddleware {
    pub fn new() -> Self {
        Self {
            success_level: LogLevel::Info,
            error_level: LogLevel::Error,
            include_timing: true,
            include_metadata: true,
        }
    }

    pub fn success_level(mut self, level: LogLevel) -> Self {
        self.success_level = level;
        self
    }

    pub fn error_level(mut self, level: LogLevel) -> Self {
        self.error_level = level;
        self
    }

    pub fn with_timing(mut self, enabled: bool) -> Self {
        self.include_timing = enabled;
        self
    }

    pub fn with_metadata(mut self, enabled: bool) -> Self {
        self.include_metadata = enabled;
        self
    }

    fn log_start(&self, op: &Operation, ctx: &Context) {
        let span = tracing::info_span!(
            "operation",
            op = %op.name(),
            request_id = %ctx.request_id,
            resource_id = ?op.resource_id().map(|id| id.as_str()),
        );
        let _enter = span.enter();

        info!(
            operation = %op.name(),
            resource = ?op.resource_id().map(|id| id.as_str()),
            mutating = %op.is_mutating(),
            "Starting operation"
        );
    }

    fn log_success(&self, op: &Operation, ctx: &Context, elapsed: std::time::Duration) {
        match self.success_level {
            LogLevel::Trace => tracing::trace!(
                operation = %op.name(),
                elapsed_ms = %elapsed.as_millis(),
                request_id = %ctx.request_id,
                "Operation completed"
            ),
            LogLevel::Debug => tracing::debug!(
                operation = %op.name(),
                elapsed_ms = %elapsed.as_millis(),
                request_id = %ctx.request_id,
                "Operation completed"
            ),
            LogLevel::Info => info!(
                operation = %op.name(),
                elapsed_ms = %elapsed.as_millis(),
                request_id = %ctx.request_id,
                "Operation completed"
            ),
            LogLevel::Warn => warn!(
                operation = %op.name(),
                elapsed_ms = %elapsed.as_millis(),
                request_id = %ctx.request_id,
                "Operation completed"
            ),
            LogLevel::Error => error!(
                operation = %op.name(),
                elapsed_ms = %elapsed.as_millis(),
                request_id = %ctx.request_id,
                "Operation completed"
            ),
        }
    }

    fn log_error(&self, op: &Operation, ctx: &Context, err: &dtx_core::Error, elapsed: std::time::Duration) {
        error!(
            operation = %op.name(),
            error = %err,
            elapsed_ms = %elapsed.as_millis(),
            request_id = %ctx.request_id,
            retryable = %err.is_retryable(),
            "Operation failed"
        );
    }
}

impl Default for LoggingMiddleware {
    fn default() -> Self {
        Self::new()
    }
}

#[async_trait]
impl Middleware for LoggingMiddleware {
    fn name(&self) -> &'static str {
        "logging"
    }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        let start = Instant::now();

        self.log_start(&op, &ctx);

        let result = next.run(op.clone(), ctx.clone()).await;
        let elapsed = start.elapsed();

        match &result {
            Ok(_) => self.log_success(&op, &ctx, elapsed),
            Err(err) => self.log_error(&op, &ctx, err, elapsed),
        }

        result
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use dtx_core::middleware::{Handler, MiddlewareStackBuilder};

    struct TestHandler;

    #[async_trait]
    impl Handler for TestHandler {
        async fn handle(&self, _op: Operation, _ctx: Context) -> Result<Response> {
            Ok(Response::ok())
        }
    }

    #[tokio::test]
    async fn logging_middleware_success() {
        let chain = MiddlewareStackBuilder::new()
            .layer(LoggingMiddleware::new())
            .build(TestHandler);

        let result = chain.execute(Operation::StartAll, Context::new()).await;
        assert!(result.is_ok());
    }
}
```
