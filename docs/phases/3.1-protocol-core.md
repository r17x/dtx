# Phase 3.1: Protocol Core

> MCP-compatible JSON-RPC protocol and configuration schema.

---

## Goal

Define the core protocol types for communication between clients and dtx, and establish the configuration schema for `.dtx/config.yaml`.

---

## User Value

```
- AI agents can control dtx
- IDEs can integrate dtx
- Remote control possible
- Consistent API across transports
- Configuration hierarchy like git (global + project)
- Seamless config without manual YAML writing
```

---

## Scope

```
IN SCOPE:
  - JSON-RPC message types
  - Request/Response envelopes
  - Method definitions
  - Error codes
  - .dtx/config.yaml schema
  - Global vs Project configuration hierarchy
  - Config loading and merging

OUT OF SCOPE:
  - Transport layer (Phase 3.2)
  - MCP resource/tool schemas (Phase 3.3)
  - Configuration import from external formats (Phase 6.4)
```

```rust
use serde::{Deserialize, Serialize};
use serde_json::Value;

/// JSON-RPC version
pub const JSONRPC_VERSION: &str = "2.0";

/// JSON-RPC Request
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct Request {
    pub jsonrpc: String,
    pub method: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub params: Option<Value>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub id: Option<RequestId>,
}

impl Request {
    pub fn new(method: impl Into<String>, params: Option<Value>) -> Self {
        Self {
            jsonrpc: JSONRPC_VERSION.to_string(),
            method: method.into(),
            params,
            id: Some(RequestId::Number(rand::random())),
        }
    }

    pub fn notification(method: impl Into<String>, params: Option<Value>) -> Self {
        Self {
            jsonrpc: JSONRPC_VERSION.to_string(),
            method: method.into(),
            params,
            id: None,
        }
    }

    pub fn is_notification(&self) -> bool {
        self.id.is_none()
    }
}

/// JSON-RPC Response
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct Response {
    pub jsonrpc: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub result: Option<Value>,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub error: Option<ErrorObject>,
    pub id: Option<RequestId>,
}

impl Response {
    pub fn success(id: Option<RequestId>, result: Value) -> Self {
        Self {
            jsonrpc: JSONRPC_VERSION.to_string(),
            result: Some(result),
            error: None,
            id,
        }
    }

    pub fn error(id: Option<RequestId>, error: ErrorObject) -> Self {
        Self {
            jsonrpc: JSONRPC_VERSION.to_string(),
            result: None,
            error: Some(error),
            id,
        }
    }

    pub fn is_error(&self) -> bool {
        self.error.is_some()
    }
}

/// Request ID (can be string or number)
#[derive(Clone, Debug, Serialize, Deserialize, Eq, PartialEq, Hash)]
#[serde(untagged)]
pub enum RequestId {
    Number(i64),
    String(String),
}

/// JSON-RPC Error
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ErrorObject {
    pub code: i32,
    pub message: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub data: Option<Value>,
}

impl ErrorObject {
    pub fn new(code: i32, message: impl Into<String>) -> Self {
        Self {
            code,
            message: message.into(),
            data: None,
        }
    }

    pub fn with_data(mut self, data: impl Serialize) -> Self {
        self.data = Some(serde_json::to_value(data).unwrap_or(Value::Null));
        self
    }
}

/// Standard JSON-RPC error codes
pub mod error_codes {
    pub const PARSE_ERROR: i32 = -32700;
    pub const INVALID_REQUEST: i32 = -32600;
    pub const METHOD_NOT_FOUND: i32 = -32601;
    pub const INVALID_PARAMS: i32 = -32602;
    pub const INTERNAL_ERROR: i32 = -32603;

    // dtx custom error codes (-32000 to -32099)
    pub const RESOURCE_NOT_FOUND: i32 = -32000;
    pub const RESOURCE_EXISTS: i32 = -32001;
    pub const INVALID_STATE: i32 = -32002;
    pub const OPERATION_FAILED: i32 = -32003;
    pub const TIMEOUT: i32 = -32004;
    pub const CANCELLED: i32 = -32005;
}
```

### 3.1.3 Protocol Handler (3h)

**File:** `crates/dtx-protocol/src/handler.rs`

```rust
use async_trait::async_trait;
use serde_json::Value;

use crate::jsonrpc::{Request, Response, ErrorObject, error_codes};
use crate::methods::*;

/// Trait for handling protocol requests
#[async_trait]
pub trait ProtocolHandler: Send + Sync {
    // Lifecycle
    async fn resource_start(&self, params: ResourceParams) -> Result<Value, ErrorObject>;
    async fn resource_stop(&self, params: ResourceParams) -> Result<Value, ErrorObject>;
    async fn resource_restart(&self, params: ResourceParams) -> Result<Value, ErrorObject>;
    async fn resource_kill(&self, params: ResourceParams) -> Result<Value, ErrorObject>;

    // Query
    async fn resource_status(&self, params: ResourceParams) -> Result<ResourceStatusResult, ErrorObject>;
    async fn resource_list(&self) -> Result<ResourceListResult, ErrorObject>;
    async fn resource_logs(&self, params: LogsParams) -> Result<Vec<LogEntry>, ErrorObject>;

    // Batch
    async fn start_all(&self) -> Result<Value, ErrorObject>;
    async fn stop_all(&self) -> Result<Value, ErrorObject>;

    // Events
    async fn events_subscribe(&self, params: SubscribeParams) -> Result<SubscribeResult, ErrorObject>;
    async fn events_unsubscribe(&self, subscription_id: String) -> Result<Value, ErrorObject>;
}

/// Dispatch a request to the appropriate handler method
pub async fn dispatch<H: ProtocolHandler>(
    handler: &H,
    request: Request,
) -> Response {
    let id = request.id.clone();

    let result = match request.method.as_str() {
        methods::RESOURCE_START => {
            let params: ResourceParams = parse_params(&request)?;
            handler.resource_start(params).await.map(|v| v)
        }
        methods::RESOURCE_STOP => {
            let params: ResourceParams = parse_params(&request)?;
            handler.resource_stop(params).await.map(|v| v)
        }
        methods::RESOURCE_RESTART => {
            let params: ResourceParams = parse_params(&request)?;
            handler.resource_restart(params).await.map(|v| v)
        }
        methods::RESOURCE_STATUS => {
            let params: ResourceParams = parse_params(&request)?;
            handler.resource_status(params).await
                .map(|r| serde_json::to_value(r).unwrap())
        }
        methods::RESOURCE_LIST => {
            handler.resource_list().await
                .map(|r| serde_json::to_value(r).unwrap())
        }
        methods::START_ALL => {
            handler.start_all().await
        }
        methods::STOP_ALL => {
            handler.stop_all().await
        }
        methods::EVENTS_SUBSCRIBE => {
            let params: SubscribeParams = parse_params_or_default(&request);
            handler.events_subscribe(params).await
                .map(|r| serde_json::to_value(r).unwrap())
        }
        _ => {
            Err(ErrorObject::new(
                error_codes::METHOD_NOT_FOUND,
                format!("Method not found: {}", request.method),
            ))
        }
    };

    match result {
        Ok(value) => Response::success(id, value),
        Err(error) => Response::error(id, error),
    }
}

fn parse_params<T: serde::de::DeserializeOwned>(request: &Request) -> Result<T, ErrorObject> {
    let params = request.params.as_ref()
        .ok_or_else(|| ErrorObject::new(error_codes::INVALID_PARAMS, "Missing params"))?;

    serde_json::from_value(params.clone())
        .map_err(|e| ErrorObject::new(error_codes::INVALID_PARAMS, e.to_string()))
}

fn parse_params_or_default<T: serde::de::DeserializeOwned + Default>(request: &Request) -> T {
    request.params.as_ref()
        .and_then(|p| serde_json::from_value(p.clone()).ok())
        .unwrap_or_default()
}
```

### 3.1.6 Configuration Hierarchy (2h)

**File:** `crates/dtx-core/src/config/loader.rs`

```rust
//! Configuration loading with hierarchical merging.
//!
//! Precedence (highest to lowest):
//! 1. Project: .dtx/config.yaml
//! 2. Global: ~/.config/dtx/config.yaml
//! 3. System: /etc/dtx/config.yaml

use std::path::{Path, PathBuf};

use super::schema::{DtxConfig, SettingsConfig, ConfigError};

/// Configuration levels
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum ConfigLevel {
    System,
    Global,
    Project,
}

impl ConfigLevel {
    /// Get default path for this level
    pub fn default_path(&self) -> Option<PathBuf> {
        match self {
            Self::System => Some(PathBuf::from("/etc/dtx/config.yaml")),
            Self::Global => dirs::config_dir().map(|p| p.join("dtx/config.yaml")),
            Self::Project => None, // Determined by project discovery
        }
    }
}

/// Configuration loader
pub struct ConfigLoader {
    system_path: Option<PathBuf>,
    global_path: Option<PathBuf>,
    project_path: Option<PathBuf>,
}

impl ConfigLoader {
    /// Create loader with default paths
    pub fn new() -> Self {
        Self {
            system_path: ConfigLevel::System.default_path(),
            global_path: ConfigLevel::Global.default_path(),
            project_path: None,
        }
    }

    /// Set project config path
    pub fn with_project_path(mut self, path: impl Into<PathBuf>) -> Self {
        self.project_path = Some(path.into());
        self
    }

    /// Discover project path by walking up directory tree
    pub fn discover_project(&mut self, start: &Path) -> Option<PathBuf> {
        let mut current = start.to_path_buf();
        loop {
            let config_path = current.join(".dtx/config.yaml");
            if config_path.exists() {
                self.project_path = Some(config_path.clone());
                return Some(config_path);
            }
            if !current.pop() {
                break;
            }
        }
        None
    }

    /// Load merged configuration
    pub fn load(&self) -> Result<DtxConfig, ConfigError> {
        let mut config = DtxConfig::default();

        // Load system config
        if let Some(path) = &self.system_path {
            if path.exists() {
                let system = DtxConfig::load(path)?;
                config = merge_config(config, system);
            }
        }

        // Load global config
        if let Some(path) = &self.global_path {
            if path.exists() {
                let global = DtxConfig::load(path)?;
                config = merge_config(config, global);
            }
        }

        // Load project config
        if let Some(path) = &self.project_path {
            if path.exists() {
                let project = DtxConfig::load(path)?;
                config = merge_config(config, project);
            }
        }

        Ok(config)
    }

    /// Load config at specific level only
    pub fn load_level(&self, level: ConfigLevel) -> Result<Option<DtxConfig>, ConfigError> {
        let path = match level {
            ConfigLevel::System => self.system_path.as_ref(),
            ConfigLevel::Global => self.global_path.as_ref(),
            ConfigLevel::Project => self.project_path.as_ref(),
        };

        match path {
            Some(p) if p.exists() => Ok(Some(DtxConfig::load(p)?)),
            _ => Ok(None),
        }
    }

    /// Save config at specific level
    pub fn save_level(&self, config: &DtxConfig, level: ConfigLevel) -> Result<(), ConfigError> {
        let path = match level {
            ConfigLevel::System => self.system_path.as_ref(),
            ConfigLevel::Global => self.global_path.as_ref(),
            ConfigLevel::Project => self.project_path.as_ref(),
        };

        match path {
            Some(p) => {
                // Ensure parent directory exists
                if let Some(parent) = p.parent() {
                    std::fs::create_dir_all(parent)
                        .map_err(|e| ConfigError::Io(e.to_string()))?;
                }
                config.save(p)
            }
            None => Err(ConfigError::Validation(format!(
                "No path configured for {:?} level",
                level
            ))),
        }
    }
}

/// Merge two configs (later values override earlier)
fn merge_config(base: DtxConfig, overlay: DtxConfig) -> DtxConfig {
    DtxConfig {
        version: if overlay.version.is_empty() {
            base.version
        } else {
            overlay.version
        },
        project: if overlay.project.name.is_empty() {
            base.project
        } else {
            overlay.project
        },
        settings: merge_settings(base.settings, overlay.settings),
        resources: {
            let mut merged = base.resources;
            merged.extend(overlay.resources);
            merged
        },
    }
}

fn merge_settings(base: SettingsConfig, overlay: SettingsConfig) -> SettingsConfig {
    SettingsConfig {
        log_level: overlay.log_level,
        health_check_interval: overlay.health_check_interval,
        shutdown_timeout: overlay.shutdown_timeout,
        log_dir: overlay.log_dir.or(base.log_dir),
    }
}

impl Default for ConfigLoader {
    fn default() -> Self {
        Self::new()
    }
}
```
