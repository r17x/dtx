## Goal

Redesign EventBus for the new Resource abstraction with improved filtering, replay, and observability.


```rust
use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

use crate::resource::{ResourceId, ResourceKind, HealthStatus, LogStreamKind};

/// Events emitted during resource lifecycle
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(tag = "event", rename_all = "snake_case")]
pub enum LifecycleEvent {
    // State transitions
    Starting {
        id: ResourceId,
        kind: ResourceKind,
        timestamp: DateTime<Utc>,
    },

    Running {
        id: ResourceId,
        kind: ResourceKind,
        pid: Option<u32>,
        timestamp: DateTime<Utc>,
    },

    Stopping {
        id: ResourceId,
        kind: ResourceKind,
        timestamp: DateTime<Utc>,
    },

    Stopped {
        id: ResourceId,
        kind: ResourceKind,
        exit_code: Option<i32>,
        timestamp: DateTime<Utc>,
    },

    Failed {
        id: ResourceId,
        kind: ResourceKind,
        error: String,
        exit_code: Option<i32>,
        timestamp: DateTime<Utc>,
    },

    Restarting {
        id: ResourceId,
        kind: ResourceKind,
        attempt: u32,
        max_attempts: Option<u32>,
        timestamp: DateTime<Utc>,
    },

    // Health
    HealthCheckPassed {
        id: ResourceId,
        timestamp: DateTime<Utc>,
    },

    HealthCheckFailed {
        id: ResourceId,
        reason: String,
        timestamp: DateTime<Utc>,
    },

    // Logs
    Log {
        id: ResourceId,
        stream: LogStreamKind,
        line: String,
        timestamp: DateTime<Utc>,
    },

    // Dependencies
    DependencyWaiting {
        id: ResourceId,
        dependency: ResourceId,
        condition: DependencyCondition,
        timestamp: DateTime<Utc>,
    },

    DependencyResolved {
        id: ResourceId,
        dependency: ResourceId,
        timestamp: DateTime<Utc>,
    },

    // Configuration
    ConfigChanged {
        project_id: String,
        timestamp: DateTime<Utc>,
    },
}

#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum DependencyCondition {
    Started,
    Healthy,
    Completed,
}

impl LifecycleEvent {
    pub fn resource_id(&self) -> Option<&ResourceId> {
        match self {
            Self::Starting { id, .. }
            | Self::Running { id, .. }
            | Self::Stopping { id, .. }
            | Self::Stopped { id, .. }
            | Self::Failed { id, .. }
            | Self::Restarting { id, .. }
            | Self::HealthCheckPassed { id, .. }
            | Self::HealthCheckFailed { id, .. }
            | Self::Log { id, .. }
            | Self::DependencyWaiting { id, .. }
            | Self::DependencyResolved { id, .. } => Some(id),
            Self::ConfigChanged { .. } => None,
        }
    }

    pub fn timestamp(&self) -> DateTime<Utc> {
        match self {
            Self::Starting { timestamp, .. }
            | Self::Running { timestamp, .. }
            | Self::Stopping { timestamp, .. }
            | Self::Stopped { timestamp, .. }
            | Self::Failed { timestamp, .. }
            | Self::Restarting { timestamp, .. }
            | Self::HealthCheckPassed { timestamp, .. }
            | Self::HealthCheckFailed { timestamp, .. }
            | Self::Log { timestamp, .. }
            | Self::DependencyWaiting { timestamp, .. }
            | Self::DependencyResolved { timestamp, .. }
            | Self::ConfigChanged { timestamp, .. } => *timestamp,
        }
    }

    pub fn event_type(&self) -> &'static str {
        match self {
            Self::Starting { .. } => "starting",
            Self::Running { .. } => "running",
            Self::Stopping { .. } => "stopping",
            Self::Stopped { .. } => "stopped",
            Self::Failed { .. } => "failed",
            Self::Restarting { .. } => "restarting",
            Self::HealthCheckPassed { .. } => "health_check_passed",
            Self::HealthCheckFailed { .. } => "health_check_failed",
            Self::Log { .. } => "log",
            Self::DependencyWaiting { .. } => "dependency_waiting",
            Self::DependencyResolved { .. } => "dependency_resolved",
            Self::ConfigChanged { .. } => "config_changed",
        }
    }
}
```

### 1.2.2 EventFilter (2h)

**File:** `crates/dtx-core/src/events/filter.rs`

```rust
use super::LifecycleEvent;
use crate::resource::{ResourceId, ResourceKind};

/// Filter for subscribing to specific events
#[derive(Clone, Debug, Default)]
pub struct EventFilter {
    /// Filter by resource IDs (empty = all)
    pub resource_ids: Vec<ResourceId>,
    /// Filter by resource kinds (empty = all)
    pub kinds: Vec<ResourceKind>,
    /// Filter by event types (empty = all)
    pub event_types: Vec<String>,
    /// Include log events
    pub include_logs: bool,
}

impl EventFilter {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn all() -> Self {
        Self {
            include_logs: true,
            ..Default::default()
        }
    }

    pub fn resource(mut self, id: impl Into<ResourceId>) -> Self {
        self.resource_ids.push(id.into());
        self
    }

    pub fn kind(mut self, kind: ResourceKind) -> Self {
        self.kinds.push(kind);
        self
    }

    pub fn event_type(mut self, event_type: impl Into<String>) -> Self {
        self.event_types.push(event_type.into());
        self
    }

    pub fn with_logs(mut self) -> Self {
        self.include_logs = true;
        self
    }

    pub fn without_logs(mut self) -> Self {
        self.include_logs = false;
        self
    }

    pub fn matches(&self, event: &LifecycleEvent) -> bool {
        // Check log filter
        if matches!(event, LifecycleEvent::Log { .. }) && !self.include_logs {
            return false;
        }

        // Check resource ID filter
        if !self.resource_ids.is_empty() {
            if let Some(id) = event.resource_id() {
                if !self.resource_ids.contains(id) {
                    return false;
                }
            }
        }

        // Check kind filter
        if !self.kinds.is_empty() {
            if let Some(kind) = Self::event_kind(event) {
                if !self.kinds.contains(&kind) {
                    return false;
                }
            }
        }

        // Check event type filter
        if !self.event_types.is_empty() {
            if !self.event_types.iter().any(|t| t == event.event_type()) {
                return false;
            }
        }

        true
    }

    fn event_kind(event: &LifecycleEvent) -> Option<ResourceKind> {
        match event {
            LifecycleEvent::Starting { kind, .. }
            | LifecycleEvent::Running { kind, .. }
            | LifecycleEvent::Stopping { kind, .. }
            | LifecycleEvent::Stopped { kind, .. }
            | LifecycleEvent::Failed { kind, .. }
            | LifecycleEvent::Restarting { kind, .. } => Some(*kind),
            _ => None,
        }
    }
}
```

- [ ] Builder pattern for filters

### 1.2.3 EventBus Core (4h)

**File:** `crates/dtx-core/src/events/bus.rs`

```rust
use std::sync::Arc;
use tokio::sync::broadcast;
use parking_lot::RwLock;

use super::{LifecycleEvent, EventFilter};

const DEFAULT_CAPACITY: usize = 1000;
const REPLAY_BUFFER_SIZE: usize = 500;

/// Central event bus for lifecycle events
pub struct EventBus {
    sender: broadcast::Sender<LifecycleEvent>,
    replay_buffer: Arc<RwLock<ReplayBuffer>>,
    metrics: Arc<EventBusMetrics>,
}

struct ReplayBuffer {
    events: Vec<LifecycleEvent>,
    capacity: usize,
}

impl ReplayBuffer {
    fn new(capacity: usize) -> Self {
        Self {
            events: Vec::with_capacity(capacity),
            capacity,
        }
    }

    fn push(&mut self, event: LifecycleEvent) {
        if self.events.len() >= self.capacity {
            self.events.remove(0);
        }
        self.events.push(event);
    }

    fn replay(&self, filter: &EventFilter) -> Vec<LifecycleEvent> {
        self.events
            .iter()
            .filter(|e| filter.matches(e))
            .cloned()
            .collect()
    }
}

#[derive(Default)]
struct EventBusMetrics {
    published: std::sync::atomic::AtomicU64,
    dropped: std::sync::atomic::AtomicU64,
}

impl EventBus {
    pub fn new() -> Self {
        Self::with_capacity(DEFAULT_CAPACITY)
    }

    pub fn with_capacity(capacity: usize) -> Self {
        let (sender, _) = broadcast::channel(capacity);
        Self {
            sender,
            replay_buffer: Arc::new(RwLock::new(ReplayBuffer::new(REPLAY_BUFFER_SIZE))),
            metrics: Arc::new(EventBusMetrics::default()),
        }
    }

    /// Publish an event to all subscribers
    pub fn publish(&self, event: LifecycleEvent) -> usize {
        // Add to replay buffer
        self.replay_buffer.write().push(event.clone());

        // Broadcast to subscribers
        let receivers = self.sender.send(event).unwrap_or(0);

        self.metrics.published.fetch_add(1, std::sync::atomic::Ordering::Relaxed);

        receivers
    }

    /// Create a new subscriber
    pub fn subscribe(&self) -> EventSubscriber {
        EventSubscriber {
            receiver: self.sender.subscribe(),
            filter: EventFilter::all(),
        }
    }

    /// Create a subscriber with filter
    pub fn subscribe_filtered(&self, filter: EventFilter) -> EventSubscriber {
        EventSubscriber {
            receiver: self.sender.subscribe(),
            filter,
        }
    }

    /// Get recent events matching filter
    pub fn replay(&self, filter: &EventFilter) -> Vec<LifecycleEvent> {
        self.replay_buffer.read().replay(filter)
    }

    /// Get metrics
    pub fn metrics(&self) -> (u64, u64) {
        (
            self.metrics.published.load(std::sync::atomic::Ordering::Relaxed),
            self.metrics.dropped.load(std::sync::atomic::Ordering::Relaxed),
        )
    }
}

impl Default for EventBus {
    fn default() -> Self {
        Self::new()
    }
}

/// Subscriber to EventBus
pub struct EventSubscriber {
    receiver: broadcast::Receiver<LifecycleEvent>,
    filter: EventFilter,
}

impl EventSubscriber {
    /// Receive next event (blocking)
    pub async fn recv(&mut self) -> Option<LifecycleEvent> {
        loop {
            match self.receiver.recv().await {
                Ok(event) => {
                    if self.filter.matches(&event) {
                        return Some(event);
                    }
                }
                Err(broadcast::error::RecvError::Closed) => return None,
                Err(broadcast::error::RecvError::Lagged(n)) => {
                    tracing::warn!("EventSubscriber lagged by {} events", n);
                    continue;
                }
            }
        }
    }

    /// Try to receive without blocking
    pub fn try_recv(&mut self) -> Option<LifecycleEvent> {
        loop {
            match self.receiver.try_recv() {
                Ok(event) => {
                    if self.filter.matches(&event) {
                        return Some(event);
                    }
                }
                Err(_) => return None,
            }
        }
    }

    /// Update filter
    pub fn set_filter(&mut self, filter: EventFilter) {
        self.filter = filter;
    }
}
```

```rust
#[tokio::test]
async fn eventbus_publish_receive() {
    let bus = EventBus::new();
    let mut sub = bus.subscribe();

    let event = LifecycleEvent::Starting {
        id: ResourceId::new("api"),
        kind: ResourceKind::Process,
        timestamp: Utc::now(),
    };

    bus.publish(event.clone());

    let received = sub.recv().await.unwrap();
    assert_eq!(received.resource_id(), Some(&ResourceId::new("api")));
}

#[tokio::test]
async fn eventbus_filtered_subscriber() {
    let bus = EventBus::new();
    let filter = EventFilter::new().resource("api");
    let mut sub = bus.subscribe_filtered(filter);

    // This should be filtered out
    bus.publish(LifecycleEvent::Starting {
        id: ResourceId::new("db"),
        kind: ResourceKind::Process,
        timestamp: Utc::now(),
    });

    // This should pass
    bus.publish(LifecycleEvent::Starting {
        id: ResourceId::new("api"),
        kind: ResourceKind::Process,
        timestamp: Utc::now(),
    });

    let received = sub.try_recv().unwrap();
    assert_eq!(received.resource_id(), Some(&ResourceId::new("api")));
}

#[test]
fn eventbus_replay() {
    let bus = EventBus::new();

    for i in 0..10 {
        bus.publish(LifecycleEvent::Starting {
            id: ResourceId::new(format!("svc-{}", i)),
            kind: ResourceKind::Process,
            timestamp: Utc::now(),
        });
    }

    let filter = EventFilter::new().resource("svc-5");
    let replayed = bus.replay(&filter);
    assert_eq!(replayed.len(), 1);
}
```

```rust
use chrono::Utc;
use crate::resource::{ResourceId, ResourceKind, LogStreamKind};
use super::LifecycleEvent;

/// Builder helpers for common events
impl LifecycleEvent {
    pub fn starting(id: impl Into<ResourceId>, kind: ResourceKind) -> Self {
        Self::Starting {
            id: id.into(),
            kind,
            timestamp: Utc::now(),
        }
    }

    pub fn running(id: impl Into<ResourceId>, kind: ResourceKind, pid: Option<u32>) -> Self {
        Self::Running {
            id: id.into(),
            kind,
            pid,
            timestamp: Utc::now(),
        }
    }

    pub fn stopped(id: impl Into<ResourceId>, kind: ResourceKind, exit_code: Option<i32>) -> Self {
        Self::Stopped {
            id: id.into(),
            kind,
            exit_code,
            timestamp: Utc::now(),
        }
    }

    pub fn failed(
        id: impl Into<ResourceId>,
        kind: ResourceKind,
        error: impl Into<String>,
        exit_code: Option<i32>,
    ) -> Self {
        Self::Failed {
            id: id.into(),
            kind,
            error: error.into(),
            exit_code,
            timestamp: Utc::now(),
        }
    }

    pub fn log(id: impl Into<ResourceId>, stream: LogStreamKind, line: impl Into<String>) -> Self {
        Self::Log {
            id: id.into(),
            stream,
            line: line.into(),
            timestamp: Utc::now(),
        }
    }

    pub fn stdout(id: impl Into<ResourceId>, line: impl Into<String>) -> Self {
        Self::log(id, LogStreamKind::Stdout, line)
    }

    pub fn stderr(id: impl Into<ResourceId>, line: impl Into<String>) -> Self {
        Self::log(id, LogStreamKind::Stderr, line)
    }
}
```
