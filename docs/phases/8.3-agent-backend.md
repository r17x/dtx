# Phase 8.3: Agent Backend

> AI agent orchestration as a Resource backend.

---

## Goal

Implement AI agents as Resources, enabling orchestration of AI-powered workflows alongside processes, containers, and VMs with full lifecycle management, tool integration, and MCP compatibility.

---

## User Value

```
- Orchestrate AI agents alongside services
- Build agentic workflows with dependencies
- Consistent management API for AI agents
- Tool registration and execution
- Multi-model support (Claude, GPT, Ollama)
- MCP protocol compatibility
- Local and cloud agent execution
```

```rust
use futures::Stream;
use std::pin::Pin;

use crate::config::{AgentConfig, ToolConfig};

use dtx_core::resource::HealthStatus;
use dtx_core::error::Result;

/// Agent runtime information.
#[derive(Clone, Debug)]
pub struct AgentInfo {
    /// Agent identifier.
    pub id: String,
    /// Current state.
    pub state: AgentState,
    /// Model being used.
    pub model: String,
    /// Number of turns processed.
    pub turns: u64,

    pub input_tokens: u64,
    /// Tokens used (output).
    pub output_tokens: u64,

    /// Last activity.
    pub last_activity: Option<chrono::DateTime<chrono::Utc>>,
}

/// Agent state.
#[derive(Clone, Debug, PartialEq)]
pub enum AgentState {
    Idle,
    Processing,
    WaitingForTool,
    Paused,
    Stopped,
    Error(String),
}

/// Response from the agent.
#[derive(Clone, Debug)]
pub struct AgentResponse {
    /// Response content.
    pub content: String,
    /// Tool calls requested.
    pub tool_calls: Vec<ToolCall>,
    /// Stop reason.
    pub stop_reason: StopReason,
    /// Token usage.
    pub usage: TokenUsage,
    /// Latency in milliseconds.
    pub latency_ms: u64,
}

/// Stop reason for a turn.
#[derive(Clone, Debug)]
pub enum StopReason {
    EndTurn,
    ToolUse,
    MaxTokens,
    StopSequence,
    Error(String),
}

/// Token usage statistics.
#[derive(Clone, Debug, Default)]
pub struct TokenUsage {
    pub input_tokens: u64,
    pub output_tokens: u64,
    pub cache_read_tokens: u64,
    pub cache_write_tokens: u64,
}

/// Streaming event from agent.
#[derive(Clone, Debug)]
pub enum StreamEvent {
    /// Content block started.
    ContentStart { index: usize },
    /// Text delta.
    TextDelta { index: usize, text: String },
    /// Tool use started.
    ToolStart { index: usize, id: String, name: String },
    /// Tool input delta.
    ToolInputDelta { index: usize, delta: String },
    /// Content block completed.
    ContentEnd { index: usize },
    /// Message completed.
    MessageEnd { usage: TokenUsage },
    /// Error occurred.
    Error { error: String },
}

/// Abstract agent runtime interface.
#[async_trait]
pub trait AgentRuntime: Send + Sync {
    /// Runtime name.
    fn name(&self) -> &str;

    /// Check if runtime is available.
    async fn is_available(&self) -> bool;

    /// Initialize the runtime with configuration.
    async fn initialize(&mut self, config: &AgentConfig) -> Result<()>;

    /// Send a message and get a response.
    async fn send(
        &self,
        messages: &[Message],
        tools: &[ToolConfig],
    ) -> Result<AgentResponse>;

    /// Send a message and stream the response.
    async fn send_streaming(
        &self,
        messages: &[Message],
        tools: &[ToolConfig],
    ) -> Result<Pin<Box<dyn Stream<Item = StreamEvent> + Send>>>;

    /// Execute a tool call.
    async fn execute_tool(&self, tool_call: &ToolCall) -> Result<ToolResult>;

    /// Get agent info.
    fn info(&self) -> AgentInfo;

    /// Cancel ongoing operation.
    async fn cancel(&self) -> Result<()>;

    /// Get health status.
    async fn health(&self) -> Result<HealthStatus>;

    /// Count tokens in a message.
    async fn count_tokens(&self, text: &str) -> Result<u64>;

    /// Shutdown the runtime.
    async fn shutdown(&mut self) -> Result<()>;
}

/// Select the best available agent runtime.
pub async fn detect_runtime(config: &AgentConfig) -> Option<Box<dyn AgentRuntime>> {
    match config.runtime {
        crate::config::AgentRuntimeType::Auto => {
            // Try Claude first
            if let Ok(runtime) = super::claude::ClaudeRuntime::new(config).await {
                if runtime.is_available().await {
                    return Some(Box::new(runtime));
                }
            }
            // Try OpenAI
            if let Ok(runtime) = super::openai::OpenAIRuntime::new(config).await {
                if runtime.is_available().await {
                    return Some(Box::new(runtime));
                }
            }
            // Try Ollama
            if let Ok(runtime) = super::ollama::OllamaRuntime::new(config).await {
                if runtime.is_available().await {
                    return Some(Box::new(runtime));
                }
            }
            None
        }
        crate::config::AgentRuntimeType::Claude => {
            super::claude::ClaudeRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
        crate::config::AgentRuntimeType::OpenAI => {
            super::openai::OpenAIRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
        crate::config::AgentRuntimeType::Ollama => {
            super::ollama::OllamaRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
        crate::config::AgentRuntimeType::LlamaCpp => {
            super::llamacpp::LlamaCppRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
        crate::config::AgentRuntimeType::OpenAICompatible => {
            super::openai::OpenAIRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
        crate::config::AgentRuntimeType::Mcp => {
            super::mcp::McpAgentRuntime::new(config).await.ok().map(|r| Box::new(r) as _)
        }
    }
}
```

```rust
use serde::{Deserialize, Serialize};

/// A message in the conversation.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct Message {
    /// Role of the message sender.
    pub role: Role,
    /// Content of the message.
    pub content: Content,
    /// Optional name for the sender.
    #[serde(skip_serializing_if = "Option::is_none")]
    pub name: Option<String>,
}

/// Message role.
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(rename_all = "lowercase")]
pub enum Role {
    System,
    User,
    Assistant,
    Tool,
}

/// Message content.
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(untagged)]
pub enum Content {
    Text(String),
    Blocks(Vec<ContentBlock>),
}

#[serde(tag = "type", rename_all = "snake_case")]

    Text { text: String },
    Image { source: ImageSource },
    ToolUse { id: String, name: String, input: serde_json::Value },
    ToolResult { tool_use_id: String, content: String, is_error: bool },
}

/// Image source.
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "snake_case")]
pub enum ImageSource {
    Base64 { media_type: String, data: String },
    Url { url: String },
}

/// Tool call request.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ToolCall {
    /// Unique identifier for this call.
    pub id: String,
    /// Tool name.
    pub name: String,
    /// Tool arguments.
    pub input: serde_json::Value,
}

/// Tool execution result.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ToolResult {
    /// Tool call ID this result corresponds to.
    pub tool_use_id: String,
    /// Result content.
    pub content: String,
    /// Whether this is an error result.
    pub is_error: bool,
}

impl Message {
    pub fn user(text: impl Into<String>) -> Self {
        Self {
            role: Role::User,
            content: Content::Text(text.into()),
            name: None,
        }
    }

    pub fn assistant(text: impl Into<String>) -> Self {
        Self {
            role: Role::Assistant,
            content: Content::Text(text.into()),
            name: None,
        }
    }

    pub fn system(text: impl Into<String>) -> Self {
        Self {
            role: Role::System,
            content: Content::Text(text.into()),
            name: None,
        }
    }

    pub fn tool_result(tool_use_id: impl Into<String>, content: impl Into<String>, is_error: bool) -> Self {
        Self {
            role: Role::Tool,
            content: Content::Blocks(vec![ContentBlock::ToolResult {
                tool_use_id: tool_use_id.into(),
                content: content.into(),
                is_error,
            }]),
            name: None,
        }
    }
}
```

```rust
use async_trait::async_trait;
use futures::{Stream, StreamExt};
use reqwest::Client;
use std::pin::Pin;
use std::sync::atomic::{AtomicU64, Ordering};
use std::sync::Arc;
use tokio::sync::RwLock;

use super::runtime::{AgentRuntime, AgentInfo, AgentState, AgentResponse, StreamEvent, StopReason, TokenUsage};
use super::config::{AgentConfig, ToolConfig, ApiKeyConfig};
use super::message::{Message, Role, Content, ContentBlock, ToolCall, ToolResult};
use super::tool::ToolExecutor;
use dtx_core::resource::HealthStatus;
use dtx_core::error::{Error, Result};

const ANTHROPIC_API_URL: &str = "https://api.anthropic.com/v1";
const ANTHROPIC_VERSION: &str = "2023-06-01";

/// Claude runtime using Anthropic API.
pub struct ClaudeRuntime {
    client: Client,
    api_key: String,
    base_url: String,
    config: AgentConfig,
    state: Arc<RwLock<AgentState>>,
    turns: AtomicU64,
    input_tokens: AtomicU64,
    output_tokens: AtomicU64,
    started_at: Option<chrono::DateTime<chrono::Utc>>,
    tool_executor: Arc<ToolExecutor>,
}

impl ClaudeRuntime {
    pub async fn new(config: &AgentConfig) -> Result<Self> {
        let api_key = match &config.model.api_key {
            Some(ApiKeyConfig::Direct(key)) => key.clone(),
            Some(ApiKeyConfig::EnvVar { env }) => {
                std::env::var(env).map_err(|_| Error::config(format!("Missing env var: {}", env)))?
            }
            Some(ApiKeyConfig::File { file }) => {
                tokio::fs::read_to_string(file).await
                    .map_err(|e| Error::config(format!("Failed to read key file: {}", e)))?
                    .trim().to_string()
            }
            Some(ApiKeyConfig::Secret { secret }) => {
                // Would integrate with secret manager
                return Err(Error::config("Secret manager not implemented"));
            }
            None => {
                std::env::var("ANTHROPIC_API_KEY")
                    .map_err(|_| Error::config("ANTHROPIC_API_KEY not set"))?
            }
        };

        let base_url = config.model.base_url.clone()
            .unwrap_or_else(|| ANTHROPIC_API_URL.to_string());

        let client = Client::builder()
            .timeout(config.timeout)
            .build()
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(Self {
            client,
            api_key,
            base_url,
            config: config.clone(),
            state: Arc::new(RwLock::new(AgentState::Idle)),
            turns: AtomicU64::new(0),
            input_tokens: AtomicU64::new(0),
            output_tokens: AtomicU64::new(0),
            started_at: None,
            tool_executor: Arc::new(ToolExecutor::new(config.tools.clone())),
        })
    }

    fn build_tools_schema(&self, tools: &[ToolConfig]) -> Vec<serde_json::Value> {
        tools.iter().filter_map(|t| {
            match t {
                ToolConfig::Builtin { name, config } => {
                    Some(self.builtin_tool_schema(name, config))
                }
                ToolConfig::Command { name, description, args_schema, .. } => {
                    Some(serde_json::json!({
                        "name": name,
                        "description": description,
                        "input_schema": args_schema.clone().unwrap_or(serde_json::json!({
                            "type": "object",
                            "properties": {}
                        }))
                    }))
                }
                ToolConfig::Http { name, description, body_schema, .. } => {
                    Some(serde_json::json!({
                        "name": name,
                        "description": description,
                        "input_schema": body_schema.clone().unwrap_or(serde_json::json!({
                            "type": "object",
                            "properties": {}
                        }))
                    }))
                }
                ToolConfig::Custom { name, description, schema, .. } => {
                    Some(serde_json::json!({
                        "name": name,
                        "description": description,
                        "input_schema": schema.clone().unwrap_or(serde_json::json!({
                            "type": "object",
                            "properties": {}
                        }))
                    }))
                }
                _ => None,
            }
        }).collect()
    }

    fn builtin_tool_schema(&self, name: &str, _config: &serde_json::Value) -> serde_json::Value {
        match name {
            "read_file" => serde_json::json!({
                "name": "read_file",
                "description": "Read contents of a file",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "path": { "type": "string", "description": "Path to the file" }
                    },
                    "required": ["path"]
                }
            }),
            "write_file" => serde_json::json!({
                "name": "write_file",
                "description": "Write contents to a file",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "path": { "type": "string", "description": "Path to the file" },
                        "content": { "type": "string", "description": "Content to write" }
                    },
                    "required": ["path", "content"]
                }
            }),
            "execute_command" => serde_json::json!({
                "name": "execute_command",
                "description": "Execute a shell command",
                "input_schema": {
                    "type": "object",
                    "properties": {
                        "command": { "type": "string", "description": "Command to execute" }
                    },
                    "required": ["command"]
                }
            }),
            _ => serde_json::json!({
                "name": name,
                "description": "Unknown builtin tool",
                "input_schema": { "type": "object", "properties": {} }
            }),
        }
    }

    fn convert_messages(&self, messages: &[Message]) -> Vec<serde_json::Value> {
        messages.iter().filter_map(|m| {
            match m.role {
                Role::System => None, // System is handled separately
                Role::User => Some(serde_json::json!({
                    "role": "user",
                    "content": self.convert_content(&m.content)
                })),
                Role::Assistant => Some(serde_json::json!({
                    "role": "assistant",
                    "content": self.convert_content(&m.content)
                })),
                Role::Tool => {
                    // Tool results are embedded in user messages
                    None
                }
            }
        }).collect()
    }

    fn convert_content(&self, content: &Content) -> serde_json::Value {
        match content {
            Content::Text(text) => serde_json::json!(text),
            Content::Blocks(blocks) => {
                serde_json::json!(blocks.iter().map(|b| {
                    match b {
                        ContentBlock::Text { text } => serde_json::json!({
                            "type": "text",
                            "text": text
                        }),
                        ContentBlock::Image { source } => serde_json::json!({
                            "type": "image",
                            "source": source
                        }),
                        ContentBlock::ToolUse { id, name, input } => serde_json::json!({
                            "type": "tool_use",
                            "id": id,
                            "name": name,
                            "input": input
                        }),
                        ContentBlock::ToolResult { tool_use_id, content, is_error } => serde_json::json!({
                            "type": "tool_result",
                            "tool_use_id": tool_use_id,
                            "content": content,
                            "is_error": is_error
                        }),
                    }
                }).collect::<Vec<_>>())
            }
        }
    }

    fn extract_system_prompt(&self, messages: &[Message]) -> Option<String> {
        messages.iter()
            .find(|m| matches!(m.role, Role::System))
            .and_then(|m| match &m.content {
                Content::Text(text) => Some(text.clone()),
                _ => None,
            })
            .or_else(|| self.config.system_prompt.clone())
    }
}

impl AgentRuntime for ClaudeRuntime {
    fn name(&self) -> &str { "claude" }

    async fn is_available(&self) -> bool {

            .header("anthropic-version", ANTHROPIC_VERSION)

            .await;

    async fn initialize(&mut self, config: &AgentConfig) -> Result<()> {
        self.config = config.clone();
        self.started_at = Some(chrono::Utc::now());
        *self.state.write().await = AgentState::Idle;
        Ok(())
    }

    async fn send(
        &self,
        messages: &[Message],
        tools: &[ToolConfig],
    ) -> Result<AgentResponse> {
        *self.state.write().await = AgentState::Processing;

        let start = std::time::Instant::now();

        let system = self.extract_system_prompt(messages);
        let converted_messages = self.convert_messages(messages);
        let tools_schema = self.build_tools_schema(tools);

        let mut body = serde_json::json!({
            "model": self.config.model.name,
            "messages": converted_messages,
            "max_tokens": self.config.max_tokens.unwrap_or(4096),
        });

        if let Some(system) = system {
            body["system"] = serde_json::json!(system);
        }

        if let Some(temp) = self.config.temperature {
            body["temperature"] = serde_json::json!(temp);
        }

        if !tools_schema.is_empty() {
            body["tools"] = serde_json::json!(tools_schema);
        }

        let response = self.client
            .post(format!("{}/messages", self.base_url))
            .header("x-api-key", &self.api_key)
            .header("anthropic-version", ANTHROPIC_VERSION)
            .header("content-type", "application/json")
            .json(&body)
            .send()
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        if !response.status().is_success() {
            let error = response.text().await.unwrap_or_default();
            *self.state.write().await = AgentState::Error(error.clone());
            return Err(Error::backend(error));
        }

        let result: serde_json::Value = response.json().await
            .map_err(|e| Error::backend(e.to_string()))?;

        // Parse response
        let content = result["content"].as_array()
            .map(|arr| {
                arr.iter()
                    .filter_map(|c| c["text"].as_str())
                    .collect::<Vec<_>>()
                    .join("")
            })
            .unwrap_or_default();

        let tool_calls: Vec<ToolCall> = result["content"].as_array()
            .map(|arr| {
                arr.iter()
                    .filter(|c| c["type"] == "tool_use")
                    .map(|c| ToolCall {
                        id: c["id"].as_str().unwrap_or_default().to_string(),
                        name: c["name"].as_str().unwrap_or_default().to_string(),
                        input: c["input"].clone(),
                    })
                    .collect()
            })
            .unwrap_or_default();

        let stop_reason = match result["stop_reason"].as_str() {
            Some("end_turn") => StopReason::EndTurn,
            Some("tool_use") => StopReason::ToolUse,
            Some("max_tokens") => StopReason::MaxTokens,
            Some("stop_sequence") => StopReason::StopSequence,
            _ => StopReason::EndTurn,
        };

        let usage = TokenUsage {
            input_tokens: result["usage"]["input_tokens"].as_u64().unwrap_or(0),
            output_tokens: result["usage"]["output_tokens"].as_u64().unwrap_or(0),
            cache_read_tokens: result["usage"]["cache_read_input_tokens"].as_u64().unwrap_or(0),
            cache_write_tokens: result["usage"]["cache_creation_input_tokens"].as_u64().unwrap_or(0),
        };

        self.turns.fetch_add(1, Ordering::SeqCst);
        self.input_tokens.fetch_add(usage.input_tokens, Ordering::SeqCst);
        self.output_tokens.fetch_add(usage.output_tokens, Ordering::SeqCst);

        *self.state.write().await = if tool_calls.is_empty() {
            AgentState::Idle
        } else {
            AgentState::WaitingForTool
        };

        Ok(AgentResponse {
            content,
            tool_calls,
            stop_reason,
            usage,
            latency_ms: start.elapsed().as_millis() as u64,
        })
    }

    async fn send_streaming(
        &self,
        messages: &[Message],
        tools: &[ToolConfig],
    ) -> Result<Pin<Box<dyn Stream<Item = StreamEvent> + Send>>> {
        *self.state.write().await = AgentState::Processing;

        let system = self.extract_system_prompt(messages);
        let converted_messages = self.convert_messages(messages);
        let tools_schema = self.build_tools_schema(tools);

        let mut body = serde_json::json!({
            "model": self.config.model.name,
            "messages": converted_messages,
            "max_tokens": self.config.max_tokens.unwrap_or(4096),
            "stream": true,
        });

        if let Some(system) = system {
            body["system"] = serde_json::json!(system);
        }

        if let Some(temp) = self.config.temperature {
            body["temperature"] = serde_json::json!(temp);
        }

        if !tools_schema.is_empty() {
            body["tools"] = serde_json::json!(tools_schema);
        }

        let response = self.client
            .post(format!("{}/messages", self.base_url))
            .header("x-api-key", &self.api_key)
            .header("anthropic-version", ANTHROPIC_VERSION)
            .header("content-type", "application/json")
            .json(&body)
            .send()
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        if !response.status().is_success() {
            let error = response.text().await.unwrap_or_default();
            return Err(Error::backend(error));
        }

        // Parse SSE stream
        let stream = response.bytes_stream().map(|result| {
            match result {
                Ok(bytes) => {
                    let text = String::from_utf8_lossy(&bytes);
                    // Parse SSE events
                    for line in text.lines() {
                        if let Some(data) = line.strip_prefix("data: ") {
                            if let Ok(event) = serde_json::from_str::<serde_json::Value>(data) {
                                let event_type = event["type"].as_str().unwrap_or("");
                                match event_type {
                                    "content_block_delta" => {
                                        if let Some(text) = event["delta"]["text"].as_str() {
                                            return StreamEvent::TextDelta {
                                                index: event["index"].as_u64().unwrap_or(0) as usize,
                                                text: text.to_string(),
                                            };
                                        }
                                    }
                                    "message_stop" => {
                                        return StreamEvent::MessageEnd {
                                            usage: TokenUsage::default(),
                                        };
                                    }
                                    _ => {}
                                }
                            }
                        }
                    }
                    StreamEvent::TextDelta { index: 0, text: String::new() }
                }
                Err(e) => StreamEvent::Error { error: e.to_string() },
            }
        });

        Ok(Box::pin(stream))
    }

    async fn execute_tool(&self, tool_call: &ToolCall) -> Result<ToolResult> {
        self.tool_executor.execute(tool_call).await
    }

    fn info(&self) -> AgentInfo {
        AgentInfo {
            id: self.config.id.as_str().to_string(),
            state: self.state.try_read().map(|s| s.clone()).unwrap_or(AgentState::Idle),
            model: self.config.model.name.clone(),
            turns: self.turns.load(Ordering::SeqCst),
            input_tokens: self.input_tokens.load(Ordering::SeqCst),
            output_tokens: self.output_tokens.load(Ordering::SeqCst),
            started_at: self.started_at,
            last_activity: Some(chrono::Utc::now()),
        }
    }

    async fn cancel(&self) -> Result<()> {
        *self.state.write().await = AgentState::Idle;
        Ok(())
    }

    async fn health(&self) -> Result<HealthStatus> {
        if self.is_available().await {
            Ok(HealthStatus::Healthy)
        } else {
            Ok(HealthStatus::Unhealthy {
                reason: "API not reachable".to_string(),
            })
        }
    }

    async fn count_tokens(&self, text: &str) -> Result<u64> {
        // Rough approximation: ~4 chars per token for Claude
        Ok((text.len() / 4) as u64)
    }

    async fn shutdown(&mut self) -> Result<()> {
        *self.state.write().await = AgentState::Stopped;
        Ok(())
    }
}
```

### 8.3.5 Tool Executor (3h)

**File:** `crates/dtx-agent/src/tool.rs`

```rust
use std::collections::HashMap;
use std::process::Stdio;
use tokio::process::Command;

use super::config::ToolConfig;
use super::message::{ToolCall, ToolResult};
use dtx_core::error::{Error, Result};

/// Executor for agent tools.
pub struct ToolExecutor {
    tools: HashMap<String, ToolConfig>,
}

impl ToolExecutor {
    pub fn new(tools: Vec<ToolConfig>) -> Self {
        let tools = tools.into_iter()
            .map(|t| (Self::tool_name(&t), t))
            .collect();
        Self { tools }
    }

    fn tool_name(tool: &ToolConfig) -> String {
        match tool {
            ToolConfig::Builtin { name, .. } => name.clone(),
            ToolConfig::Command { name, .. } => name.clone(),
            ToolConfig::Http { name, .. } => name.clone(),
            ToolConfig::Resource { name, .. } => name.clone(),
            ToolConfig::Mcp { server, tool } => format!("{}:{}", server, tool),
            ToolConfig::Custom { name, .. } => name.clone(),
        }
    }

    pub async fn execute(&self, tool_call: &ToolCall) -> Result<ToolResult> {
        let tool = self.tools.get(&tool_call.name)
            .ok_or_else(|| Error::config(format!("Unknown tool: {}", tool_call.name)))?;

        match tool {
            ToolConfig::Builtin { name, .. } => {
                self.execute_builtin(name, &tool_call.input).await
            }
            ToolConfig::Command { command, timeout, .. } => {
                self.execute_command(command, &tool_call.input, *timeout).await
            }
            ToolConfig::Http { url, method, headers, .. } => {
                self.execute_http(url, method, headers, &tool_call.input).await
            }
            ToolConfig::Custom { handler, .. } => {
                self.execute_custom(handler, &tool_call.input).await
            }
            _ => {
                Ok(ToolResult {
                    tool_use_id: tool_call.id.clone(),
                    content: "Tool type not implemented".to_string(),
                    is_error: true,
                })
            }
        }
        .map(|content| ToolResult {
            tool_use_id: tool_call.id.clone(),
            content,
            is_error: false,
        })
        .or_else(|e| Ok(ToolResult {
            tool_use_id: tool_call.id.clone(),
            content: e.to_string(),
            is_error: true,
        }))
    }

    async fn execute_builtin(&self, name: &str, input: &serde_json::Value) -> Result<String> {
        match name {
            "read_file" => {
                let path = input["path"].as_str()
                    .ok_or_else(|| Error::config("Missing path"))?;
                tokio::fs::read_to_string(path).await
                    .map_err(|e| Error::backend(e.to_string()))
            }
            "write_file" => {
                let path = input["path"].as_str()
                    .ok_or_else(|| Error::config("Missing path"))?;
                let content = input["content"].as_str()
                    .ok_or_else(|| Error::config("Missing content"))?;
                tokio::fs::write(path, content).await
                    .map_err(|e| Error::backend(e.to_string()))?;
                Ok(format!("Written to {}", path))
            }
            "execute_command" => {
                let command = input["command"].as_str()
                    .ok_or_else(|| Error::config("Missing command"))?;
                let output = Command::new("sh")
                    .args(["-c", command])
                    .output()
                    .await
                    .map_err(|e| Error::backend(e.to_string()))?;

                let stdout = String::from_utf8_lossy(&output.stdout);
                let stderr = String::from_utf8_lossy(&output.stderr);

                Ok(format!(
                    "Exit code: {}\nStdout:\n{}\nStderr:\n{}",
                    output.status.code().unwrap_or(-1),
                    stdout,
                    stderr
                ))
            }
            "list_files" => {
                let path = input["path"].as_str().unwrap_or(".");
                let mut entries = tokio::fs::read_dir(path).await
                    .map_err(|e| Error::backend(e.to_string()))?;

                let mut files = Vec::new();
                while let Some(entry) = entries.next_entry().await? {
                    files.push(entry.file_name().to_string_lossy().to_string());
                }

                Ok(files.join("\n"))
            }
            "search_files" => {
                let pattern = input["pattern"].as_str()
                    .ok_or_else(|| Error::config("Missing pattern"))?;
                let path = input["path"].as_str().unwrap_or(".");

                let output = Command::new("grep")
                    .args(["-r", "-l", pattern, path])
                    .output()
                    .await
                    .map_err(|e| Error::backend(e.to_string()))?;

                Ok(String::from_utf8_lossy(&output.stdout).to_string())
            }
            _ => Err(Error::config(format!("Unknown builtin: {}", name))),
        }
    }

    async fn execute_command(
        &self,
        command: &str,
        input: &serde_json::Value,
        timeout: Option<std::time::Duration>,
    ) -> Result<String> {
        // Interpolate input into command
        let mut cmd = command.to_string();
        if let Some(obj) = input.as_object() {
            for (key, value) in obj {
                let placeholder = format!("{{{}}}", key);
                let value_str = match value {
                    serde_json::Value::String(s) => s.clone(),
                    v => v.to_string(),
                };
                cmd = cmd.replace(&placeholder, &value_str);
            }
        }

        let mut child = Command::new("sh")
            .args(["-c", &cmd])
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .spawn()
            .map_err(|e| Error::backend(e.to_string()))?;

        let output = match timeout {
            Some(t) => {
                match tokio::time::timeout(t, child.wait_with_output()).await {
                    Ok(result) => result.map_err(|e| Error::backend(e.to_string()))?,
                    Err(_) => {
                        child.kill().await.ok();
                        return Err(Error::timeout(t));
                    }
                }
            }
            None => child.wait_with_output().await.map_err(|e| Error::backend(e.to_string()))?,
        };

        Ok(format!(
            "Exit: {}\n{}{}",
            output.status.code().unwrap_or(-1),
            String::from_utf8_lossy(&output.stdout),
            String::from_utf8_lossy(&output.stderr)
        ))
    }

    async fn execute_http(
        &self,
        url: &str,
        method: &str,
        headers: &HashMap<String, String>,
        input: &serde_json::Value,
    ) -> Result<String> {
        let client = reqwest::Client::new();

        let mut request = match method.to_uppercase().as_str() {
            "GET" => client.get(url),
            "POST" => client.post(url).json(input),
            "PUT" => client.put(url).json(input),
            "DELETE" => client.delete(url),
            "PATCH" => client.patch(url).json(input),
            _ => return Err(Error::config(format!("Unknown HTTP method: {}", method))),
        };

        for (key, value) in headers {
            request = request.header(key, value);
        }

        let response = request.send().await
            .map_err(|e| Error::backend(e.to_string()))?;

        let status = response.status();
        let body = response.text().await
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(format!("Status: {}\n{}", status, body))
    }

    async fn execute_custom(
        &self,
        handler: &std::path::Path,
        input: &serde_json::Value,
    ) -> Result<String> {
        let input_json = serde_json::to_string(input)
            .map_err(|e| Error::backend(e.to_string()))?;

        let output = Command::new(handler)
            .stdin(Stdio::piped())
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .spawn()
            .map_err(|e| Error::backend(e.to_string()))?
            .wait_with_output()
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(String::from_utf8_lossy(&output.stdout).to_string())
    }
}
```

```rust
use std::sync::Arc;
use tokio::sync::RwLock;

use crate::config::{AgentConfig, AgentMode};
use crate::runtime::{AgentRuntime, AgentState, detect_runtime};

/// AI Agent resource implementing the Resource trait.
pub struct Agent {
    config: AgentConfig,
    state: ResourceState,
    runtime: Option<Box<dyn AgentRuntime>>,
    event_bus: Arc<EventBus>,
    conversation: Arc<RwLock<Vec<Message>>>,
    shutdown_signal: tokio::sync::watch::Sender<bool>,
    worker_handle: Option<tokio::task::JoinHandle<()>>,
}

impl Agent {
    pub async fn new(
        config: AgentConfig,
        event_bus: Arc<EventBus>,
    ) -> Result<Self> {
        let (shutdown_tx, _) = tokio::sync::watch::channel(false);

        Ok(Self {
            config,
            state: ResourceState::Pending,
            runtime: None,
            event_bus,
            conversation: Arc::new(RwLock::new(Vec::new())),
            shutdown_signal: shutdown_tx,
            worker_handle: None,
        })
    }

    pub fn config(&self) -> &AgentConfig {
        &self.config
    }

    /// Send a message to the agent and get a response.
    pub async fn chat(&self, message: &str) -> Result<String> {
        let runtime = self.runtime.as_ref()
            .ok_or_else(|| Error::backend("Agent not started"))?;

        // Add user message to conversation
        {
            let mut conv = self.conversation.write().await;
            conv.push(Message::user(message));
        }

        // Get response
        let messages = self.conversation.read().await.clone();
        let response = runtime.send(&messages, &self.config.tools).await?;

        // Handle tool calls
        let final_response = if !response.tool_calls.is_empty() {
            let mut conv = self.conversation.write().await;
            conv.push(Message::assistant(&response.content));

            for tool_call in &response.tool_calls {
                let result = runtime.execute_tool(tool_call).await?;
                conv.push(Message::tool_result(&tool_call.id, &result.content, result.is_error));
            }

            // Get final response after tool execution
            let messages = conv.clone();
            drop(conv);
            let final_resp = runtime.send(&messages, &self.config.tools).await?;
            final_resp.content
        } else {
            response.content
        };

        // Add assistant response to conversation
        {
            let mut conv = self.conversation.write().await;
            conv.push(Message::assistant(&final_response));

            // Apply memory limits
            if conv.len() > self.config.memory.max_history * 2 {
                conv.drain(0..conv.len() - self.config.memory.max_history * 2);
            }
        }

        Ok(final_response)
    }

    /// Get agent statistics.
    pub fn stats(&self) -> Option<crate::runtime::AgentInfo> {
        self.runtime.as_ref().map(|r| r.info())
    }

    /// Clear conversation history.
    pub async fn clear_history(&self) {
        self.conversation.write().await.clear();
    }

    async fn run_daemon(&self, mut shutdown_rx: tokio::sync::watch::Receiver<bool>) {
        loop {
            tokio::select! {
                _ = shutdown_rx.changed() => {
                    if *shutdown_rx.borrow() {
                        break;
                    }
                }
                // In daemon mode, agent waits for external triggers
                // This could be via EventBus, HTTP, or other mechanisms
                _ = tokio::time::sleep(std::time::Duration::from_secs(1)) => {
                    // Check for pending tasks, events, etc.
                }
            }
        }
    }

    async fn run_worker(
        &self,
        queue: &str,
        concurrency: usize,
        mut shutdown_rx: tokio::sync::watch::Receiver<bool>,
    ) {
        // Worker mode implementation would connect to a queue system
        // and process tasks concurrently
        loop {
            tokio::select! {
                _ = shutdown_rx.changed() => {
                    if *shutdown_rx.borrow() {
                        break;
                    }
                }
                _ = tokio::time::sleep(std::time::Duration::from_secs(1)) => {
                    // Poll queue for tasks
                }
            }
        }
    }
}

#[async_trait]
impl Resource for Agent {
    fn id(&self) -> &ResourceId {
        &self.config.id
    }

    fn kind(&self) -> ResourceKind {
        ResourceKind::Agent
    }

    fn state(&self) -> &ResourceState {
        &self.state
    }

    async fn start(&mut self, _ctx: &Context) -> Result<()> {
        let started_at = chrono::Utc::now();
        self.state = ResourceState::Starting { started_at };
        self.event_bus.publish(LifecycleEvent::starting(&self.config.id, ResourceKind::Agent));

        // Initialize runtime
        tracing::info!(id = %self.config.id, model = %self.config.model.name, "Initializing agent");
        let mut runtime = detect_runtime(&self.config).await
            .ok_or_else(|| Error::backend("No agent runtime available"))?;

        runtime.initialize(&self.config).await?;

        // Load system prompt
        if let Some(file) = &self.config.system_prompt_file {
            let prompt = tokio::fs::read_to_string(file).await
                .map_err(|e| Error::config(format!("Failed to read system prompt: {}", e)))?;
            self.conversation.write().await.push(Message::system(prompt));
        } else if let Some(prompt) = &self.config.system_prompt {
            self.conversation.write().await.push(Message::system(prompt.clone()));
        }

        self.runtime = Some(runtime);

        // Start mode-specific worker
        let shutdown_rx = self.shutdown_signal.subscribe();
        match &self.config.mode {
            AgentMode::Daemon => {
                let agent = self.conversation.clone();
                self.worker_handle = Some(tokio::spawn(async move {
                    // Daemon loop
                }));
            }
            AgentMode::Worker { queue, concurrency } => {
                let queue = queue.clone();
                let concurrency = *concurrency;
                self.worker_handle = Some(tokio::spawn(async move {
                    // Worker loop
                }));
            }
            AgentMode::Cron { schedule, task } => {
                // Set up cron scheduler
            }
            _ => {
                // OneShot and Interactive don't need background workers
            }
        }

        self.state = ResourceState::Running {
            pid: None,
            started_at,
        };
        self.event_bus.publish(LifecycleEvent::running(&self.config.id, ResourceKind::Agent, None));

        Ok(())
    }

    async fn stop(&mut self, _ctx: &Context) -> Result<()> {
        let started_at = match &self.state {
            ResourceState::Running { started_at, .. } => *started_at,
            _ => chrono::Utc::now(),
        };
        let stopping_at = chrono::Utc::now();

        self.state = ResourceState::Stopping { started_at, stopping_at };
        self.event_bus.publish(LifecycleEvent::stopping(&self.config.id));

        // Signal shutdown
        let _ = self.shutdown_signal.send(true);

        // Wait for worker
        if let Some(handle) = self.worker_handle.take() {
            handle.abort();
        }

        // Shutdown runtime
        if let Some(ref mut runtime) = self.runtime {
            runtime.shutdown().await?;
        }
        self.runtime = None;

        self.state = ResourceState::Stopped {
            exit_code: Some(0),
            started_at,
            stopped_at: chrono::Utc::now(),
        };
        self.event_bus.publish(LifecycleEvent::stopped(&self.config.id, Some(0)));

        Ok(())
    }

    async fn kill(&mut self, _ctx: &Context) -> Result<()> {
        let _ = self.shutdown_signal.send(true);

        if let Some(handle) = self.worker_handle.take() {
            handle.abort();
        }

        self.runtime = None;
        self.state = ResourceState::Stopped {
            exit_code: Some(-9),
            started_at: chrono::Utc::now(),
            stopped_at: chrono::Utc::now(),
        };

        Ok(())
    }

    async fn health(&self) -> HealthStatus {
        match &self.runtime {
            Some(runtime) => runtime.health().await.unwrap_or(HealthStatus::Unknown),
            None => HealthStatus::Unhealthy {
                reason: "Agent not running".to_string(),
            },
        }
    }

    fn logs(&self) -> Option<Box<dyn dtx_core::resource::LogStream>> {
        None
    }

    fn as_any(&self) -> &dyn std::any::Any {
        self
    }

    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
        self
    }
}
```

```rust
            metrics: true,
            hot_restart: false,
        }
```
