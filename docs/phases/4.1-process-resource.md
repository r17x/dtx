### 4.1.2 Process Resource Implementation (4h)

```rust
// crates/dtx-process/src/process.rs

pub struct Process {
    config: ProcessConfig,
    state: ResourceState,
    child: Option<Child>,
    event_bus: Arc<EventBus>,
    stdout_reader: Option<JoinHandle<()>>,
    stderr_reader: Option<JoinHandle<()>>,
}

#[async_trait]
impl Resource for Process {
    fn id(&self) -> &ResourceId {
        &self.config.id
    }

    fn kind(&self) -> ResourceKind {
        ResourceKind::Process
    }

    fn state(&self) -> &ResourceState {
        &self.state
    }

    async fn start(&mut self, ctx: &Context) -> Result<()> {
        // 1. Run init command if configured
        if let Some(init) = &self.config.init_command {
            self.run_init(init, ctx).await?;
        }

        // 2. Build command with environment
        let mut cmd = self.build_command();

        // 3. Spawn process
        self.state = ResourceState::Starting { started_at: Utc::now() };
        self.event_bus.publish(LifecycleEvent::starting(&self.config.id, ResourceKind::Process));

        let child = cmd.spawn()?;
        let pid = child.id();

        // 4. Setup IO readers
        self.setup_io_readers(&child);
        self.child = Some(child);

        // 5. Wait for readiness probe
        if let Some(probe) = &self.config.readiness_probe {
            self.wait_for_readiness(probe, ctx).await?;
        }

        self.state = ResourceState::Running { pid, started_at: Utc::now() };
        self.event_bus.publish(LifecycleEvent::running(&self.config.id, ResourceKind::Process, pid));

        Ok(())
    }

    async fn stop(&mut self, ctx: &Context) -> Result<()> {
        if let Some(mut child) = self.child.take() {
            self.state = ResourceState::Stopping { ... };

            // Run shutdown command if configured
            if let Some(cmd) = &self.config.shutdown.command {
                self.run_shutdown_command(cmd).await;
            }

            // Send signal
            send_signal(&child, self.config.shutdown.signal)?;

            // Wait with timeout
            match timeout(self.config.shutdown.timeout, child.wait()).await {
                Ok(status) => {
                    self.state = ResourceState::Stopped { exit_code: status.code(), ... };
                }
                Err(_) => {
                    // Force kill
                    child.kill().await?;
                    self.state = ResourceState::Stopped { exit_code: None, ... };
                }
            }

            self.event_bus.publish(LifecycleEvent::stopped(...));
        }
        Ok(())
    }

    async fn health(&self) -> HealthStatus {
        if let Some(probe) = &self.config.liveness_probe {
            match self.run_probe(probe).await {
                Ok(true) => HealthStatus::Healthy,
                Ok(false) => HealthStatus::Unhealthy { reason: "Probe failed".into() },
                Err(e) => HealthStatus::Unhealthy { reason: e.to_string() },
            }
        } else {
            // Default: healthy if running
            if self.state.is_running() {
                HealthStatus::Healthy
            } else {
                HealthStatus::Unknown
            }
        }
    }

    fn as_any(&self) -> &dyn Any { self }
    fn as_any_mut(&mut self) -> &mut dyn Any { self }
}
```

**Acceptance:**
- [ ] Implements Resource trait
- [ ] Init command support
- [ ] Graceful shutdown
- [ ] Health probes
- [ ] IO streaming to EventBus

### 4.1.3 Probe Runner (2h)

```rust
// crates/dtx-process/src/probe.rs

pub enum ProbeConfig {
    Exec { command: ShellCommand, settings: ProbeSettings },
    HttpGet { url: String, settings: ProbeSettings },
    TcpSocket { host: String, port: u16, settings: ProbeSettings },
}

pub struct ProbeSettings {
    pub initial_delay: Duration,
    pub period: Duration,
    pub timeout: Duration,
    pub success_threshold: u32,
    pub failure_threshold: u32,
}

impl Process {
    async fn run_probe(&self, probe: &ProbeConfig) -> Result<bool> {
        match probe {
            ProbeConfig::Exec { command, settings } => {
                let output = timeout(settings.timeout,
                    Command::new("sh").args(["-c", command.as_str()]).output()
                ).await??;
                Ok(output.status.success())
            }
            ProbeConfig::HttpGet { url, settings } => {
                let response = timeout(settings.timeout,
                    reqwest::get(url)
                ).await??;
                Ok(response.status().is_success())
            }
            ProbeConfig::TcpSocket { host, port, settings } => {
                let result = timeout(settings.timeout,
                    TcpStream::connect(format!("{}:{}", host, port))
                ).await;
                Ok(result.is_ok())
            }
        }
    }

    async fn wait_for_readiness(&self, probe: &ProbeConfig, ctx: &Context) -> Result<()> {
        let settings = probe.settings();
        tokio::time::sleep(settings.initial_delay).await;

        let mut successes = 0;
        let mut failures = 0;

        loop {
            if ctx.is_expired() {
                return Err(Error::Timeout { elapsed: ctx.elapsed() });
            }

            match self.run_probe(probe).await {
                Ok(true) => {
                    successes += 1;
                    failures = 0;
                    if successes >= settings.success_threshold {
                        return Ok(());
                    }
                }
                Ok(false) | Err(_) => {
                    failures += 1;
                    successes = 0;
                    if failures >= settings.failure_threshold {
                        return Err(Error::HealthCheckFailed { ... });
                    }
                }
            }

            tokio::time::sleep(settings.period).await;
        }
    }
}
```
