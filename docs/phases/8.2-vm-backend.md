# Phase 8.2: VM Backend

> Virtual machine and microVM support as a Resource backend.

---

## Goal

Implement VMs as Resources, enabling virtual machine orchestration alongside native processes and containers with full lifecycle management, supporting QEMU, Firecracker microVMs, and NixOS VMs.

---

## User Value

```
- Run isolated dev environments in VMs
- Test NixOS configurations locally before deployment
- Fast microVM boot for integration testing
- Consistent VM management API
- Hardware-level isolation for security testing
- Production environment simulation
```

---

## Dependencies

```
Phase 1.1 (Resource Trait) - VM implements Resource
Phase 1.2 (EventBus v2) - Lifecycle events
Phase 4.1 (Process Resource) - Reference implementation
Phase 7.2 (Backend Plugins) - BackendPlugin trait
```

### 8.2.2 VM Runtime Trait (3h)

**File:** `crates/dtx-vm/src/runtime.rs`

```rust
use async_trait::async_trait;
use std::path::Path;

use crate::config::VmConfig;
use dtx_core::resource::HealthStatus;
use dtx_core::error::Result;

/// VM information returned by inspect.
#[derive(Clone, Debug)]
pub struct VmInfo {
    /// VM identifier.
    pub id: String,
    /// Current state.
    pub state: VmState,
    /// Process ID of the VM hypervisor.
    pub pid: Option<u32>,
    /// Boot time.
    pub started_at: Option<chrono::DateTime<chrono::Utc>>,
    /// IP address (if known).
    pub ip_address: Option<String>,
    /// SSH port on host.
    pub ssh_port: Option<u16>,
    /// VNC/SPICE port.
    pub display_port: Option<u16>,
    /// QMP socket path (QEMU).
    pub qmp_socket: Option<std::path::PathBuf>,
}

/// VM state.
#[derive(Clone, Debug, PartialEq)]
pub enum VmState {
    Pending,
    Booting,
    Running,
    Paused,
    Shutoff,
    Crashed,
}

/// Abstract VM runtime interface.
#[async_trait]
pub trait VmRuntime: Send + Sync {
    /// Runtime name.
    fn name(&self) -> &str;

    /// Check if runtime is available.
    async fn is_available(&self) -> bool;

    /// Prepare VM image (download, build, etc.).
    async fn prepare_image(&self, config: &VmConfig) -> Result<std::path::PathBuf>;

    /// Create VM (without starting).
    async fn create(&self, config: &VmConfig) -> Result<String>;

    /// Start VM.
    async fn start(&self, id: &str) -> Result<()>;

    /// Stop VM gracefully (ACPI shutdown).
    async fn stop(&self, id: &str, timeout: std::time::Duration) -> Result<()>;

    /// Force stop VM.
    async fn kill(&self, id: &str) -> Result<()>;

    /// Pause VM.
    async fn pause(&self, id: &str) -> Result<()>;

    /// Resume paused VM.
    async fn resume(&self, id: &str) -> Result<()>;

    /// Restart VM.
    async fn restart(&self, id: &str) -> Result<()>;

    /// Get VM info.
    async fn inspect(&self, id: &str) -> Result<VmInfo>;

    /// Check if VM is running.
    async fn is_running(&self, id: &str) -> Result<bool>;

    /// Wait for VM to boot (SSH available or other check).
    async fn wait_for_boot(
        &self,
        id: &str,
        timeout: std::time::Duration,
    ) -> Result<()>;

    /// Execute command in VM via SSH.
    async fn exec(
        &self,
        id: &str,
        command: &[String],
        config: &VmConfig,
    ) -> Result<ExecResult>;

    /// Get VM console output.
    async fn console_log(&self, id: &str, lines: Option<usize>) -> Result<String>;

    /// Remove VM and cleanup.
    async fn remove(&self, id: &str, config: &VmConfig) -> Result<()>;

    /// Get health status.
    async fn health(&self, id: &str, config: &VmConfig) -> Result<HealthStatus>;

    /// Create snapshot.
    async fn snapshot(&self, id: &str, name: &str) -> Result<String>;

    /// Restore snapshot.
    async fn restore_snapshot(&self, id: &str, name: &str) -> Result<()>;

    /// List snapshots.
    async fn list_snapshots(&self, id: &str) -> Result<Vec<SnapshotInfo>>;
}

/// Result of command execution.
#[derive(Clone, Debug)]
pub struct ExecResult {
    pub exit_code: i32,
    pub stdout: String,
    pub stderr: String,
}

/// Snapshot information.
#[derive(Clone, Debug)]
pub struct SnapshotInfo {
    pub name: String,
    pub created_at: chrono::DateTime<chrono::Utc>,
    pub size_bytes: u64,
}

/// Select the best available VM runtime.
pub async fn detect_runtime() -> Option<Box<dyn VmRuntime>> {
    // Try QEMU first (most common)
    let qemu = super::qemu::QemuRuntime::new();
    if qemu.is_available().await {
        return Some(Box::new(qemu));
    }

    // Try Firecracker (fast microVMs)
    let firecracker = super::firecracker::FirecrackerRuntime::new();
    if firecracker.is_available().await {
        return Some(Box::new(firecracker));
    }

    // Try NixOS VM
    let nixos = super::nixos::NixosVmRuntime::new();
    if nixos.is_available().await {
        return Some(Box::new(nixos));
    }

    None
}
```

### 8.2.3 QEMU Runtime Implementation (4h)

**File:** `crates/dtx-vm/src/qemu.rs`

```rust
use async_trait::async_trait;
use std::collections::HashMap;
use std::path::{Path, PathBuf};
use std::process::Stdio;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};
use tokio::process::{Child, Command};
use tokio::net::UnixStream;

use super::runtime::{VmRuntime, VmInfo, VmState, ExecResult, SnapshotInfo};
use super::config::{VmConfig, VmImage, NetworkMode, GraphicsConfig};
use dtx_core::resource::HealthStatus;
use dtx_core::error::{Error, Result};

/// QEMU runtime implementation.
pub struct QemuRuntime {
    /// Path to QEMU binary.
    qemu_binary: PathBuf,
    /// Running VMs.
    vms: tokio::sync::RwLock<HashMap<String, QemuVm>>,
    /// Base directory for VM state.
    state_dir: PathBuf,
}

struct QemuVm {
    config: VmConfig,
    process: Child,
    qmp_socket: PathBuf,
    console_socket: PathBuf,
    ssh_port: Option<u16>,
}

impl QemuRuntime {
    pub fn new() -> Self {
        let qemu_binary = which::which("qemu-system-x86_64")
            .or_else(|_| which::which("qemu-kvm"))
            .unwrap_or_else(|_| PathBuf::from("qemu-system-x86_64"));

        Self {
            qemu_binary,
            vms: tokio::sync::RwLock::new(HashMap::new()),
            state_dir: std::env::temp_dir().join("dtx-vm"),
        }
    }

    pub fn with_binary(binary: PathBuf) -> Self {
        Self {
            qemu_binary: binary,
            vms: tokio::sync::RwLock::new(HashMap::new()),
            state_dir: std::env::temp_dir().join("dtx-vm"),
        }
    }

    fn build_qemu_args(&self, config: &VmConfig, image_path: &Path) -> Vec<String> {
        let mut args = Vec::new();

        // Machine type
        args.extend(["-machine", "q35,accel=kvm:tcg"]);

        // CPU
        args.extend(["-cpu", config.cpu.model.as_deref().unwrap_or("host")]);
        args.extend(["-smp", &config.cpu.count.to_string()]);

        // Memory
        args.extend(["-m", &config.memory.size]);

        // Enable KVM if available
        if config.kvm {
            args.push("-enable-kvm");
        }

        // Boot drive
        args.extend([
            "-drive",
            &format!(
                "file={},format={},if=virtio",
                image_path.display(),
                match config.image {
                    VmImage::File { ref format, .. } => format.as_str(),
                    _ => "qcow2",
                }
            ),
        ]);

        // Additional disks
        for (i, disk) in config.disks.iter().enumerate() {
            args.extend([
                "-drive",
                &format!(
                    "file={},format={},if={},index={}{}",
                    disk.path.display(),
                    disk.format.as_str(),
                    disk.interface.as_str(),
                    i + 1,
                    if disk.read_only { ",readonly=on" } else { "" }
                ),
            ]);
        }

        // Network
        match &config.network.mode {
            NetworkMode::User => {
                let mut netdev = "user,id=net0".to_string();
                for pf in &config.port_forwards {
                    netdev.push_str(&format!(
                        ",hostfwd={}::{}-:{}",
                        pf.protocol.as_str(),
                        pf.host,
                        pf.guest
                    ));
                }
                // Add SSH forwarding
                if let Some(ssh) = &config.ssh {
                    if let Some(host_port) = ssh.host_port {
                        netdev.push_str(&format!(",hostfwd=tcp::{}-:{}", host_port, ssh.port));
                    }
                }
                args.extend(["-netdev", &netdev]);
                args.extend(["-device", "virtio-net-pci,netdev=net0"]);
            }
            NetworkMode::Bridged => {
                if let Some(bridge) = &config.network.bridge {
                    args.extend(["-netdev", &format!("bridge,id=net0,br={}", bridge)]);
                    args.extend(["-device", "virtio-net-pci,netdev=net0"]);
                }
            }
            NetworkMode::Tap => {
                if let Some(tap) = &config.network.tap {
                    args.extend(["-netdev", &format!("tap,id=net0,ifname={},script=no,downscript=no", tap)]);
                    args.extend(["-device", "virtio-net-pci,netdev=net0"]);
                }
            }
            NetworkMode::None => {
                args.extend(["-net", "none"]);
            }
        }

        // Shared directories (virtio-fs)
        for share in &config.shared_dirs {
            args.extend([
                "-virtfs",
                &format!(
                    "local,path={},mount_tag={},security_model=mapped-xattr{}",
                    share.source.display(),
                    share.tag,
                    if share.read_only { ",readonly=on" } else { "" }
                ),
            ]);
        }

        // Graphics
        match &config.graphics {
            GraphicsConfig::None => {
                args.extend(["-display", "none"]);
            }
            GraphicsConfig::Vnc { port, password } => {
                let display = port.map(|p| format!(":{}", p - 5900)).unwrap_or(":0".to_string());
                if password.is_some() {
                    args.extend(["-vnc", &format!("{},password=on", display)]);
                } else {
                    args.extend(["-vnc", &display]);
                }
            }
            GraphicsConfig::Spice { port } => {
                let p = port.unwrap_or(5930);
                args.extend(["-spice", &format!("port={},disable-ticketing=on", p)]);
            }
            GraphicsConfig::Gtk => {
                args.extend(["-display", "gtk"]);
            }
        }

        // QMP socket for control
        let qmp_socket = self.state_dir.join(format!("{}.qmp", config.id.as_str()));
        args.extend(["-qmp", &format!("unix:{},server,nowait", qmp_socket.display())]);

        // Console output
        let console_socket = self.state_dir.join(format!("{}.console", config.id.as_str()));
        args.extend(["-serial", &format!("unix:{},server,nowait", console_socket.display())]);

        // Cloud-init/NoCloud
        if let Some(init_config) = &config.init_config {
            // Create cloud-init ISO or config drive
            // (implementation omitted for brevity)
        }

        // Extra args
        args.extend(config.extra_args.iter().cloned());

        args
    }

    async fn qmp_command(&self, socket: &Path, command: &str) -> Result<serde_json::Value> {
        let mut stream = UnixStream::connect(socket).await
            .map_err(|e| Error::backend(format!("QMP connect failed: {}", e)))?;

        // Read greeting
        let mut reader = BufReader::new(&mut stream);
        let mut line = String::new();
        reader.read_line(&mut line).await?;

        // Enter command mode
        stream.write_all(b"{\"execute\":\"qmp_capabilities\"}\n").await?;
        line.clear();
        reader.read_line(&mut line).await?;

        // Send command
        let cmd = format!("{{\"execute\":\"{}\"}}\n", command);
        stream.write_all(cmd.as_bytes()).await?;
        line.clear();
        reader.read_line(&mut line).await?;

        serde_json::from_str(&line).map_err(|e| Error::backend(e.to_string()))
    }

    async fn ssh_exec(&self, config: &VmConfig, command: &[String]) -> Result<ExecResult> {
        let ssh = config.ssh.as_ref()
            .ok_or_else(|| Error::config("SSH not configured"))?;

        let host_port = ssh.host_port
            .ok_or_else(|| Error::config("SSH host port not configured"))?;

        let mut cmd = Command::new("ssh");
        cmd.args([
            "-o", "StrictHostKeyChecking=no",
            "-o", "UserKnownHostsFile=/dev/null",
            "-o", "ConnectTimeout=5",
            "-p", &host_port.to_string(),
        ]);

        if let Some(key) = &ssh.identity_file {
            cmd.args(["-i", &key.to_string_lossy()]);
        }

        cmd.arg(format!("{}@localhost", ssh.user));
        cmd.args(command);

        let output = cmd.output().await
            .map_err(|e| Error::backend(format!("SSH failed: {}", e)))?;

        Ok(ExecResult {
            exit_code: output.status.code().unwrap_or(-1),
            stdout: String::from_utf8_lossy(&output.stdout).to_string(),
            stderr: String::from_utf8_lossy(&output.stderr).to_string(),
        })
    }
}

    async fn prepare_image(&self, config: &VmConfig) -> Result<PathBuf> {
        match &config.image {
            VmImage::File { path, .. } => {
                if !path.exists() {
                    return Err(Error::config(format!("Image not found: {}", path.display())));
                }
                Ok(path.clone())
            }
            VmImage::NixosFlake { flake, attribute } => {
                let attr = attribute.as_deref().unwrap_or("vm");
                let output = Command::new("nix")
                    .args(["build", "--no-link", "--print-out-paths", &format!("{}#{}", flake, attr)])
                    .output()
                    .await
                    .map_err(|e| Error::backend(format!("Nix build failed: {}", e)))?;

                if !output.status.success() {
                    return Err(Error::backend(String::from_utf8_lossy(&output.stderr).to_string()));
                }

                let path = String::from_utf8_lossy(&output.stdout).trim().to_string();
                Ok(PathBuf::from(path).join("nixos.qcow2"))
            }
            VmImage::Url { url, checksum } => {
                let cache_dir = self.state_dir.join("images");
                tokio::fs::create_dir_all(&cache_dir).await?;

                let filename = url.split('/').last().unwrap_or("image.qcow2");
                let target = cache_dir.join(filename);

                if !target.exists() {
                    tracing::info!(url = %url, "Downloading VM image");
                    let output = Command::new("curl")
                        .args(["-L", "-o", &target.to_string_lossy(), url])
                        .output()
                        .await?;

                    if !output.status.success() {
                        return Err(Error::backend("Download failed"));
                    }
                }

                Ok(target)
            }
            VmImage::Container { image } => {
                Err(Error::config("Container images not supported by QEMU runtime"))
            }
            VmImage::NixBuild { expression, attribute } => {
                let output = Command::new("nix-build")
                    .args(["-E", expression, "-A", attribute, "--no-out-link"])
                    .output()
                    .await?;

                if !output.status.success() {
                    return Err(Error::backend(String::from_utf8_lossy(&output.stderr).to_string()));
                }

                let path = String::from_utf8_lossy(&output.stdout).trim().to_string();
                Ok(PathBuf::from(path))
            }
        }
    }

        if vms.contains_key(id) {
            return Err(Error::backend("VM already running"));

        drop(vms);

        // Config needs to be passed in - for now return error
        Err(Error::backend("Use start_with_config for QEMU"))

    async fn stop(&self, id: &str, timeout: std::time::Duration) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            // Send ACPI shutdown via QMP
            self.qmp_command(&vm.qmp_socket, "system_powerdown").await?;
        }
        drop(vms);

        // Wait for process to exit
        let deadline = tokio::time::Instant::now() + timeout;
        loop {
            let vms = self.vms.read().await;
            if let Some(vm) = vms.get(id) {
                // Check if still running
                if tokio::time::Instant::now() > deadline {
                    drop(vms);
                    return self.kill(id).await;
                }
            } else {
                break;
            }
            drop(vms);
            tokio::time::sleep(std::time::Duration::from_millis(500)).await;
        }

        Ok(())
    }

    async fn kill(&self, id: &str) -> Result<()> {
        let mut vms = self.vms.write().await;
        if let Some(mut vm) = vms.remove(id) {
            vm.process.kill().await.ok();
        }
        Ok(())
    }

    async fn pause(&self, id: &str) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            self.qmp_command(&vm.qmp_socket, "stop").await?;
        }
        Ok(())
    }

    async fn resume(&self, id: &str) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            self.qmp_command(&vm.qmp_socket, "cont").await?;
        }
        Ok(())
    }

    async fn restart(&self, id: &str) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            self.qmp_command(&vm.qmp_socket, "system_reset").await?;
        }
        Ok(())
    }

    async fn inspect(&self, id: &str) -> Result<VmInfo> {
        let vms = self.vms.read().await;
        let vm = vms.get(id)
            .ok_or_else(|| Error::backend("VM not found"))?;

        Ok(VmInfo {

            ip_address: None,
            ssh_port: vm.ssh_port,
            display_port: None,
            qmp_socket: Some(vm.qmp_socket.clone()),
        })
    }

    async fn is_running(&self, id: &str) -> Result<bool> {
        let vms = self.vms.read().await;
        Ok(vms.contains_key(id))
    }

    async fn wait_for_boot(&self, id: &str, timeout: std::time::Duration) -> Result<()> {
        let deadline = tokio::time::Instant::now() + timeout;

        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;
        let ssh_port = vm.ssh_port;
        let config = vm.config.clone();
        drop(vms);

        if let Some(port) = ssh_port {
            // Wait for SSH to be available
            loop {
                if tokio::time::Instant::now() > deadline {
                    return Err(Error::timeout(timeout));
                }

                let result = tokio::time::timeout(
                    std::time::Duration::from_secs(2),
                    tokio::net::TcpStream::connect(format!("127.0.0.1:{}", port)),
                ).await;

                if let Ok(Ok(_)) = result {
                    return Ok(());
                }

                tokio::time::sleep(std::time::Duration::from_secs(1)).await;
            }
        }

        Ok(())
    }

    async fn exec(&self, id: &str, command: &[String], config: &VmConfig) -> Result<ExecResult> {
        self.ssh_exec(config, command).await
    }

    async fn console_log(&self, id: &str, lines: Option<usize>) -> Result<String> {
        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;

        // Read from console socket
        let mut stream = UnixStream::connect(&vm.console_socket).await?;
        let mut buffer = Vec::new();
        tokio::io::AsyncReadExt::read_to_end(&mut stream, &mut buffer).await?;

        let output = String::from_utf8_lossy(&buffer).to_string();
        if let Some(n) = lines {
            Ok(output.lines().rev().take(n).collect::<Vec<_>>().into_iter().rev().collect::<Vec<_>>().join("\n"))
        } else {
            Ok(output)
        }
    }

    async fn remove(&self, id: &str, _config: &VmConfig) -> Result<()> {
        self.kill(id).await?;

        // Cleanup state files
        let _ = tokio::fs::remove_file(self.state_dir.join(format!("{}.qmp", id))).await;
        let _ = tokio::fs::remove_file(self.state_dir.join(format!("{}.console", id))).await;

        Ok(())
    }

    async fn health(&self, id: &str, config: &VmConfig) -> Result<HealthStatus> {
        if !self.is_running(id).await? {
            return Ok(HealthStatus::Unhealthy {
                reason: "VM not running".to_string(),
            });
        }

        match &config.health_check {
            Some(VmHealthCheck::Ssh { .. }) => {
                match self.ssh_exec(config, &["true".to_string()]).await {
                    Ok(result) if result.exit_code == 0 => Ok(HealthStatus::Healthy),
                    _ => Ok(HealthStatus::Unhealthy {
                        reason: "SSH check failed".to_string(),
                    }),
                }
            }
            Some(VmHealthCheck::Exec { command, .. }) => {
                match self.exec(id, command, config).await {
                    Ok(result) if result.exit_code == 0 => Ok(HealthStatus::Healthy),
                    _ => Ok(HealthStatus::Unhealthy {
                        reason: "Exec check failed".to_string(),
                    }),
                }
            }
            Some(VmHealthCheck::Qmp { .. }) => {
                let vms = self.vms.read().await;
                if let Some(vm) = vms.get(id) {
                    match self.qmp_command(&vm.qmp_socket, "query-status").await {
                        Ok(_) => Ok(HealthStatus::Healthy),
                        Err(e) => Ok(HealthStatus::Unhealthy {
                            reason: format!("QMP check failed: {}", e),
                        }),
                    }
                } else {
                    Ok(HealthStatus::Unknown)
                }
            }
            _ => Ok(HealthStatus::Unknown),
        }
    }

    async fn snapshot(&self, id: &str, name: &str) -> Result<String> {
        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;

        // Use QMP to create snapshot
        let cmd = format!("savevm {}", name);

    }

        Ok(Vec::new())
```

```rust
            "snapshot_path": snapshot_path.to_string_lossy(),
            "mem_file_path": self.state_dir.join(format!("{}-{}.mem", id, name)).to_string_lossy(),
        });

        self.call_api(&vm.socket, "PUT", "/snapshot/create", Some(&body.to_string())).await?;
        Ok(snapshot_path.to_string_lossy().to_string())
    }

    async fn restore_snapshot(&self, id: &str, name: &str) -> Result<()> {
        let snapshot_path = self.state_dir.join(format!("{}-{}.snap", id, name));
        let mem_path = self.state_dir.join(format!("{}-{}.mem", id, name));

        // Restore requires recreating the VM
        Err(Error::backend("Firecracker snapshot restore requires VM recreation"))
    }

    async fn list_snapshots(&self, id: &str) -> Result<Vec<SnapshotInfo>> {
        let mut snapshots = Vec::new();
        let mut entries = tokio::fs::read_dir(&self.state_dir).await?;

        while let Some(entry) = entries.next_entry().await? {
            let name = entry.file_name().to_string_lossy().to_string();
            if name.starts_with(id) && name.ends_with(".snap") {
                let metadata = entry.metadata().await?;
                snapshots.push(SnapshotInfo {
                    name: name.strip_prefix(&format!("{}-", id))
                        .and_then(|s| s.strip_suffix(".snap"))
                        .unwrap_or(&name)
                        .to_string(),
                    created_at: metadata.modified().ok()
                        .map(|t| chrono::DateTime::from(t))
                        .unwrap_or_else(chrono::Utc::now),
                    size_bytes: metadata.len(),
                });
            }
        }

        Ok(snapshots)
    }
}

fn parse_memory_mib(mem: &str) -> u64 {
    let mem = mem.to_lowercase();
    if mem.ends_with('g') {
        mem.trim_end_matches('g').parse::<u64>().unwrap_or(1) * 1024
    } else if mem.ends_with('m') {
        mem.trim_end_matches('m').parse::<u64>().unwrap_or(512)
    } else {
        mem.parse::<u64>().unwrap_or(512)
    }
}

fn generate_mac() -> String {
    use rand::Rng;
    let mut rng = rand::thread_rng();
    format!(
        "AA:FC:00:{:02X}:{:02X}:{:02X}",
        rng.gen::<u8>(),
        rng.gen::<u8>(),
        rng.gen::<u8>()
    )
}
```

```rust
use async_trait::async_trait;
use std::collections::HashMap;
use std::path::PathBuf;
use tokio::process::{Child, Command};

use super::runtime::{VmRuntime, VmInfo, VmState, ExecResult, SnapshotInfo};
use super::config::{VmConfig, VmImage};
use dtx_core::resource::HealthStatus;
use dtx_core::error::{Error, Result};

/// NixOS VM runtime using nixos-rebuild build-vm.
///
/// Provides first-class NixOS integration for testing configurations.
pub struct NixosVmRuntime {
    state_dir: PathBuf,
    vms: tokio::sync::RwLock<HashMap<String, NixosVm>>,
}

struct NixosVm {
    config: VmConfig,
    process: Child,
    ssh_port: u16,
    qemu_pid: Option<u32>,
}

impl NixosVmRuntime {
    pub fn new() -> Self {
        Self {
            state_dir: std::env::temp_dir().join("dtx-vm-nixos"),
            vms: tokio::sync::RwLock::new(HashMap::new()),
        }
    }

    async fn build_nixos_vm(&self, flake: &str, attribute: Option<&str>) -> Result<PathBuf> {
        let attr = attribute.unwrap_or("nixosConfigurations.default.config.system.build.vm");
        let output = Command::new("nix")
            .args([
                "build",
                "--no-link",
                "--print-out-paths",
                &format!("{}#{}", flake, attr),
            ])
            .output()
            .await
            .map_err(|e| Error::backend(format!("nix build failed: {}", e)))?;

        if !output.status.success() {
            return Err(Error::backend(format!(
                "nix build failed: {}",
                String::from_utf8_lossy(&output.stderr)
            )));
        }

        let path = String::from_utf8_lossy(&output.stdout).trim().to_string();
        Ok(PathBuf::from(path))
    }
}

#[async_trait]
impl VmRuntime for NixosVmRuntime {
    fn name(&self) -> &str { "nixos" }

    async fn is_available(&self) -> bool {
        Command::new("nix")
            .args(["--version"])
            .output()
            .await
            .map(|o| o.status.success())
            .unwrap_or(false)
    }

    async fn prepare_image(&self, config: &VmConfig) -> Result<PathBuf> {
        match &config.image {
            VmImage::NixosFlake { flake, attribute } => {
                self.build_nixos_vm(flake, attribute.as_deref()).await
            }
            VmImage::NixBuild { expression, attribute } => {
                let output = Command::new("nix-build")
                    .args(["-E", expression, "-A", attribute, "--no-out-link"])
                    .output()
                    .await?;

                if !output.status.success() {
                    return Err(Error::backend(String::from_utf8_lossy(&output.stderr).to_string()));
                }

                Ok(PathBuf::from(String::from_utf8_lossy(&output.stdout).trim()))
            }
            _ => Err(Error::config("NixOS runtime requires NixosFlake or NixBuild image")),
        }
    }

    async fn create(&self, config: &VmConfig) -> Result<String> {
        tokio::fs::create_dir_all(&self.state_dir).await?;
        Ok(config.id.as_str().to_string())
    }

    async fn start(&self, id: &str) -> Result<()> {
        Err(Error::backend("Use start_with_config for NixOS VMs"))
    }

    async fn stop(&self, id: &str, timeout: std::time::Duration) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            // SSH shutdown
            let _ = Command::new("ssh")
                .args([
                    "-o", "StrictHostKeyChecking=no",
                    "-p", &vm.ssh_port.to_string(),
                    "root@localhost",
                    "poweroff",
                ])
                .output()
                .await;
        }
        drop(vms);

        tokio::time::sleep(timeout).await;
        self.kill(id).await
    }

    async fn kill(&self, id: &str) -> Result<()> {
        let mut vms = self.vms.write().await;
        if let Some(mut vm) = vms.remove(id) {
            vm.process.kill().await.ok();
        }
        Ok(())
    }

    async fn pause(&self, _id: &str) -> Result<()> {
        Err(Error::backend("NixOS VM pause not supported"))
    }

    async fn resume(&self, _id: &str) -> Result<()> {
        Err(Error::backend("NixOS VM resume not supported"))
    }

    async fn restart(&self, id: &str) -> Result<()> {
        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            let _ = Command::new("ssh")
                .args([
                    "-o", "StrictHostKeyChecking=no",
                    "-p", &vm.ssh_port.to_string(),
                    "root@localhost",
                    "reboot",
                ])
                .output()
                .await;
        }
        Ok(())
    }

    async fn inspect(&self, id: &str) -> Result<VmInfo> {
        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;

        Ok(VmInfo {
            id: id.to_string(),
            state: VmState::Running,
            pid: vm.qemu_pid,
            started_at: None,
            ip_address: Some("localhost".to_string()),
            ssh_port: Some(vm.ssh_port),
            display_port: None,
            qmp_socket: None,
        })
    }

    async fn is_running(&self, id: &str) -> Result<bool> {
        let vms = self.vms.read().await;
        Ok(vms.contains_key(id))
    }

    async fn wait_for_boot(&self, id: &str, timeout: std::time::Duration) -> Result<()> {
        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;
        let ssh_port = vm.ssh_port;
        drop(vms);

        let deadline = tokio::time::Instant::now() + timeout;
        loop {
            if tokio::time::Instant::now() > deadline {
                return Err(Error::timeout(timeout));
            }

            let result = Command::new("ssh")
                .args([
                    "-o", "StrictHostKeyChecking=no",
                    "-o", "ConnectTimeout=2",
                    "-p", &ssh_port.to_string(),
                    "root@localhost",
                    "true",
                ])
                .output()
                .await;

            if result.map(|o| o.status.success()).unwrap_or(false) {
                return Ok(());
            }

            tokio::time::sleep(std::time::Duration::from_secs(1)).await;
        }
    }

    async fn exec(&self, id: &str, command: &[String], _config: &VmConfig) -> Result<ExecResult> {
        let vms = self.vms.read().await;
        let vm = vms.get(id).ok_or_else(|| Error::backend("VM not found"))?;

        let output = Command::new("ssh")
            .args([
                "-o", "StrictHostKeyChecking=no",
                "-p", &vm.ssh_port.to_string(),
                "root@localhost",
            ])
            .args(command)
            .output()
            .await?;

        Ok(ExecResult {
            exit_code: output.status.code().unwrap_or(-1),
            stdout: String::from_utf8_lossy(&output.stdout).to_string(),
            stderr: String::from_utf8_lossy(&output.stderr).to_string(),
        })
    }

    async fn console_log(&self, id: &str, lines: Option<usize>) -> Result<String> {
        // NixOS VM console goes to stdout
        let log_path = self.state_dir.join(format!("{}.log", id));
        let content = tokio::fs::read_to_string(log_path).await.unwrap_or_default();

        if let Some(n) = lines {
            Ok(content.lines().rev().take(n).collect::<Vec<_>>().into_iter().rev().collect::<Vec<_>>().join("\n"))
        } else {
            Ok(content)
        }
    }

    async fn remove(&self, id: &str, _config: &VmConfig) -> Result<()> {
        self.kill(id).await?;
        let _ = tokio::fs::remove_file(self.state_dir.join(format!("{}.log", id))).await;
        Ok(())
    }

    async fn health(&self, id: &str, _config: &VmConfig) -> Result<HealthStatus> {
        if !self.is_running(id).await? {
            return Ok(HealthStatus::Unhealthy {
                reason: "VM not running".to_string(),
            });
        }

        let vms = self.vms.read().await;
        if let Some(vm) = vms.get(id) {
            let result = Command::new("ssh")
                .args([
                    "-o", "StrictHostKeyChecking=no",
                    "-o", "ConnectTimeout=2",
                    "-p", &vm.ssh_port.to_string(),
                    "root@localhost",
                    "true",
                ])
                .output()
                .await;

            match result {
                Ok(o) if o.status.success() => Ok(HealthStatus::Healthy),
                _ => Ok(HealthStatus::Unhealthy {
                    reason: "SSH check failed".to_string(),
                }),
            }
        } else {
            Ok(HealthStatus::Unknown)
        }
    }

    async fn snapshot(&self, _id: &str, _name: &str) -> Result<String> {
        Err(Error::backend("NixOS VM snapshots not supported"))
    }

    async fn restore_snapshot(&self, _id: &str, _name: &str) -> Result<()> {
        Err(Error::backend("NixOS VM snapshots not supported"))
    }

    async fn list_snapshots(&self, _id: &str) -> Result<Vec<SnapshotInfo>> {
        Ok(Vec::new())
    }
}
```

```rust
use std::sync::Arc;

/// VM resource implementing the Resource trait.
pub struct Vm {

impl Vm {

        Self {

            state: ResourceState::Pending,
            runtime,
            event_bus,
            image_path: None,
        }
    }

    pub fn config(&self) -> &VmConfig {
        &self.config
    }

    pub async fn exec(&self, command: &[String]) -> Result<crate::runtime::ExecResult> {
        let id = self.config.id.as_str();
        self.runtime.exec(id, command, &self.config).await
    }

    pub async fn console_log(&self, lines: Option<usize>) -> Result<String> {
        self.runtime.console_log(self.config.id.as_str(), lines).await
    }

    pub async fn snapshot(&self, name: &str) -> Result<String> {
        self.runtime.snapshot(self.config.id.as_str(), name).await
    }

    pub async fn restore_snapshot(&self, name: &str) -> Result<()> {
        self.runtime.restore_snapshot(self.config.id.as_str(), name).await
    }
}

#[async_trait]
impl Resource for Vm {
    fn id(&self) -> &ResourceId {
        &self.config.id
    }

    fn kind(&self) -> ResourceKind {
        ResourceKind::VM
    }

    fn state(&self) -> &ResourceState {
        &self.state
    }

    async fn start(&mut self, ctx: &Context) -> Result<()> {
        let started_at = chrono::Utc::now();
        self.state = ResourceState::Starting { started_at };
        self.event_bus.publish(LifecycleEvent::starting(&self.config.id, ResourceKind::VM));

        // Prepare image
        tracing::info!(id = %self.config.id, "Preparing VM image");
        let image_path = self.runtime.prepare_image(&self.config).await?;
        self.image_path = Some(image_path.clone());

        // Create VM
        tracing::info!(id = %self.config.id, "Creating VM");
        self.runtime.create(&self.config).await?;

        // Start VM
        tracing::info!(id = %self.config.id, "Starting VM");
        // Note: Real implementation would pass config to runtime.start_with_config
        // self.runtime.start_with_config(&self.config, &image_path).await?;

        // Wait for boot
        tracing::info!(id = %self.config.id, "Waiting for VM to boot");
        self.runtime.wait_for_boot(
            self.config.id.as_str(),
            self.config.boot_timeout,
        ).await?;

        // Get VM info
        let info = self.runtime.inspect(self.config.id.as_str()).await?;

        self.state = ResourceState::Running {
            pid: info.pid,
            started_at,
        };
        self.event_bus.publish(LifecycleEvent::running(
            &self.config.id,
            ResourceKind::VM,
            info.pid,
        ));

        Ok(())
    }

    async fn stop(&mut self, _ctx: &Context) -> Result<()> {
        let id = self.config.id.as_str();

        let started_at = match &self.state {
            ResourceState::Running { started_at, .. } => *started_at,
            _ => chrono::Utc::now(),
        };
        let stopping_at = chrono::Utc::now();

        self.state = ResourceState::Stopping { started_at, stopping_at };
        self.event_bus.publish(LifecycleEvent::stopping(&self.config.id));

        tracing::info!(id = %self.config.id, "Stopping VM");
        self.runtime.stop(id, self.config.shutdown_timeout).await?;

        self.state = ResourceState::Stopped {
            exit_code: Some(0),
            started_at,
            stopped_at: chrono::Utc::now(),
        };
        self.event_bus.publish(LifecycleEvent::stopped(&self.config.id, Some(0)));

        Ok(())
    }

    async fn kill(&mut self, _ctx: &Context) -> Result<()> {
        tracing::info!(id = %self.config.id, "Killing VM");
        self.runtime.kill(self.config.id.as_str()).await?;

        self.state = ResourceState::Stopped {
            exit_code: Some(-9),
            started_at: chrono::Utc::now(),
            stopped_at: chrono::Utc::now(),
        };

        Ok(())
    }

    async fn restart(&mut self, ctx: &Context) -> Result<()> {
        tracing::info!(id = %self.config.id, "Restarting VM");
        self.runtime.restart(self.config.id.as_str()).await?;

        // Wait for boot again
        self.runtime.wait_for_boot(
            self.config.id.as_str(),
            self.config.boot_timeout,
        ).await?;

        Ok(())
    }

    async fn health(&self) -> HealthStatus {
        self.runtime.health(self.config.id.as_str(), &self.config)
            .await
            .unwrap_or(HealthStatus::Unknown)
    }

    fn logs(&self) -> Option<Box<dyn dtx_core::resource::LogStream>> {
        None
    }

    fn as_any(&self) -> &dyn std::any::Any {
        self
    }

    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
        self
    }
}
```
