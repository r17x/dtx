# Phase 6.4: Configuration Import

> Import configurations from process-compose, docker-compose, and Procfile.

---

## Goal

Implement importers that convert external configuration formats into dtx resources, enabling seamless migration and interoperability.

---

## User Value

```
- Import from process-compose: dtx import process-compose.yaml
- Import from docker-compose: dtx import docker-compose.yml
- Import from Procfile: dtx import Procfile
- Automatic Nix package inference with confirmation
- Zero-config migration from existing tools
```

### 6.4.1 Import Types (2h)

**File:** `crates/dtx-core/src/import/types.rs`

```rust
//! Import format types and traits.

use std::path::Path;

use super::error::{ImportError, ImportResult};
use crate::config::schema::ResourceConfig;

/// Supported import formats.
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum ImportFormat {
    /// process-compose.yaml
    ProcessCompose,
    /// docker-compose.yml
    DockerCompose,
    /// Procfile (Heroku)
    Procfile,
    /// Auto-detect from file content
    Auto,
}

impl ImportFormat {
    /// Detect format from file path
    pub fn from_path(path: &Path) -> Self {
        let name = path.file_name()
            .and_then(|n| n.to_str())
            .unwrap_or("");

        match name.to_lowercase().as_str() {
            "process-compose.yaml" | "process-compose.yml" => Self::ProcessCompose,
            "docker-compose.yaml" | "docker-compose.yml" | "compose.yaml" | "compose.yml" => {
                Self::DockerCompose
            }
            "procfile" => Self::Procfile,
            _ => Self::Auto,
        }
    }

    /// Detect format from content
    pub fn from_content(content: &str) -> Self {
        // Check for process-compose markers
        if content.contains("processes:") && !content.contains("services:") {
            return Self::ProcessCompose;
        }

        // Check for docker-compose markers
        if content.contains("services:") {
            return Self::DockerCompose;
        }

        // Check for Procfile format (NAME: command)
        let lines: Vec<&str> = content.lines().collect();
        if !lines.is_empty() && lines.iter().all(|l| {
            l.trim().is_empty() || l.starts_with('#') || l.contains(':')
        }) {
            let non_empty: Vec<_> = lines.iter()
                .filter(|l| !l.trim().is_empty() && !l.starts_with('#'))
                .collect();
            if !non_empty.is_empty() && non_empty.iter().all(|l| {
                let parts: Vec<_> = l.splitn(2, ':').collect();
                parts.len() == 2 && !parts[0].contains(' ')
            }) {
                return Self::Procfile;
            }
        }

        Self::Auto
    }
}

/// Trait for importers.
pub trait Importer: Send + Sync {
    /// Import format.
    fn format(&self) -> ImportFormat;

    /// Import from string content.
    fn import(&self, content: &str) -> ImportResult<ImportedConfig>;

    /// Import from file.
    fn import_file(&self, path: &Path) -> ImportResult<ImportedConfig> {
        let content = std::fs::read_to_string(path)
            .map_err(|e| ImportError::Io(e.to_string()))?;
        self.import(&content)
    }
}

/// Result of import operation.
#[derive(Clone, Debug, Default)]
pub struct ImportedConfig {
    /// Project name (if detected)
    pub project_name: Option<String>,

    /// Imported resources
    pub resources: Vec<ImportedResource>,

    /// Warnings generated during import
    pub warnings: Vec<String>,
}

/// Single imported resource.
#[derive(Clone, Debug)]
pub struct ImportedResource {
    /// Resource name
    pub name: String,

    /// Resource configuration
    pub config: ResourceConfig,

    /// Inferred Nix packages (requires confirmation)
    pub inferred_nix_packages: Vec<InferredPackage>,

    /// Source line number (for error reporting)
    pub source_line: Option<usize>,
}

/// Inferred Nix package.
#[derive(Clone, Debug)]
pub struct InferredPackage {
    /// Package name in nixpkgs
    pub package: String,

    /// Confidence level
    pub confidence: Confidence,

    /// Reason for inference
    pub reason: String,
}

/// Confidence level for inference.
#[derive(Clone, Copy, Debug, PartialEq, Eq)]
pub enum Confidence {
    High,
    Medium,
    Low,
}

impl std::fmt::Display for Confidence {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            Self::High => write!(f, "high"),
            Self::Medium => write!(f, "medium"),
            Self::Low => write!(f, "low"),
        }
    }
}
```

```rust
//! Import from process-compose.yaml format.

use std::collections::HashMap;

use serde::Deserialize;

use super::error::{ImportError, ImportResult};
use super::types::{
    Confidence, ImportFormat, ImportedConfig, ImportedResource, InferredPackage, Importer,
};
use crate::config::schema::{
    DependencyConfig, HealthConfig, NixConfig, ResourceConfig, RestartConfig,
};
use crate::translation::inference::infer_nix_packages;

/// Process-compose v1.x importer.
pub struct ProcessComposeImporter;

impl Importer for ProcessComposeImporter {
    fn format(&self) -> ImportFormat {
        ImportFormat::ProcessCompose
    }

    fn import(&self, content: &str) -> ImportResult<ImportedConfig> {
        let pc: ProcessComposeFile = serde_yaml::from_str(content)
            .map_err(|e| ImportError::Parse {
                line: e.location().map(|l| l.line()).unwrap_or(0),
                message: e.to_string(),
            })?;

        let mut resources = Vec::new();
        let mut warnings = Vec::new();

        for (name, process) in pc.processes {
            match self.convert_process(&name, &process) {
                Ok(resource) => resources.push(resource),
                Err(e) => warnings.push(format!("{}: {}", name, e)),
            }
        }

        Ok(ImportedConfig {
            project_name: None,
            resources,
            warnings,
        })
    }
}

impl ProcessComposeImporter {
    fn convert_process(
        &self,
        name: &str,
        process: &ProcessComposeProcess,
    ) -> ImportResult<ImportedResource> {
        // Map depends_on
        let depends_on: Vec<DependencyConfig> = process
            .depends_on
            .as_ref()
            .map(|deps| {
                deps.iter()
                    .map(|(target, dep)| {
                        let condition = match dep.condition.as_str() {
                            "process_healthy" => "healthy",
                            "process_started" => "started",
                            "process_completed_successfully" => "completed",
                            _ => "started",
                        };
                        DependencyConfig::WithCondition(
                            [(target.clone(), condition.to_string())].into_iter().collect(),
                        )
                    })
                    .collect()
            })
            .unwrap_or_default();

        // Map readiness_probe to health
        let health = process.readiness_probe.as_ref().map(|probe| HealthConfig {
            exec: probe.exec.as_ref().map(|e| e.command.clone()),
            http: probe.http_get.as_ref().map(|h| h.path.clone()),
            tcp: None,
            interval: probe
                .period_seconds
                .map(|s| format!("{}s", s))
                .unwrap_or_else(|| "10s".to_string()),
            timeout: probe
                .timeout_seconds
                .map(|s| format!("{}s", s))
                .unwrap_or_else(|| "5s".to_string()),
            retries: probe.failure_threshold.unwrap_or(3),
            initial_delay: probe.initial_delay_seconds.map(|s| format!("{}s", s)),
        });

        // Map availability to restart
        let restart = process.availability.as_ref().map(|avail| {
            RestartConfig::Extended {
                policy: avail.restart.clone(),
                max_attempts: avail.max_restarts,
                backoff: avail.backoff_seconds.map(|s| format!("{}s", s)),
                grace_period: None,
            }
        });

        // Infer Nix packages from command
        let inferred_nix_packages = infer_nix_packages(&process.command);

        let config = ResourceConfig {
            kind: "process".to_string(),
            command: Some(process.command.clone()),
            port: process.readiness_probe.as_ref()
                .and_then(|p| p.http_get.as_ref())
                .map(|h| h.port),
            working_dir: process.working_dir.clone().map(Into::into),
            environment: process.environment.clone().unwrap_or_default(),
            depends_on,
            health,
            liveness: None,
            restart,
            nix: None, // Will be set after user confirmation
            image: None,
            runtime: None,
            model: None,
            tools: vec![],
            enabled: !process.disabled.unwrap_or(false),
        };

        Ok(ImportedResource {
            name: name.to_string(),
            config,
            inferred_nix_packages,
            source_line: None,
        })
    }
}
```

```rust
//! Import from docker-compose.yml format.

use std::collections::HashMap;

use serde::Deserialize;

use super::error::{ImportError, ImportResult};
use super::types::{
    Confidence, ImportFormat, ImportedConfig, ImportedResource, InferredPackage, Importer,
};
use crate::config::schema::{DependencyConfig, HealthConfig, ResourceConfig};
use crate::translation::inference::infer_nix_from_image;

/// Docker Compose v3.x importer.
pub struct DockerComposeImporter {
    /// Convert containers to processes where possible
    pub prefer_processes: bool,
}

impl Default for DockerComposeImporter {
    fn default() -> Self {
        Self {
            prefer_processes: true,
        }
    }
}

impl Importer for DockerComposeImporter {
    fn format(&self) -> ImportFormat {
        ImportFormat::DockerCompose
    }

    fn import(&self, content: &str) -> ImportResult<ImportedConfig> {
        let compose: ComposeFile = serde_yaml::from_str(content)
            .map_err(|e| ImportError::Parse {
                line: e.location().map(|l| l.line()).unwrap_or(0),
                message: e.to_string(),
            })?;

        let mut resources = Vec::new();
        let mut warnings = Vec::new();

        for (name, service) in compose.services.unwrap_or_default() {
            match self.convert_service(&name, &service) {
                Ok(resource) => resources.push(resource),
                Err(e) => warnings.push(format!("{}: {}", name, e)),
            }
        }

        Ok(ImportedConfig {
            project_name: compose.name,
            resources,
            warnings,
        })
    }
}

impl DockerComposeImporter {
    fn convert_service(
        &self,
        name: &str,
        service: &ComposeService,
    ) -> ImportResult<ImportedResource> {
        // Determine kind based on presence of image vs command
        let kind = if self.prefer_processes && service.image.is_none() && service.command.is_some() {
            "process"
        } else {
            "container"
        };

        // Extract command
        let command = service.command.clone().map(|c| match c {
            CommandOrList::String(s) => s,
            CommandOrList::List(l) => l.join(" "),
        });

        // Map ports (take first port as primary)
        let port = service.ports.as_ref().and_then(|ports| {
            ports.first().and_then(|p| {
                // Parse "3000:3000" or "3000" format
                let parts: Vec<&str> = p.split(':').collect();
                parts.last().and_then(|s| s.parse().ok())
            })
        });

        // Map depends_on
        let depends_on: Vec<DependencyConfig> = service
            .depends_on
            .as_ref()
            .map(|deps| match deps {
                DependsOn::List(list) => list
                    .iter()
                    .map(|d| DependencyConfig::Simple(d.clone()))
                    .collect(),
                DependsOn::Map(map) => map
                    .iter()
                    .map(|(target, cond)| {
                        let condition = match cond.condition.as_str() {
                            "service_healthy" => "healthy",
                            "service_started" => "started",
                            "service_completed_successfully" => "completed",
                            _ => "started",
                        };
                        DependencyConfig::WithCondition(
                            [(target.clone(), condition.to_string())].into_iter().collect(),
                        )
                    })
                    .collect(),
            })
            .unwrap_or_default();

        // Map healthcheck
        let health = service.healthcheck.as_ref().map(|hc| {
            let exec = hc.test.as_ref().map(|t| match t {
                HealthTest::List(l) => {
                    // ["CMD", "pg_isready"] or ["CMD-SHELL", "command"]
                    if l.len() >= 2 {
                        if l[0] == "CMD-SHELL" {
                            l[1..].join(" ")
                        } else if l[0] == "CMD" {
                            l[1..].join(" ")
                        } else {
                            l.join(" ")
                        }
                    } else {
                        l.join(" ")
                    }
                }
                HealthTest::String(s) => s.clone(),
            });

            HealthConfig {
                exec,
                http: None,
                tcp: None,
                interval: hc.interval.clone().unwrap_or_else(|| "30s".to_string()),
                timeout: hc.timeout.clone().unwrap_or_else(|| "30s".to_string()),
                retries: hc.retries.unwrap_or(3),
                initial_delay: hc.start_period.clone(),
            }
        });

        // Infer Nix packages from image
        let inferred_nix_packages = service
            .image
            .as_ref()
            .map(|img| infer_nix_from_image(img))
            .unwrap_or_default();

        let config = ResourceConfig {
            kind: kind.to_string(),
            command,
            port,
            working_dir: service.working_dir.clone().map(Into::into),
            environment: service.environment.clone().unwrap_or_default(),
            depends_on,
            health,
            liveness: None,
            restart: service.restart.clone().map(|r| crate::config::schema::RestartConfig::Simple(r)),
            nix: None,
            image: service.image.clone(),
            runtime: None,
            model: None,
            tools: vec![],
            enabled: true,
        };

        Ok(ImportedResource {
            name: name.to_string(),
            config,
            inferred_nix_packages,
            source_line: None,
        })
    }
}
```

### 6.4.5 Procfile Importer (2h)

**File:** `crates/dtx-core/src/import/procfile.rs`

```rust
//! Import from Heroku-style Procfile.

use super::error::{ImportError, ImportResult};
use super::types::{ImportFormat, ImportedConfig, ImportedResource, Importer};
use crate::config::schema::ResourceConfig;
use crate::translation::inference::infer_nix_packages;

/// Procfile importer.
pub struct ProcfileImporter;

impl Importer for ProcfileImporter {
    fn format(&self) -> ImportFormat {
        ImportFormat::Procfile
    }

    fn import(&self, content: &str) -> ImportResult<ImportedConfig> {
        let mut resources = Vec::new();
        let mut warnings = Vec::new();

        for (line_num, line) in content.lines().enumerate() {
            let line = line.trim();

            // Skip empty lines and comments
            if line.is_empty() || line.starts_with('#') {
                continue;
            }

            // Parse "name: command" format
            let parts: Vec<&str> = line.splitn(2, ':').collect();
            if parts.len() != 2 {
                warnings.push(format!("Line {}: invalid format", line_num + 1));
                continue;
            }

            let name = parts[0].trim().to_string();
            let command = parts[1].trim().to_string();

            if name.is_empty() || command.is_empty() {
                warnings.push(format!("Line {}: empty name or command", line_num + 1));
                continue;
            }

            // Infer Nix packages from command
            let inferred_nix_packages = infer_nix_packages(&command);

            let config = ResourceConfig {
                kind: "process".to_string(),
                command: Some(command),
                port: None, // Procfile doesn't specify ports
                working_dir: None,
                environment: Default::default(),
                depends_on: vec![],
                health: None,
                liveness: None,
                restart: None,
                nix: None,
                image: None,
                runtime: None,
                model: None,
                tools: vec![],
                enabled: true,
            };

            resources.push(ImportedResource {
                name,
                config,
                inferred_nix_packages,
                source_line: Some(line_num + 1),
            });
        }

        Ok(ImportedConfig {
            project_name: None,
            resources,
            warnings,
        })
    }
}
```

### 6.4.6 CLI Import Command (2h)

**File:** `crates/dtx/src/cmd/import.rs`

```rust
//! Import command implementation.

use std::path::PathBuf;

use anyhow::Result;
use clap::Args;

use dtx_core::config::loader::{ConfigLoader, ConfigLevel};
use dtx_core::import::{
    DockerComposeImporter, ImportFormat, ImportedConfig, Importer,
    ProcessComposeImporter, ProcfileImporter,
};

use crate::context::CmdContext;

#[derive(Args)]
pub struct ImportArgs {
    /// File to import.
    pub file: PathBuf,

    /// Force format (auto-detect if not specified).
    #[arg(short, long)]
    pub format: Option<String>,

    /// Skip Nix package inference.
    #[arg(long)]
    pub no_nix: bool,

    /// Skip confirmation prompts.
    #[arg(short = 'y', long)]
    pub yes: bool,

    /// Dry run (show what would be imported).
    #[arg(long)]
    pub dry_run: bool,
}

impl ImportArgs {
    pub async fn run(&self, ctx: &CmdContext) -> Result<()> {
        // Detect or use specified format
        let format = if let Some(f) = &self.format {
            match f.to_lowercase().as_str() {
                "process-compose" | "pc" => ImportFormat::ProcessCompose,
                "docker-compose" | "docker" | "compose" => ImportFormat::DockerCompose,
                "procfile" => ImportFormat::Procfile,
                _ => anyhow::bail!("Unknown format: {}", f),
            }
        } else {
            ImportFormat::from_path(&self.file)
        };

        // Read file content
        let content = std::fs::read_to_string(&self.file)?;

        // Auto-detect if needed
        let format = if format == ImportFormat::Auto {
            ImportFormat::from_content(&content)
        } else {
            format
        };

        if format == ImportFormat::Auto {
            anyhow::bail!("Could not detect format. Use --format to specify.");
        }

        // Create importer
        let imported: ImportedConfig = match format {
            ImportFormat::ProcessCompose => ProcessComposeImporter.import(&content)?,
            ImportFormat::DockerCompose => DockerComposeImporter::default().import(&content)?,
            ImportFormat::Procfile => ProcfileImporter.import(&content)?,
            ImportFormat::Auto => unreachable!(),
        };

        // Show what was imported
        println!("Importing from {:?}:", format);
        println!();

        for resource in &imported.resources {
            println!("  {} ({})", resource.name, resource.config.kind);
            if let Some(cmd) = &resource.config.command {
                println!("    command: {}", cmd);
            }
            if let Some(port) = resource.config.port {
                println!("    port: {}", port);
            }

            // Show inferred Nix packages
            if !self.no_nix && !resource.inferred_nix_packages.is_empty() {
                println!("    nix packages (inferred):");
                for pkg in &resource.inferred_nix_packages {
                    println!("      - {} ({} confidence: {})",
                        pkg.package, pkg.confidence, pkg.reason);
                }
            }
        }

        // Show warnings
        if !imported.warnings.is_empty() {
            println!();
            println!("Warnings:");
            for warning in &imported.warnings {
                println!("  ! {}", warning);
            }
        }

        if self.dry_run {
            println!();
            println!("(dry run - no changes made)");
            return Ok(());
        }

        // Confirm Nix packages if not skipped and not auto-yes
        if !self.no_nix && !self.yes {
            let has_inferred = imported.resources.iter()
                .any(|r| !r.inferred_nix_packages.is_empty());

            if has_inferred {
                println!();
                print!("Apply inferred Nix packages? [y/N] ");
                std::io::Write::flush(&mut std::io::stdout())?;

                let mut input = String::new();
                std::io::stdin().read_line(&mut input)?;

                if !input.trim().eq_ignore_ascii_case("y") {
                    println!("Skipping Nix packages.");
                    // Clear inferred packages
                    // (would need mutable access in real implementation)
                }
            }
        }

        // Load existing config and merge
        let mut loader = ConfigLoader::new();
        loader.discover_project(&std::env::current_dir()?);

        let mut config = loader.load()?;

        // Add imported resources
        for resource in imported.resources {
            let mut resource_config = resource.config;

            // Apply confirmed Nix packages
            if !self.no_nix && !resource.inferred_nix_packages.is_empty() {
                let packages: Vec<String> = resource.inferred_nix_packages
                    .iter()
                    .filter(|p| p.confidence == dtx_core::import::Confidence::High || self.yes)
                    .map(|p| p.package.clone())
                    .collect();

                if !packages.is_empty() {
                    resource_config.nix = Some(crate::config::schema::NixConfig {
                        packages,
                        expr: None,
                        shell: None,
                    });
                }
            }

            config.resources.insert(resource.name, resource_config);
        }

        // Save merged config
        loader.save_level(&config, ConfigLevel::Project)?;

        println!();
        println!("Imported {} resources to .dtx/config.yaml", imported.resources.len());

        Ok(())
    }
}
```
