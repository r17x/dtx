### 4.2.1 Orchestrator Core (4h)

```rust
// crates/dtx-core/src/orchestrator/mod.rs

pub struct Orchestrator {
    resources: HashMap<ResourceId, Arc<RwLock<dyn Resource>>>,
    dependencies: DependencyGraph,
    event_bus: Arc<EventBus>,
    middleware: MiddlewareChain,
}

impl Orchestrator {
    pub fn new(event_bus: Arc<EventBus>, middleware: MiddlewareStack) -> Self {
        Self {
            resources: HashMap::new(),
            dependencies: DependencyGraph::new(),
            event_bus,
            middleware: middleware.build(OrchestratorHandler::new()),
        }
    }

    pub fn register<R: Resource + 'static>(&mut self, resource: R) {
        let id = resource.id().clone();
        self.resources.insert(id, Arc::new(RwLock::new(resource)));
    }

    pub async fn start(&self, id: &ResourceId, ctx: &Context) -> Result<Response> {
        self.middleware.execute(
            Operation::Start { id: id.clone() },
            ctx.clone(),
        ).await
    }

    pub async fn stop(&self, id: &ResourceId, ctx: &Context) -> Result<Response> {
        self.middleware.execute(
            Operation::Stop { id: id.clone() },
            ctx.clone(),
        ).await
    }

    pub async fn start_all(&self, ctx: &Context) -> Result<Response> {
        self.middleware.execute(Operation::StartAll, ctx.clone()).await
    }

    pub async fn stop_all(&self, ctx: &Context) -> Result<Response> {
        self.middleware.execute(Operation::StopAll, ctx.clone()).await
    }
}
```

### 4.2.2 Dependency Resolution (3h)

```rust
// crates/dtx-core/src/orchestrator/graph.rs

pub struct DependencyGraph {
    edges: HashMap<ResourceId, Vec<Dependency>>,
}

pub struct Dependency {
    pub target: ResourceId,
    pub condition: DependencyCondition,
}

#[derive(Clone, Debug)]
pub enum DependencyCondition {
    Started,       // Process has started
    Healthy,       // Health check passed
    Completed,     // Exited with code 0
}

impl DependencyGraph {
    pub fn startup_order(&self) -> Result<Vec<Vec<ResourceId>>, Error> {
        // Returns layers of resources that can start in parallel
        // Layer 0: no dependencies
        // Layer 1: depends only on layer 0
        // etc.
        self.topological_sort()
    }

    pub fn shutdown_order(&self) -> Vec<Vec<ResourceId>> {
        // Reverse of startup order
        let mut order = self.startup_order().unwrap_or_default();
        order.reverse();
        order
    }

    fn topological_sort(&self) -> Result<Vec<Vec<ResourceId>>, Error> {
        // Kahn's algorithm with layer detection
        let mut in_degree: HashMap<ResourceId, usize> = HashMap::new();
        let mut layers: Vec<Vec<ResourceId>> = Vec::new();

        // Calculate in-degrees
        for id in self.edges.keys() {
            in_degree.entry(id.clone()).or_insert(0);
            for dep in self.edges.get(id).unwrap_or(&vec![]) {
                *in_degree.entry(dep.target.clone()).or_insert(0) += 1;
            }
        }

        // Find roots (no incoming edges)
        let mut current_layer: Vec<ResourceId> = in_degree
            .iter()
            .filter(|(_, &deg)| deg == 0)
            .map(|(id, _)| id.clone())
            .collect();

        while !current_layer.is_empty() {
            layers.push(current_layer.clone());
            let mut next_layer = Vec::new();

            for id in &current_layer {
                for dep in self.edges.get(id).unwrap_or(&vec![]) {
                    let degree = in_degree.get_mut(&dep.target).unwrap();
                    *degree -= 1;
                    if *degree == 0 {
                        next_layer.push(dep.target.clone());
                    }
                }
            }

            current_layer = next_layer;
        }

        // Check for cycles
        if layers.iter().map(|l| l.len()).sum::<usize>() != in_degree.len() {
            return Err(Error::DependencyCycle { cycle: vec![] });
        }

        Ok(layers)
    }
}
```

**Acceptance:**
- [ ] Topological sort with layers
- [ ] Cycle detection
- [ ] Startup/shutdown order

### 4.2.3 Orchestrator Handler (3h)

```rust
// crates/dtx-core/src/orchestrator/handler.rs

struct OrchestratorHandler {
    // Will hold reference to orchestrator internals
}

#[async_trait]
impl Handler for OrchestratorHandler {
    async fn handle(&self, op: Operation, ctx: Context) -> Result<Response> {
        match op {
            Operation::Start { id } => self.do_start(&id, &ctx).await,
            Operation::Stop { id } => self.do_stop(&id, &ctx).await,
            Operation::StartAll => self.do_start_all(&ctx).await,
            Operation::StopAll => self.do_stop_all(&ctx).await,
            Operation::Status { id } => self.do_status(&id).await,
            _ => Err(Error::InvalidState { ... }),
        }
    }
}

impl OrchestratorHandler {
    async fn do_start_all(&self, ctx: &Context) -> Result<Response> {
        let layers = self.graph.startup_order()?;

        for layer in layers {
            // Start all resources in this layer in parallel
            let futures: Vec<_> = layer.iter().map(|id| {
                self.start_with_deps(id, ctx)
            }).collect();

            let results = futures::future::join_all(futures).await;

            // Check for errors
            for result in results {
                result?;
            }
        }

        Ok(Response::ok())
    }

    async fn start_with_deps(&self, id: &ResourceId, ctx: &Context) -> Result<()> {
        // Wait for dependencies
        if let Some(deps) = self.graph.edges.get(id) {
            for dep in deps {
                self.wait_for_condition(&dep.target, &dep.condition, ctx).await?;
            }
        }

        // Start resource
        let resource = self.resources.get(id)
            .ok_or_else(|| Error::ResourceNotFound { id: id.clone() })?;

        let mut resource = resource.write().await;
        resource.start(ctx).await
    }

    async fn wait_for_condition(
        &self,
        id: &ResourceId,
        condition: &DependencyCondition,
        ctx: &Context,
    ) -> Result<()> {
        let mut subscriber = self.event_bus.subscribe_filtered(
            EventFilter::new().resource(id.clone())
        );

        loop {
            if ctx.is_expired() {
                return Err(Error::Timeout { elapsed: ctx.elapsed() });
            }

            let resource = self.resources.get(id)
                .ok_or_else(|| Error::ResourceNotFound { id: id.clone() })?;

            let state = resource.read().await.state().clone();

            match condition {
                DependencyCondition::Started => {
                    if matches!(state, ResourceState::Running { .. }) {
                        return Ok(());
                    }
                }
                DependencyCondition::Healthy => {
                    if matches!(state, ResourceState::Running { .. }) {
                        let health = resource.read().await.health().await;
                        if health.is_healthy() {
                            return Ok(());
                        }
                    }
                }
                DependencyCondition::Completed => {
                    if let ResourceState::Stopped { exit_code: Some(0), .. } = state {
                        return Ok(());
                    }
                }
            }

            // Wait for next event
            tokio::select! {
                _ = subscriber.recv() => continue,
                _ = tokio::time::sleep(Duration::from_millis(100)) => continue,
                _ = ctx.cancelled() => return Err(Error::Cancelled),
            }
        }
    }
}
```

**Acceptance:**
- [ ] Layer-based parallel startup
- [ ] Dependency condition waiting
- [ ] Timeout/cancellation support

### 4.2.4 Polling Loop (2h)

```rust
// crates/dtx-core/src/orchestrator/poll.rs

impl Orchestrator {
    pub async fn poll(&self) {
        for (id, resource) in &self.resources {
            let mut resource = resource.write().await;

            // Check if process exited
            if let Some(exit_status) = resource.try_wait() {
                // Handle exit
                self.handle_exit(id, exit_status).await;
            }

            // Collect logs
            if let Some(logs) = resource.logs() {
                while let Some(entry) = logs.try_recv() {
                    self.event_bus.publish(LifecycleEvent::log(id, entry.stream, entry.line));
                }
            }

            // Check restart
            if resource.should_restart() {
                self.schedule_restart(id).await;
            }
        }
    }

    pub fn run_poll_loop(self: Arc<Self>) -> JoinHandle<()> {
        tokio::spawn(async move {
            let mut interval = tokio::time::interval(Duration::from_millis(100));
            loop {
                interval.tick().await;
                self.poll().await;
            }
        })
    }
}
```
