# Phase 8.1: Container Backend

> Docker/Podman support as a Resource backend.

---

## Goal

Implement containers as Resources, enabling container orchestration alongside native processes with full lifecycle management.

---

## User Value

```
- Mix processes and containers in same project
- Use existing Docker/Podman images
- Consistent management API across resource types
- Volume and network management
- Container-specific health checks
```

---

## Dependencies

```
Phase 1.1 (Resource Trait) - Container implements Resource
Phase 1.2 (EventBus v2) - Lifecycle events
Phase 4.1 (Process Resource) - Reference implementation

```

```rust
/// Configuration for a container resource.
#[derive(Clone, Debug, Serialize, Deserialize, JsonSchema)]
pub struct ContainerConfig {

    #[serde(default)]
    pub working_dir: Option<String>,

    /// Environment variables.
```

### 8.1.2 Container Runtime Trait (3h)

**File:** `crates/dtx-container/src/runtime.rs`

```rust
use async_trait::async_trait;
use futures::Stream;
use std::pin::Pin;

use crate::config::ContainerConfig;
use dtx_core::resource::{LogEntry, HealthStatus};
use dtx_core::error::Result;

/// Container information returned by inspect.
#[derive(Clone, Debug)]
pub struct ContainerInfo {
    pub id: String,
    pub name: String,
    pub image: String,
    pub state: ContainerState,
    pub created_at: chrono::DateTime<chrono::Utc>,
    pub started_at: Option<chrono::DateTime<chrono::Utc>>,
    pub finished_at: Option<chrono::DateTime<chrono::Utc>>,
    pub exit_code: Option<i32>,
    pub pid: Option<u32>,
    pub ports: Vec<PortInfo>,
    pub mounts: Vec<MountInfo>,
}

#[derive(Clone, Debug)]
pub enum ContainerState {
    Created,
    Running,
    Paused,
    Restarting,
    Exited,
    Dead,
}

#[derive(Clone, Debug)]
pub struct PortInfo {
    pub container_port: u16,
    pub host_port: Option<u16>,
    pub host_ip: Option<String>,
}

#[derive(Clone, Debug)]
pub struct MountInfo {
    pub source: String,
    pub destination: String,
    pub mode: String,
}

/// Abstract container runtime interface.
///
/// Implementations exist for Docker and Podman.
#[async_trait]
pub trait ContainerRuntime: Send + Sync {
    /// Runtime name (docker, podman).
    fn name(&self) -> &str;

    /// Check if runtime is available.
    async fn is_available(&self) -> bool;

    /// Pull an image.
    async fn pull(&self, image: &str) -> Result<()>;

    /// Check if image exists locally.
    async fn image_exists(&self, image: &str) -> Result<bool>;

    /// Create a container.
    async fn create(&self, config: &ContainerConfig) -> Result<String>;

    /// Start a container.
    async fn start(&self, id: &str) -> Result<()>;

    /// Stop a container with timeout.
    async fn stop(&self, id: &str, timeout: std::time::Duration) -> Result<()>;

    /// Kill a container.
    async fn kill(&self, id: &str, signal: Option<&str>) -> Result<()>;

    /// Remove a container.
    async fn remove(&self, id: &str, force: bool) -> Result<()>;

    /// Get container logs as a stream.
    async fn logs(
        &self,
        id: &str,
        follow: bool,
        tail: Option<usize>,
    ) -> Result<Pin<Box<dyn Stream<Item = LogEntry> + Send>>>;

    /// Inspect a container.
    async fn inspect(&self, id: &str) -> Result<ContainerInfo>;

    /// Execute a command in a running container.
    async fn exec(
        &self,
        id: &str,
        command: &[String],
        interactive: bool,
    ) -> Result<ExecResult>;

    /// Wait for container to exit.
    async fn wait(&self, id: &str) -> Result<i32>;

    /// Get container health status.
    async fn health(&self, id: &str) -> Result<HealthStatus>;

    /// List containers (optionally filtered by labels).
    async fn list(&self, labels: Option<&[(&str, &str)]>) -> Result<Vec<ContainerInfo>>;

    /// Create a network.
    async fn network_create(&self, name: &str) -> Result<String>;

    /// Remove a network.
    async fn network_remove(&self, name: &str) -> Result<()>;

    /// Connect container to network.
    async fn network_connect(&self, network: &str, container: &str) -> Result<()>;

    /// Create a volume.
    async fn volume_create(&self, name: &str) -> Result<String>;

    /// Remove a volume.
    async fn volume_remove(&self, name: &str, force: bool) -> Result<()>;
}

/// Result of exec command.
#[derive(Clone, Debug)]
pub struct ExecResult {
    pub exit_code: i32,
    pub stdout: String,
    pub stderr: String,
}

/// Select the best available runtime.
pub async fn detect_runtime() -> Option<Box<dyn ContainerRuntime>> {
    // Try Docker first
    let docker = super::docker::DockerRuntime::new();
    if docker.is_available().await {
        return Some(Box::new(docker));
    }

    // Try Podman
    let podman = super::podman::PodmanRuntime::new();
    if podman.is_available().await {
        return Some(Box::new(podman));
    }

    None
}
```

### 8.1.3 Docker Runtime Implementation (4h)

**File:** `crates/dtx-container/src/docker.rs`

```rust
use async_trait::async_trait;
use bollard::Docker;
use bollard::container::{
    Config, CreateContainerOptions, StartContainerOptions, StopContainerOptions,
    RemoveContainerOptions, LogsOptions, WaitContainerOptions,
};
use bollard::exec::{CreateExecOptions, StartExecResults};
use bollard::image::CreateImageOptions;
use bollard::network::{CreateNetworkOptions, ConnectNetworkOptions};
use bollard::volume::CreateVolumeOptions;
use futures::{Stream, StreamExt, TryStreamExt};
use std::collections::HashMap;
use std::pin::Pin;

use super::runtime::{ContainerRuntime, ContainerInfo, ContainerState, ExecResult};
use super::config::ContainerConfig;
use dtx_core::resource::{LogEntry, LogStreamKind, HealthStatus};
use dtx_core::error::{Error, Result};

/// Docker runtime using bollard.
pub struct DockerRuntime {
    client: Docker,
}

impl DockerRuntime {
    pub fn new() -> Self {
        let client = Docker::connect_with_local_defaults()
            .expect("Failed to connect to Docker");
        Self { client }
    }

    pub fn with_socket(socket: &str) -> Result<Self> {
        let client = Docker::connect_with_socket(socket, 120, bollard::API_DEFAULT_VERSION)?;
        Ok(Self { client })
    }

    fn build_container_config(&self, config: &ContainerConfig) -> Config<String> {
        let mut env: Vec<String> = config.environment
            .iter()
            .map(|(k, v)| format!("{}={}", k, v))
            .collect();

        let exposed_ports: HashMap<String, HashMap<(), ()>> = config.ports
            .iter()
            .map(|p| (format!("{}/{}", p.container, p.protocol.as_str()), HashMap::new()))
            .collect();

        let host_config = bollard::models::HostConfig {
            port_bindings: Some(
                config.ports
                    .iter()
                    .map(|p| {
                        let binding = bollard::models::PortBinding {
                            host_ip: p.host_ip.clone(),
                            host_port: Some(p.host.to_string()),
                        };
                        (format!("{}/{}", p.container, p.protocol.as_str()), Some(vec![binding]))
                    })
                    .collect()
            ),
            binds: Some(
                config.volumes
                    .iter()
                    .map(|v| {
                        let ro = if v.read_only { ":ro" } else { "" };
                        format!("{}:{}{}", v.source, v.target, ro)
                    })
                    .collect()
            ),
            memory: config.resources.as_ref().and_then(|r| parse_memory(&r.memory)),
            nano_cpus: config.resources.as_ref().and_then(|r| parse_cpus(&r.cpus)),
            privileged: Some(config.privileged),
            restart_policy: Some(config.restart.to_bollard()),
            network_mode: Some(config.network.mode.to_string()),
            ..Default::default()
        };

        let healthcheck = config.health_check.as_ref().map(|h| {
            bollard::models::HealthConfig {
                test: Some(h.test.clone()),
                interval: Some(h.interval.as_nanos() as i64),
                timeout: Some(h.timeout.as_nanos() as i64),
                retries: Some(h.retries as i64),
                start_period: Some(h.start_period.as_nanos() as i64),
            }
        });

        Config {
            image: Some(config.image.clone()),
            cmd: config.command.clone(),
            entrypoint: config.args.clone(),
            working_dir: config.working_dir.clone(),
            env: Some(env),
            exposed_ports: Some(exposed_ports),
            host_config: Some(host_config),
            healthcheck,
            labels: Some(config.labels.clone()),
            user: config.user.clone(),
            hostname: config.hostname.clone(),
            ..Default::default()
        }
    }
}

#[async_trait]
impl ContainerRuntime for DockerRuntime {
    fn name(&self) -> &str { "docker" }

    async fn is_available(&self) -> bool {
        self.client.ping().await.is_ok()
    }

    async fn pull(&self, image: &str) -> Result<()> {
        let options = CreateImageOptions {
            from_image: image,
            ..Default::default()
        };

        let mut stream = self.client.create_image(Some(options), None, None);
        while let Some(result) = stream.next().await {
            result.map_err(|e| Error::backend(e.to_string()))?;
        }
        Ok(())
    }

    async fn image_exists(&self, image: &str) -> Result<bool> {
        match self.client.inspect_image(image).await {
            Ok(_) => Ok(true),
            Err(bollard::errors::Error::DockerResponseServerError { status_code: 404, .. }) => Ok(false),
            Err(e) => Err(Error::backend(e.to_string())),
        }
    }

    async fn create(&self, config: &ContainerConfig) -> Result<String> {
        let container_config = self.build_container_config(config);
        let options = CreateContainerOptions {
            name: config.id.as_str(),
            platform: None,
        };

        let response = self.client
            .create_container(Some(options), container_config)
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(response.id)
    }

    async fn start(&self, id: &str) -> Result<()> {
        self.client
            .start_container(id, None::<StartContainerOptions<String>>)
            .await
            .map_err(|e| Error::backend(e.to_string()))
    }

    async fn stop(&self, id: &str, timeout: std::time::Duration) -> Result<()> {
        let options = StopContainerOptions {
            t: timeout.as_secs() as i64,
        };
        self.client
            .stop_container(id, Some(options))
            .await
            .map_err(|e| Error::backend(e.to_string()))
    }

    async fn kill(&self, id: &str, signal: Option<&str>) -> Result<()> {
        let options = bollard::container::KillContainerOptions {
            signal: signal.unwrap_or("SIGKILL"),
        };
        self.client
            .kill_container(id, Some(options))
            .await
            .map_err(|e| Error::backend(e.to_string()))
    }

    async fn remove(&self, id: &str, force: bool) -> Result<()> {
        let options = RemoveContainerOptions {
            force,
            v: true, // Remove volumes
            ..Default::default()
        };
        self.client
            .remove_container(id, Some(options))
            .await
            .map_err(|e| Error::backend(e.to_string()))
    }

    async fn logs(
        &self,
        id: &str,
        follow: bool,
        tail: Option<usize>,
    ) -> Result<Pin<Box<dyn Stream<Item = LogEntry> + Send>>> {
        let options = LogsOptions::<String> {
            stdout: true,
            stderr: true,
            follow,
            tail: tail.map(|t| t.to_string()).unwrap_or_else(|| "all".to_string()),
            timestamps: true,
            ..Default::default()
        };

        let stream = self.client
            .logs(id, Some(options))
            .map(|result| {
                match result {
                    Ok(output) => {
                        let (stream, line) = match output {
                            bollard::container::LogOutput::StdOut { message } => {
                                (LogStreamKind::Stdout, String::from_utf8_lossy(&message).to_string())
                            }
                            bollard::container::LogOutput::StdErr { message } => {
                                (LogStreamKind::Stderr, String::from_utf8_lossy(&message).to_string())
                            }
                            _ => return LogEntry {
                                timestamp: chrono::Utc::now(),
                                stream: LogStreamKind::Stdout,
                                line: String::new(),
                            },
                        };
                        LogEntry {
                            timestamp: chrono::Utc::now(),
                            stream,
                            line,
                        }
                    }
                    Err(_) => LogEntry {
                        timestamp: chrono::Utc::now(),
                        stream: LogStreamKind::Stderr,
                        line: String::new(),
                    },
                }
            });

        Ok(Box::pin(stream))
    }

    async fn inspect(&self, id: &str) -> Result<ContainerInfo> {
        let info = self.client
            .inspect_container(id, None)
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        let state = match info.state.as_ref().and_then(|s| s.status) {
            Some(bollard::models::ContainerStateStatusEnum::RUNNING) => ContainerState::Running,
            Some(bollard::models::ContainerStateStatusEnum::EXITED) => ContainerState::Exited,
            Some(bollard::models::ContainerStateStatusEnum::PAUSED) => ContainerState::Paused,
            Some(bollard::models::ContainerStateStatusEnum::RESTARTING) => ContainerState::Restarting,
            Some(bollard::models::ContainerStateStatusEnum::DEAD) => ContainerState::Dead,
            _ => ContainerState::Created,
        };

        Ok(ContainerInfo {
            id: info.id.unwrap_or_default(),
            name: info.name.unwrap_or_default(),
            image: info.config.and_then(|c| c.image).unwrap_or_default(),
            state,
            created_at: info.created.and_then(|s| s.parse().ok()).unwrap_or_else(chrono::Utc::now),
            started_at: info.state.as_ref().and_then(|s| s.started_at.as_ref()).and_then(|s| s.parse().ok()),
            finished_at: info.state.as_ref().and_then(|s| s.finished_at.as_ref()).and_then(|s| s.parse().ok()),
            exit_code: info.state.as_ref().and_then(|s| s.exit_code).map(|c| c as i32),
            pid: info.state.as_ref().and_then(|s| s.pid).map(|p| p as u32),
            ports: vec![],
            mounts: vec![],
        })
    }

    async fn exec(
        &self,
        id: &str,
        command: &[String],
        interactive: bool,
    ) -> Result<ExecResult> {
        let options = CreateExecOptions {
            cmd: Some(command.to_vec()),
            attach_stdout: Some(true),
            attach_stderr: Some(true),
            tty: Some(interactive),
            ..Default::default()
        };

        let exec = self.client
            .create_exec(id, options)
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        let output = self.client
            .start_exec(&exec.id, None)
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        let mut stdout = String::new();
        let mut stderr = String::new();

        if let StartExecResults::Attached { mut output, .. } = output {
            while let Some(Ok(msg)) = output.next().await {
                match msg {
                    bollard::container::LogOutput::StdOut { message } => {
                        stdout.push_str(&String::from_utf8_lossy(&message));
                    }
                    bollard::container::LogOutput::StdErr { message } => {
                        stderr.push_str(&String::from_utf8_lossy(&message));
                    }
                    _ => {}
                }
            }
        }

        let inspect = self.client
            .inspect_exec(&exec.id)
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(ExecResult {
            exit_code: inspect.exit_code.unwrap_or(0) as i32,
            stdout,
            stderr,
        })
    }

    async fn wait(&self, id: &str) -> Result<i32> {
        let options = WaitContainerOptions {
            condition: "not-running",
        };

        let mut stream = self.client.wait_container(id, Some(options));
        if let Some(result) = stream.next().await {
            let response = result.map_err(|e| Error::backend(e.to_string()))?;
            Ok(response.status_code as i32)
        } else {
            Err(Error::backend("Container wait failed"))
        }
    }

    async fn health(&self, id: &str) -> Result<HealthStatus> {
        let info = self.inspect(id).await?;
        match info.state {
            ContainerState::Running => Ok(HealthStatus::Healthy),
            ContainerState::Exited => Ok(HealthStatus::Unhealthy {
                reason: format!("Container exited with code {:?}", info.exit_code),
            }),
            _ => Ok(HealthStatus::Unknown),
        }
    }

    async fn list(&self, labels: Option<&[(&str, &str)]>) -> Result<Vec<ContainerInfo>> {
        let mut filters = HashMap::new();
        if let Some(labels) = labels {
            filters.insert("label".to_string(), labels.iter().map(|(k, v)| format!("{}={}", k, v)).collect());
        }

        let options = bollard::container::ListContainersOptions {
            all: true,
            filters,
            ..Default::default()
        };

        let containers = self.client
            .list_containers(Some(options))
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        let mut results = Vec::new();
        for c in containers {
            if let Some(id) = c.id {
                if let Ok(info) = self.inspect(&id).await {
                    results.push(info);
                }
            }
        }
        Ok(results)
    }

    async fn network_create(&self, name: &str) -> Result<String> {
        let options = CreateNetworkOptions {
            name,

fn parse_memory(mem: &Option<String>) -> Option<i64> {
    mem.as_ref().and_then(|m| {
        let m = m.to_lowercase();
        if m.ends_with('g') {
            m.trim_end_matches('g').parse::<i64>().ok().map(|v| v * 1024 * 1024 * 1024)
        } else if m.ends_with('m') {
            m.trim_end_matches('m').parse::<i64>().ok().map(|v| v * 1024 * 1024)
        } else if m.ends_with('k') {
            m.trim_end_matches('k').parse::<i64>().ok().map(|v| v * 1024)
        } else {
            m.parse().ok()
        }
    })
}

fn parse_cpus(cpus: &Option<String>) -> Option<i64> {
    cpus.as_ref().and_then(|c| {
        c.parse::<f64>().ok().map(|v| (v * 1_000_000_000.0) as i64)
    })
}
```

```rust
use async_trait::async_trait;
use futures::{Stream, StreamExt};
use std::pin::Pin;
use std::process::Stdio;
use tokio::process::Command;

use super::runtime::{ContainerRuntime, ContainerInfo, ContainerState, ExecResult};
use super::config::ContainerConfig;
use dtx_core::resource::{LogEntry, LogStreamKind, HealthStatus};
use dtx_core::error::{Error, Result};

/// Podman runtime using CLI commands.
///
/// Podman is API-compatible with Docker for most operations,
/// but we use the CLI for simplicity and broader compatibility.
pub struct PodmanRuntime {
    socket: Option<String>,
}

impl PodmanRuntime {
    pub fn new() -> Self {
        Self { socket: None }
    }

    pub fn with_socket(socket: String) -> Self {
        Self { socket: Some(socket) }
    }

    async fn run_command(&self, args: &[&str]) -> Result<String> {
        let mut cmd = Command::new("podman");

        if let Some(socket) = &self.socket {
            cmd.arg("--url").arg(socket);
        }

        let output = cmd
            .args(args)
            .stdout(Stdio::piped())
            .stderr(Stdio::piped())
            .output()
            .await
            .map_err(|e| Error::backend(format!("Failed to run podman: {}", e)))?;

        if output.status.success() {
            Ok(String::from_utf8_lossy(&output.stdout).to_string())
        } else {
            Err(Error::backend(format!(
                "podman {} failed: {}",
                args.join(" "),
                String::from_utf8_lossy(&output.stderr)
            )))
        }
    }

    fn build_run_args(&self, config: &ContainerConfig) -> Vec<String> {
        let mut args = vec!["create".to_string(), "--name".to_string(), config.id.as_str().to_string()];

        // Environment
        for (k, v) in &config.environment {
            args.push("-e".to_string());
            args.push(format!("{}={}", k, v));
        }

        // Ports
        for p in &config.ports {
            args.push("-p".to_string());
            args.push(format!("{}:{}", p.host, p.container));
        }

        // Volumes
        for v in &config.volumes {
            args.push("-v".to_string());
            let ro = if v.read_only { ":ro" } else { "" };
            args.push(format!("{}:{}{}", v.source, v.target, ro));
        }

        // Resources
        if let Some(resources) = &config.resources {
            if let Some(memory) = &resources.memory {
                args.push("--memory".to_string());
                args.push(memory.clone());
            }
            if let Some(cpus) = &resources.cpus {
                args.push("--cpus".to_string());
                args.push(cpus.clone());
            }
        }

        // Privileged
        if config.privileged {
            args.push("--privileged".to_string());
        }

        // User
        if let Some(user) = &config.user {
            args.push("--user".to_string());
            args.push(user.clone());
        }

        // Labels
        for (k, v) in &config.labels {
            args.push("--label".to_string());
            args.push(format!("{}={}", k, v));
        }

        // Image
        args.push(config.image.clone());

        // Command
        if let Some(cmd) = &config.command {
            args.extend(cmd.clone());
        }

        args
    }
}

impl ContainerRuntime for PodmanRuntime {

            self.run_command(&["rm", "-f", id]).await?;

        id: &str,
        follow: bool,
        tail: Option<usize>,
    ) -> Result<Pin<Box<dyn Stream<Item = LogEntry> + Send>>> {
        let mut args = vec!["logs"];
        if follow {
            args.push("-f");
        }
        if let Some(t) = tail {
            args.push("--tail");
            // Need to handle this differently for streaming
        }
        args.push(id);

        // For now, return a simple stream
        let output = self.run_command(&args).await?;
        let lines: Vec<LogEntry> = output
            .lines()
            .map(|line| LogEntry {
                timestamp: chrono::Utc::now(),
                stream: LogStreamKind::Stdout,
                line: line.to_string(),
            })
            .collect();

        let stream = futures::stream::iter(lines);
        Ok(Box::pin(stream))
    }

    async fn inspect(&self, id: &str) -> Result<ContainerInfo> {
        let output = self.run_command(&["inspect", "--format", "json", id]).await?;
        let info: Vec<serde_json::Value> = serde_json::from_str(&output)
            .map_err(|e| Error::backend(format!("Failed to parse inspect output: {}", e)))?;

        let container = info.first().ok_or_else(|| Error::backend("Container not found"))?;

        let state_str = container["State"]["Status"].as_str().unwrap_or("unknown");
        let state = match state_str {
            "running" => ContainerState::Running,
            "exited" => ContainerState::Exited,
            "paused" => ContainerState::Paused,
            _ => ContainerState::Created,
        };

        Ok(ContainerInfo {
            id: container["Id"].as_str().unwrap_or_default().to_string(),
            name: container["Name"].as_str().unwrap_or_default().to_string(),
            image: container["ImageName"].as_str().unwrap_or_default().to_string(),
            state,
            created_at: chrono::Utc::now(),
            started_at: None,
            finished_at: None,
            exit_code: container["State"]["ExitCode"].as_i64().map(|c| c as i32),
            pid: container["State"]["Pid"].as_u64().map(|p| p as u32),
            ports: vec![],
            mounts: vec![],
        })
    }

    async fn exec(
        &self,
        id: &str,
        command: &[String],
        _interactive: bool,
    ) -> Result<ExecResult> {
        let mut args = vec!["exec", id];
        let cmd_strs: Vec<&str> = command.iter().map(|s| s.as_str()).collect();
        args.extend(cmd_strs);

        let output = Command::new("podman")
            .args(&args)
            .output()
            .await
            .map_err(|e| Error::backend(e.to_string()))?;

        Ok(ExecResult {
            exit_code: output.status.code().unwrap_or(-1),
            stdout: String::from_utf8_lossy(&output.stdout).to_string(),
            stderr: String::from_utf8_lossy(&output.stderr).to_string(),
        })
    }

    async fn wait(&self, id: &str) -> Result<i32> {
        let output = self.run_command(&["wait", id]).await?;
        output.trim().parse().map_err(|_| Error::backend("Invalid exit code"))
    }

    async fn health(&self, id: &str) -> Result<HealthStatus> {
        let info = self.inspect(id).await?;
        match info.state {
            ContainerState::Running => Ok(HealthStatus::Healthy),
            ContainerState::Exited => Ok(HealthStatus::Unhealthy {
                reason: format!("Container exited with code {:?}", info.exit_code),
            }),
            _ => Ok(HealthStatus::Unknown),
        }
    }

    async fn list(&self, labels: Option<&[(&str, &str)]>) -> Result<Vec<ContainerInfo>> {

                args.push("--filter");
                args.push(&format!("label={}={}", k, v));

        let mut results = Vec::new();
        for c in containers {
            if let Some(id) = c["Id"].as_str() {
                if let Ok(info) = self.inspect(id).await {
                    results.push(info);
                }
            }
        }
        Ok(results)
    }

    async fn network_create(&self, name: &str) -> Result<String> {
        self.run_command(&["network", "create", name]).await
    }

    async fn network_remove(&self, name: &str) -> Result<()> {
        self.run_command(&["network", "rm", name]).await?;
        Ok(())
    }

    async fn network_connect(&self, network: &str, container: &str) -> Result<()> {
        self.run_command(&["network", "connect", network, container]).await?;
        Ok(())

    async fn volume_create(&self, name: &str) -> Result<String> {

        } else {
            self.run_command(&["volume", "rm", name]).await?;
        }
        Ok(())
    }
```

```rust
use async_trait::async_trait;
use std::sync::Arc;

use crate::config::{ContainerConfig, PullPolicy};
use crate::runtime::ContainerRuntime;
use dtx_core::{
    resource::{Resource, ResourceId, ResourceKind, ResourceState, HealthStatus, LogStream, LogEntry},
    events::{EventBus, LifecycleEvent},
    context::Context,
    error::{Error, Result},
};

/// Container resource implementing the Resource trait.
pub struct Container {
    config: ContainerConfig,
    state: ResourceState,
    container_id: Option<String>,
    runtime: Arc<dyn ContainerRuntime>,
    event_bus: Arc<EventBus>,
}

impl Container {
    pub fn new(
        config: ContainerConfig,
        runtime: Arc<dyn ContainerRuntime>,
        event_bus: Arc<EventBus>,
    ) -> Self {
        Self {
            config,
            state: ResourceState::Pending,
            container_id: None,
            runtime,
            event_bus,
        }
    }

    async fn pull_image_if_needed(&self) -> Result<()> {
        match self.config.pull_policy {
            PullPolicy::Always => {
                tracing::info!(image = %self.config.image, "Pulling image (always)");
                self.runtime.pull(&self.config.image).await?;
            }
            PullPolicy::IfNotPresent => {
                if !self.runtime.image_exists(&self.config.image).await? {
                    tracing::info!(image = %self.config.image, "Pulling image (not present)");
                    self.runtime.pull(&self.config.image).await?;
                }
            }
            PullPolicy::Never => {
                if !self.runtime.image_exists(&self.config.image).await? {
                    return Err(Error::backend(format!(
                        "Image {} not found and pull_policy is Never",
                        self.config.image
                    )));
                }
            }
        }
        Ok(())
    }

    async fn wait_for_healthy(&self, ctx: &Context) -> Result<()> {
        if let Some(health_check) = &self.config.health_check {
            let start = std::time::Instant::now();
            let timeout = health_check.start_period + (health_check.interval * health_check.retries);

            loop {
                if start.elapsed() > timeout {
                    return Err(Error::timeout(timeout));
                }

                if ctx.is_expired() {
                    return Err(Error::timeout(ctx.elapsed()));
                }

                if let Some(id) = &self.container_id {
                    match self.runtime.health(id).await {
                        Ok(HealthStatus::Healthy) => return Ok(()),
                        _ => {}
                    }
                }

                tokio::time::sleep(health_check.interval).await;
            }
        }
        Ok(())
    }
}

#[async_trait]
impl Resource for Container {
    fn id(&self) -> &ResourceId {
        &self.config.id
    }

    fn kind(&self) -> ResourceKind {
        ResourceKind::Container
    }

    fn state(&self) -> &ResourceState {
        &self.state
    }

    async fn start(&mut self, ctx: &Context) -> Result<()> {
        let started_at = chrono::Utc::now();
        self.state = ResourceState::Starting { started_at };
        self.event_bus.publish(LifecycleEvent::starting(
            &self.config.id,
            ResourceKind::Container,
        ));

        // Pull image if needed
        self.pull_image_if_needed().await?;

        // Remove existing container if any
        if let Ok(info) = self.runtime.inspect(self.config.id.as_str()).await {
            tracing::info!(id = %self.config.id, "Removing existing container");
            self.runtime.remove(&info.id, true).await?;
        }

        // Create container
        tracing::info!(id = %self.config.id, image = %self.config.image, "Creating container");
        let container_id = self.runtime.create(&self.config).await?;
        self.container_id = Some(container_id.clone());

        // Start container
        tracing::info!(id = %self.config.id, container_id = %container_id, "Starting container");
        self.runtime.start(&container_id).await?;

        // Wait for health if configured
        self.wait_for_healthy(ctx).await?;

        // Get PID from inspect
        let info = self.runtime.inspect(&container_id).await?;
        let pid = info.pid;

        self.state = ResourceState::Running { pid, started_at };
        self.event_bus.publish(LifecycleEvent::running(
            &self.config.id,
            ResourceKind::Container,
            pid,
        ));

        Ok(())
    }

    async fn stop(&mut self, _ctx: &Context) -> Result<()> {
        if let Some(id) = &self.container_id {
            let started_at = match &self.state {
                ResourceState::Running { started_at, .. } => *started_at,
                _ => chrono::Utc::now(),
            };
            let stopping_at = chrono::Utc::now();

            self.state = ResourceState::Stopping { started_at, stopping_at };
            self.event_bus.publish(LifecycleEvent::stopping(&self.config.id));

            tracing::info!(id = %self.config.id, container_id = %id, "Stopping container");
            let timeout = self.config.restart.timeout();

            match tokio::time::timeout(timeout, self.runtime.stop(id, timeout)).await {
                Ok(Ok(())) => {}
                Ok(Err(e)) => {
                    tracing::warn!(error = %e, "Stop failed, killing container");
                    self.runtime.kill(id, None).await?;
                }
                Err(_) => {
                    tracing::warn!("Stop timeout, killing container");
                    self.runtime.kill(id, None).await?;
                }
            }

            // Get exit code
            let info = self.runtime.inspect(id).await.ok();
            let exit_code = info.and_then(|i| i.exit_code);

            self.state = ResourceState::Stopped {
                exit_code,
                started_at,
                stopped_at: chrono::Utc::now(),
            };

            self.event_bus.publish(LifecycleEvent::stopped(
                &self.config.id,
                exit_code,
            ));
        }
        Ok(())
    }

    async fn kill(&mut self, _ctx: &Context) -> Result<()> {
        if let Some(id) = &self.container_id {
            tracing::info!(id = %self.config.id, container_id = %id, "Killing container");
            self.runtime.kill(id, Some("SIGKILL")).await?;

            self.state = ResourceState::Stopped {
                exit_code: Some(-9),
                started_at: chrono::Utc::now(),
                stopped_at: chrono::Utc::now(),
            };
        }
        Ok(())
    }

    async fn health(&self) -> HealthStatus {
        if let Some(id) = &self.container_id {
            self.runtime.health(id).await.unwrap_or(HealthStatus::Unknown)
        } else {
            HealthStatus::Unknown
        }
    }

    fn logs(&self) -> Option<Box<dyn LogStream>> {
        // TODO: Implement log stream wrapper
        None
    }

    fn as_any(&self) -> &dyn std::any::Any {
        self
    }

    fn as_any_mut(&mut self) -> &mut dyn std::any::Any {
        self
    }
}
```

```rust
use dtx_core::translation::{Translator, TranslatorMetadata, TranslationError, TranslationResult};
use dtx_process::config::ProcessConfig;
use crate::config::{ContainerConfig, PortMapping, Protocol, RestartPolicy};

/// Translator from Process to Container configuration.
pub struct ProcessToContainerTranslator;

impl Translator<ProcessConfig, ContainerConfig> for ProcessToContainerTranslator {
    fn translate(&self, process: &ProcessConfig) -> TranslationResult<ContainerConfig> {
        // Infer image from command
        let image = infer_image_from_command(&process.command)?;

        Ok(ContainerConfig {
            id: process.id.clone(),
            image,
            command: Some(vec![process.command.as_str().to_string()]),
            args: None,
            working_dir: process.working_dir.as_ref().map(|p| p.display().to_string()),
            environment: process.environment.clone(),
            ports: process.port.map(|p| vec![PortMapping {
                host: p,
                container: p,
                protocol: Protocol::Tcp,
                host_ip: None,
            }]).unwrap_or_default(),
            restart: match &process.restart {
                dtx_process::config::RestartPolicy::Always { .. } => RestartPolicy::Always,
                dtx_process::config::RestartPolicy::OnFailure { .. } => RestartPolicy::OnFailure,
                dtx_process::config::RestartPolicy::No => RestartPolicy::No,
            },
            health_check: process.readiness_probe.as_ref().map(|p| {
                crate::config::ContainerHealthCheck {
                    test: vec!["CMD".to_string(), p.command().unwrap_or_default()],
                    interval: p.settings().period,
                    timeout: p.settings().timeout,
                    retries: p.settings().failure_threshold,
                    start_period: p.settings().initial_delay,
                }
            }),
            ..Default::default()
        })
    }

    fn reverse(&self, container: &ContainerConfig) -> TranslationResult<ProcessConfig> {
        let command = container.command.as_ref()
            .and_then(|c| c.first())
            .ok_or_else(|| TranslationError::missing_field("command"))?
            .clone();

        Ok(ProcessConfig {
            id: container.id.clone(),
            command: command.into(),
            environment: container.environment.clone(),
            port: container.ports.first().map(|p| p.host),
            ..Default::default()
        })
    }

    fn supports_reverse(&self) -> bool {
        true
    }

    fn metadata(&self) -> TranslatorMetadata {
        TranslatorMetadata::new("process-to-container")
            .with_description("Convert native process to container")
            .lossy()
    }
}

fn infer_image_from_command(command: &str) -> TranslationResult<String> {
    let parts: Vec<&str> = command.split_whitespace().collect();
    let binary = parts.first().ok_or_else(|| TranslationError::missing_field("command"))?;

    // Common mappings
    let image = match *binary {
        "node" | "npm" | "yarn" | "pnpm" => "node:20-alpine",
        "python" | "python3" | "pip" => "python:3.12-alpine",
        "ruby" | "gem" | "bundle" => "ruby:3.3-alpine",
        "go" | "gofmt" => "golang:1.22-alpine",
        "cargo" | "rustc" => "rust:1.75-alpine",
        "java" | "javac" | "mvn" | "gradle" => "eclipse-temurin:21-alpine",
        "php" | "composer" => "php:8.3-alpine",
        "postgres" | "psql" | "pg_isready" => "postgres:16-alpine",
        "redis-server" | "redis-cli" => "redis:7-alpine",
        "mysql" | "mysqld" => "mysql:8",
        "mongo" | "mongod" => "mongo:7",
        _ => return Err(TranslationError::untranslatable(
            "command",
            format!("Cannot infer image for: {}", binary),
        )),
    };

    Ok(image.to_string())
}
```
