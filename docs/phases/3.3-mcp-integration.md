```rust
use serde::{Deserialize, Serialize};

/// MCP Resource
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct Resource {
    /// URI identifying the resource
    pub uri: String,
    /// Human-readable name
    pub name: String,
    /// Optional description
    #[serde(skip_serializing_if = "Option::is_none")]
    pub description: Option<String>,
    /// MIME type
    #[serde(skip_serializing_if = "Option::is_none")]
    pub mime_type: Option<String>,
}

/// Resource content
#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct ResourceContent {
    pub uri: String,
    #[serde(skip_serializing_if = "Option::is_none")]
    pub mime_type: Option<String>,
    #[serde(flatten)]
    pub content: ResourceContentType,
}

#[derive(Clone, Debug, Serialize, Deserialize)]
#[serde(untagged)]
pub enum ResourceContentType {
    Text { text: String },
    Blob { blob: String }, // base64 encoded
}

/// List resources result
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ListResourcesResult {
    pub resources: Vec<Resource>,
}

/// Read resource params
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ReadResourceParams {
    pub uri: String,
}

/// Read resource result
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct ReadResourceResult {
    pub contents: Vec<ResourceContent>,
}

/// dtx resource URIs
pub mod uris {
    /// Project resource: dtx://project/{project_id}
    pub fn project(project_id: &str) -> String {
        format!("dtx://project/{}", project_id)
    }

    /// Resource status: dtx://project/{project_id}/resource/{resource_id}
    pub fn resource(project_id: &str, resource_id: &str) -> String {
        format!("dtx://project/{}/resource/{}", project_id, resource_id)
    }

    /// Resource logs: dtx://project/{project_id}/resource/{resource_id}/logs
    pub fn logs(project_id: &str, resource_id: &str) -> String {
        format!("dtx://project/{}/resource/{}/logs", project_id, resource_id)
    }

    /// Parse a dtx URI
    pub fn parse(uri: &str) -> Option<DxtUri> {
        if !uri.starts_with("dtx://") {
            return None;
        }

        let path = &uri[6..];
        let parts: Vec<&str> = path.split('/').collect();

        match parts.as_slice() {
            ["project", project_id] => Some(DxtUri::Project {
                project_id: project_id.to_string(),
            }),
            ["project", project_id, "resource", resource_id] => Some(DxtUri::Resource {
                project_id: project_id.to_string(),
                resource_id: resource_id.to_string(),
            }),
            ["project", project_id, "resource", resource_id, "logs"] => Some(DxtUri::Logs {
                project_id: project_id.to_string(),
                resource_id: resource_id.to_string(),
            }),
            _ => None,
        }
    }
}

#[derive(Clone, Debug)]
pub enum DxtUri {
    Project { project_id: String },
    Resource { project_id: String, resource_id: String },
    Logs { project_id: String, resource_id: String },
}
```

### 3.3.4 MCP Handler (3h)

**File:** `crates/dtx-protocol/src/mcp/handler.rs`

```rust
use async_trait::async_trait;
use serde_json::Value;

use crate::jsonrpc::{Request, Response, ErrorObject, error_codes, RequestId};
use crate::methods::methods;
use super::types::*;
use super::resources::*;
use super::tools::*;

/// MCP-specific handler trait
#[async_trait]
pub trait McpHandler: Send + Sync {
    /// Handle initialize
    async fn initialize(&self, params: InitializeParams) -> Result<InitializeResult, ErrorObject>;

    /// List available resources
    async fn list_resources(&self) -> Result<ListResourcesResult, ErrorObject>;

    /// Read a resource
    async fn read_resource(&self, params: ReadResourceParams) -> Result<ReadResourceResult, ErrorObject>;

    /// List available tools
    async fn list_tools(&self) -> Result<ListToolsResult, ErrorObject>;

    /// Call a tool
    async fn call_tool(&self, params: CallToolParams) -> Result<CallToolResult, ErrorObject>;
}

/// Default MCP handler implementation
pub struct DefaultMcpHandler<H> {
    inner: H,
    server_info: ServerInfo,
}

impl<H> DefaultMcpHandler<H> {
    pub fn new(inner: H) -> Self {
        Self {
            inner,
            server_info: ServerInfo {
                name: "dtx".to_string(),
                version: env!("CARGO_PKG_VERSION").to_string(),
            },
        }
    }
}

#[async_trait]
impl<H: crate::handler::ProtocolHandler> McpHandler for DefaultMcpHandler<H> {
    async fn initialize(&self, _params: InitializeParams) -> Result<InitializeResult, ErrorObject> {
        Ok(InitializeResult {
            protocol_version: "2024-11-05".to_string(),
            capabilities: ServerCapabilities::default(),
            server_info: self.server_info.clone(),
        })
    }

    async fn list_resources(&self) -> Result<ListResourcesResult, ErrorObject> {
        let result = self.inner.resource_list().await?;

        let resources = result.resources.iter().map(|r| {
            Resource {
                uri: uris::resource("default", &r.id),
                name: r.id.clone(),
                description: Some(format!("{} ({})", r.kind, r.state)),
                mime_type: Some("application/json".to_string()),
            }
        }).collect();

        Ok(ListResourcesResult { resources })
    }

    async fn read_resource(&self, params: ReadResourceParams) -> Result<ReadResourceResult, ErrorObject> {
        let uri = uris::parse(&params.uri)
            .ok_or_else(|| ErrorObject::new(error_codes::INVALID_PARAMS, "Invalid URI"))?;

        match uri {
            DxtUri::Resource { resource_id, .. } => {
                let status = self.inner.resource_status(crate::methods::ResourceParams {
                    id: resource_id,
                }).await?;

                Ok(ReadResourceResult {
                    contents: vec![ResourceContent {
                        uri: params.uri,
                        mime_type: Some("application/json".to_string()),
                        content: ResourceContentType::Text {
                            text: serde_json::to_string_pretty(&status).unwrap(),
                        },
                    }],
                })
            }
            DxtUri::Logs { resource_id, .. } => {
                let logs = self.inner.resource_logs(crate::methods::LogsParams {
                    id: resource_id,
                    follow: false,
                    lines: Some(100),
                }).await?;

                let text = logs.iter()
                    .map(|l| format!("[{}] {} {}", l.timestamp, l.stream, l.line))
                    .collect::<Vec<_>>()
                    .join("\n");

                Ok(ReadResourceResult {
                    contents: vec![ResourceContent {
                        uri: params.uri,
                        mime_type: Some("text/plain".to_string()),
                        content: ResourceContentType::Text { text },
                    }],
                })
            }
            _ => Err(ErrorObject::new(error_codes::RESOURCE_NOT_FOUND, "Resource not found")),
        }
    }

    async fn list_tools(&self) -> Result<ListToolsResult, ErrorObject> {
        Ok(ListToolsResult {
            tools: dtx_tools(),
        })
    }

    async fn call_tool(&self, params: CallToolParams) -> Result<CallToolResult, ErrorObject> {
        let result = match params.name.as_str() {
            "start_resource" => {
                let id = params.arguments
                    .and_then(|a| a.get("id").and_then(|v| v.as_str()).map(String::from))
                    .ok_or_else(|| ErrorObject::new(error_codes::INVALID_PARAMS, "Missing id"))?;

                self.inner.resource_start(crate::methods::ResourceParams { id }).await?;
                "Resource started".to_string()
            }
            "stop_resource" => {
                let id = params.arguments
                    .and_then(|a| a.get("id").and_then(|v| v.as_str()).map(String::from))
                    .ok_or_else(|| ErrorObject::new(error_codes::INVALID_PARAMS, "Missing id"))?;

                self.inner.resource_stop(crate::methods::ResourceParams { id }).await?;
                "Resource stopped".to_string()
            }
            "list_resources" => {
                let result = self.inner.resource_list().await?;
                serde_json::to_string_pretty(&result).unwrap()
            }
            "start_all" => {
                self.inner.start_all().await?;
                "All resources started".to_string()
            }
            "stop_all" => {
                self.inner.stop_all().await?;
                "All resources stopped".to_string()
            }
            _ => {
                return Err(ErrorObject::new(error_codes::METHOD_NOT_FOUND, "Unknown tool"));
            }
        };

        Ok(CallToolResult {
            content: vec![ToolContent::Text { text: result }],
            is_error: None,
        })
    }
}
```
