### 7.3.1 MiddlewarePlugin Trait (2h)

**File:** `crates/dtx-plugin/src/middleware.rs`

```rust
use dtx_core::middleware::{Middleware, MiddlewareStack};
use dtx_core::error::Result;

/// A plugin that provides middleware.
///
/// Middleware plugins add processing layers that wrap operations.
/// They can log, authenticate, rate limit, add metrics, etc.
pub trait MiddlewarePlugin: Send + Sync {
    /// Unique name for this middleware.
    fn name(&self) -> &str;

    /// Middleware version.
    fn version(&self) -> &str;

    /// Create the middleware instance.
    ///
    /// Configuration is passed as JSON from dtx config.
    fn create(&self, config: serde_json::Value) -> Result<Box<dyn Middleware>>;

    /// Validate configuration.
    fn validate_config(&self, config: &serde_json::Value) -> Result<()> {
        let _ = config;
        Ok(())
    }

    /// Priority in the middleware stack (lower = outer).
    ///
    /// Standard priorities:
    /// - 0-99: Security (auth, rate limiting)
    /// - 100-199: Logging, tracing
    /// - 200-299: Metrics
    /// - 300-399: Transformation
    /// - 400-499: Caching
    /// - 500+: Custom
    fn priority(&self) -> u32 {
        500
    }

    /// Middleware capabilities/behavior.
    fn metadata(&self) -> MiddlewareMetadata {
        MiddlewareMetadata::default()
    }
}

/// Metadata about middleware behavior.
#[derive(Clone, Debug, Default)]
pub struct MiddlewareMetadata {
    /// Whether this middleware may reject requests.
    pub may_reject: bool,
    /// Whether this middleware modifies operations.
    pub modifies_operation: bool,
    /// Whether this middleware modifies responses.
    pub modifies_response: bool,
    /// Whether this middleware is async-only.
    pub async_required: bool,
    /// Dependencies on other middleware.
    pub requires: Vec<String>,
    /// Description for documentation.
    pub description: Option<String>,
}

impl MiddlewareMetadata {
    pub fn new() -> Self {
        Self::default()
    }

    pub fn may_reject(mut self) -> Self {
        self.may_reject = true;
        self
    }

    pub fn modifies_operation(mut self) -> Self {
        self.modifies_operation = true;
        self
    }

    pub fn modifies_response(mut self) -> Self {
        self.modifies_response = true;
        self
    }

    pub fn requires(mut self, name: impl Into<String>) -> Self {
        self.requires.push(name.into());
        self
    }

    pub fn with_description(mut self, desc: impl Into<String>) -> Self {
        self.description = Some(desc.into());
        self
    }
}
```

### 7.3.2 Middleware Registry (2h)

**File:** `crates/dtx-plugin/src/middleware_registry.rs`

```rust
use std::collections::HashMap;
use std::sync::Arc;

use dtx_core::middleware::{Middleware, MiddlewareStack, MiddlewareStackBuilder};
use dtx_core::error::Result;

use super::middleware::{MiddlewarePlugin, MiddlewareMetadata};

/// Registry of middleware plugins.
pub struct MiddlewareRegistry {
    plugins: HashMap<String, Arc<dyn MiddlewarePlugin>>,
}

impl MiddlewareRegistry {
    pub fn new() -> Self {
        Self {
            plugins: HashMap::new(),
        }
    }

    /// Register a middleware plugin.
    pub fn register(&mut self, plugin: Arc<dyn MiddlewarePlugin>) -> &mut Self {
        self.plugins.insert(plugin.name().to_string(), plugin);
        self
    }

    /// Get a plugin by name.
    pub fn get(&self, name: &str) -> Option<Arc<dyn MiddlewarePlugin>> {
        self.plugins.get(name).cloned()
    }

    /// Check if a middleware is registered.
    pub fn has_middleware(&self, name: &str) -> bool {
        self.plugins.contains_key(name)
    }

    /// List all registered middleware.
    pub fn list(&self) -> Vec<MiddlewareInfo> {
        let mut list: Vec<_> = self.plugins
            .values()
            .map(|p| MiddlewareInfo {
                name: p.name().to_string(),
                version: p.version().to_string(),
                priority: p.priority(),
                metadata: p.metadata(),
            })
            .collect();

        // Sort by priority
        list.sort_by_key(|m| m.priority);
        list
    }

    /// Create middleware instances from configuration.
    ///
    /// Returns middleware sorted by priority (lowest first = outermost).
    pub fn create_from_config(
        &self,
        config: &MiddlewareStackConfig,
    ) -> Result<Vec<Box<dyn Middleware>>> {
        let mut middlewares: Vec<(u32, Box<dyn Middleware>)> = Vec::new();

        for (name, mw_config) in &config.middleware {
            let plugin = self.plugins.get(name).ok_or_else(|| {
                dtx_core::error::Error::config(format!("unknown middleware: {}", name))
            })?;

            // Validate config
            plugin.validate_config(mw_config)?;

            // Create instance
            let middleware = plugin.create(mw_config.clone())?;

            // Use configured priority or default
            let priority = config.priorities.get(name).copied()
                .unwrap_or_else(|| plugin.priority());

            middlewares.push((priority, middleware));
        }

        // Sort by priority (lower = added first = outermost)
        middlewares.sort_by_key(|(p, _)| *p);

        Ok(middlewares.into_iter().map(|(_, m)| m).collect())
    }

    /// Build a middleware stack from configuration.
    pub fn build_stack<H: dtx_core::middleware::Handler + 'static>(
        &self,
        config: &MiddlewareStackConfig,
        handler: H,
    ) -> Result<dtx_core::middleware::MiddlewareChain> {
        let middlewares = self.create_from_config(config)?;

        let mut builder = MiddlewareStackBuilder::new();
        for mw in middlewares {
            builder = builder.layer_arc(Arc::from(mw));
        }

        Ok(builder.build(handler))
    }
}

impl Default for MiddlewareRegistry {
    fn default() -> Self {
        Self::new()
    }
}

/// Information about registered middleware.
#[derive(Clone, Debug)]
pub struct MiddlewareInfo {
    pub name: String,
    pub version: String,
    pub priority: u32,
    pub metadata: MiddlewareMetadata,
}

/// Configuration for building a middleware stack.
#[derive(Clone, Debug, Default, serde::Deserialize)]
pub struct MiddlewareStackConfig {
    /// Middleware configurations by name.
    #[serde(default)]
    pub middleware: HashMap<String, serde_json::Value>,

    /// Override priorities for specific middleware.
    #[serde(default)]
    pub priorities: HashMap<String, u32>,

    /// Middleware to disable even if configured.
    #[serde(default)]
    pub disabled: Vec<String>,
}
```

### 7.3.4 Example: Auth Middleware Plugin (3h)

**File:** `crates/dtx-plugin/examples/auth-middleware/src/lib.rs`

```rust
//! Example authentication middleware plugin.
//!
//! Demonstrates how to create a middleware plugin that validates
//! tokens and rejects unauthorized requests.

use async_trait::async_trait;
use serde::Deserialize;
use std::sync::Arc;

use dtx_core::{
    context::Context,
    middleware::{Middleware, Next, Operation, Response},
    error::Result,
};
use dtx_plugin::{
    dtx_plugin, Plugin, MiddlewarePlugin, MiddlewareMetadata,
};

// --- Plugin Registration ---

pub struct AuthPlugin;

impl Plugin for AuthPlugin {
    fn name(&self) -> &str { "auth" }
    fn version(&self) -> &str { env!("CARGO_PKG_VERSION") }
    fn api_version(&self) -> u32 { 2 }

    fn middleware(&self) -> Vec<Box<dyn MiddlewarePlugin>> {
        vec![
            Box::new(TokenAuthMiddlewarePlugin),
            Box::new(RbacMiddlewarePlugin),
        ]
    }
}

impl AuthPlugin {
    pub fn new() -> Self { Self }
}

dtx_plugin!(AuthPlugin);

// --- Token Auth Middleware ---

pub struct TokenAuthMiddlewarePlugin;

impl MiddlewarePlugin for TokenAuthMiddlewarePlugin {
    fn name(&self) -> &str { "token-auth" }
    fn version(&self) -> &str { env!("CARGO_PKG_VERSION") }

    fn create(&self, config: serde_json::Value) -> Result<Box<dyn Middleware>> {
        let config: TokenAuthConfig = serde_json::from_value(config)?;
        Ok(Box::new(TokenAuthMiddleware::new(config)))
    }

    fn priority(&self) -> u32 { 10 } // Very early in the chain

    fn metadata(&self) -> MiddlewareMetadata {
        MiddlewareMetadata::new()
            .may_reject()
            .with_description("Bearer token authentication")
    }
}

#[derive(Clone, Debug, Deserialize)]
pub struct TokenAuthConfig {
    /// Secret for token validation.
    pub secret: String,

    /// Token header name.
    #[serde(default = "default_header")]
    pub header: String,

    /// Operations that don't require auth.
    #[serde(default)]
    pub skip_operations: Vec<String>,
}

fn default_header() -> String {
    "Authorization".to_string()
}

pub struct TokenAuthMiddleware {
    config: TokenAuthConfig,
}

impl TokenAuthMiddleware {
    pub fn new(config: TokenAuthConfig) -> Self {
        Self { config }
    }

    fn validate_token(&self, token: &str) -> bool {
        // Simplified validation - in production use proper JWT
        token.starts_with("dtx_") && token.len() > 10
    }

    fn should_skip(&self, op: &Operation) -> bool {
        let op_name = op.name();
        self.config.skip_operations.iter().any(|s| s == op_name)
    }
}

#[async_trait]
impl Middleware for TokenAuthMiddleware {
    fn name(&self) -> &'static str { "token-auth" }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        // Skip auth for configured operations
        if self.should_skip(&op) {
            return next.run(op, ctx).await;
        }

        // Get token from context
        let token = ctx.get::<String>(&self.config.header);

        match token {
            Some(t) if self.validate_token(&t) => {
                // Token valid, continue
                next.run(op, ctx).await
            }
            Some(_) => {
                // Invalid token
                tracing::warn!(op = %op.name(), "invalid auth token");
                Ok(Response::error("unauthorized: invalid token"))
            }
            None => {
                // No token
                tracing::warn!(op = %op.name(), "missing auth token");
                Ok(Response::error("unauthorized: missing token"))
            }
        }
    }
}

// --- RBAC Middleware ---

pub struct RbacMiddlewarePlugin;

impl MiddlewarePlugin for RbacMiddlewarePlugin {
    fn name(&self) -> &str { "rbac" }
    fn version(&self) -> &str { env!("CARGO_PKG_VERSION") }

    fn create(&self, config: serde_json::Value) -> Result<Box<dyn Middleware>> {
        let config: RbacConfig = serde_json::from_value(config)?;
        Ok(Box::new(RbacMiddleware::new(config)))
    }

    fn priority(&self) -> u32 { 20 } // After auth

    fn metadata(&self) -> MiddlewareMetadata {
        MiddlewareMetadata::new()
            .may_reject()
            .requires("token-auth")
            .with_description("Role-based access control")
    }
}

#[derive(Clone, Debug, Deserialize)]
pub struct RbacConfig {
    /// Role permissions.
    pub roles: std::collections::HashMap<String, Vec<String>>,
}

pub struct RbacMiddleware {
    config: RbacConfig,
}

impl RbacMiddleware {
    pub fn new(config: RbacConfig) -> Self {
        Self { config }
    }

    fn check_permission(&self, role: &str, operation: &str) -> bool {
        self.config.roles
            .get(role)
            .map(|perms| perms.iter().any(|p| p == operation || p == "*"))
            .unwrap_or(false)
    }
}

#[async_trait]
impl Middleware for RbacMiddleware {
    fn name(&self) -> &'static str { "rbac" }

    async fn handle(
        &self,
        op: Operation,
        ctx: Context,
        next: Next<'_>,
    ) -> Result<Response> {
        let role = ctx.get::<String>("user_role").unwrap_or_default();
        let op_name = op.name();

        if self.check_permission(&role, op_name) {
            next.run(op, ctx).await
        } else {
            tracing::warn!(role = %role, op = %op_name, "permission denied");
            Ok(Response::error(format!(
                "forbidden: role '{}' cannot perform '{}'",
                role, op_name
            )))
        }
    }
}
```

### 7.3.5 Middleware Plugin Configuration (2h)

**File:** `crates/dtx-core/src/config/middleware.rs`

```rust
use serde::{Deserialize, Serialize};
use std::collections::HashMap;

/// Configuration for middleware in .dtx/config.yaml.
#[derive(Clone, Debug, Default, Serialize, Deserialize)]
pub struct MiddlewareConfiguration {
    /// Enable/disable middleware stack.
    #[serde(default = "default_true")]
    pub enabled: bool,

    /// Middleware configurations.
    ///
    /// Each key is a middleware name, value is its config.
    #[serde(default)]
    pub layers: HashMap<String, MiddlewareLayerConfig>,

    /// Default middleware to include (can be removed with `!name`).
    #[serde(default = "default_middleware")]
    pub defaults: Vec<String>,
}

fn default_true() -> bool { true }

fn default_middleware() -> Vec<String> {
    vec!["logging".to_string()]
}

/// Configuration for a single middleware layer.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct MiddlewareLayerConfig {
    /// Enable/disable this middleware.
    #[serde(default = "default_true")]
    pub enabled: bool,

    /// Override default priority.
    pub priority: Option<u32>,

    /// Middleware-specific configuration.
    #[serde(flatten)]
    pub config: serde_json::Value,
}

impl MiddlewareConfiguration {
    /// Get the final list of middleware to enable.
    pub fn effective_middleware(&self) -> Vec<String> {
        if !self.enabled {
            return Vec::new();
        }

        let mut middleware: Vec<String> = self.defaults.clone();

        // Add configured middleware
        for (name, cfg) in &self.layers {
            if cfg.enabled && !middleware.contains(name) {
                middleware.push(name.clone());
            }
        }

        // Remove disabled (prefixed with !)
        middleware.retain(|m| !m.starts_with('!'));

        middleware
    }

    /// Get configuration for a specific middleware.
    pub fn get_config(&self, name: &str) -> Option<&MiddlewareLayerConfig> {
        self.layers.get(name)
    }
}

/// Parse middleware configuration from .dtx/config.yaml.
pub fn parse_middleware_config(value: &serde_yaml::Value) -> Result<MiddlewareConfiguration, serde_yaml::Error> {
    serde_yaml::from_value(value.clone())
}
```

## Middleware Ordering

```
Priority ranges (lower = executed first = outermost):

  0-49:   Security (auth, rate limiting)
 50-99:   Validation, input transformation
100-149:  Logging, tracing
150-199:  Caching
200-249:  Metrics, observability
250-299:  Retry, timeout
300-399:  Business logic hooks
400-499:  Output transformation
500+:     Custom/plugin default
```

Example stack execution order:

```
Request:  Auth(10) → Logging(100) → Metrics(200) → Timeout(250) → Handler
Response: Handler → Timeout(250) → Metrics(200) → Logging(100) → Auth(10)
```
