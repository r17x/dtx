### 3.2.1 Transport Trait (2h)

**File:** `crates/dtx-protocol/src/transport/mod.rs`

```rust
use async_trait::async_trait;
use crate::jsonrpc::{Request, Response};

/// Transport abstraction for sending/receiving messages
#[async_trait]
pub trait Transport: Send + Sync {
    /// Send a request and receive a response
    async fn send(&self, request: Request) -> Result<Response, TransportError>;

    /// Send a notification (no response expected)
    async fn notify(&self, request: Request) -> Result<(), TransportError>;

    /// Close the transport
    async fn close(&self) -> Result<(), TransportError>;
}

/// Server-side transport for receiving requests
#[async_trait]
pub trait ServerTransport: Send + Sync {
    /// Receive the next request
    async fn recv(&mut self) -> Result<Request, TransportError>;

    /// Send a response
    async fn send(&mut self, response: Response) -> Result<(), TransportError>;

    /// Send a notification to the client
    async fn notify(&mut self, request: Request) -> Result<(), TransportError>;
}

/// Transport error
#[derive(Debug, thiserror::Error)]
pub enum TransportError {
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),

    #[error("Serialization error: {0}")]
    Serialization(#[from] serde_json::Error),

    #[error("Connection closed")]
    Closed,

    #[error("Timeout")]
    Timeout,

    #[error("Transport error: {0}")]
    Other(String),
}
```

**Acceptance:**
- [ ] Transport trait for clients
- [ ] ServerTransport for servers
- [ ] Error types defined

### 3.2.2 stdio Transport (3h)

**File:** `crates/dtx-protocol/src/transport/stdio.rs`

```rust
use async_trait::async_trait;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader, BufWriter};
use tokio::io::{Stdin, Stdout};

use crate::jsonrpc::{Request, Response};
use super::{ServerTransport, TransportError};

/// stdio transport for MCP compatibility
pub struct StdioTransport {
    reader: BufReader<Stdin>,
    writer: BufWriter<Stdout>,
}

impl StdioTransport {
    pub fn new() -> Self {
        Self {
            reader: BufReader::new(tokio::io::stdin()),
            writer: BufWriter::new(tokio::io::stdout()),
        }
    }
}

impl Default for StdioTransport {
    fn default() -> Self {
        Self::new()
    }
}

#[async_trait]
impl ServerTransport for StdioTransport {
    async fn recv(&mut self) -> Result<Request, TransportError> {
        let mut line = String::new();
        let bytes_read = self.reader.read_line(&mut line).await?;

        if bytes_read == 0 {
            return Err(TransportError::Closed);
        }

        let request: Request = serde_json::from_str(&line)?;
        Ok(request)
    }

    async fn send(&mut self, response: Response) -> Result<(), TransportError> {
        let json = serde_json::to_string(&response)?;
        self.writer.write_all(json.as_bytes()).await?;
        self.writer.write_all(b"\n").await?;
        self.writer.flush().await?;
        Ok(())
    }

    async fn notify(&mut self, request: Request) -> Result<(), TransportError> {
        let json = serde_json::to_string(&request)?;
        self.writer.write_all(json.as_bytes()).await?;
        self.writer.write_all(b"\n").await?;
        self.writer.flush().await?;
        Ok(())
    }
}

/// Run a server with stdio transport
pub async fn serve_stdio<H, F>(handler: H) -> Result<(), TransportError>
where
    H: Fn(Request) -> F + Send + Sync,
    F: std::future::Future<Output = Response> + Send,
{
    let mut transport = StdioTransport::new();

    loop {
        match transport.recv().await {
            Ok(request) => {
                let is_notification = request.is_notification();
                let response = handler(request).await;

                if !is_notification {
                    transport.send(response).await?;
                }
            }
            Err(TransportError::Closed) => break,
            Err(e) => {
                eprintln!("Transport error: {}", e);
            }
        }
    }

    Ok(())
}
```

**Acceptance:**
- [ ] Read requests from stdin
- [ ] Write responses to stdout
- [ ] Newline-delimited JSON
- [ ] Handle EOF

### 3.2.3 HTTP Transport (3h)

**File:** `crates/dtx-protocol/src/transport/http.rs`

```rust
use async_trait::async_trait;
use reqwest::Client;

use crate::jsonrpc::{Request, Response};
use super::{Transport, TransportError};

/// HTTP client transport
pub struct HttpTransport {
    client: Client,
    url: String,
}

impl HttpTransport {
    pub fn new(url: impl Into<String>) -> Self {
        Self {
            client: Client::new(),
            url: url.into(),
        }
    }

    pub fn with_client(client: Client, url: impl Into<String>) -> Self {
        Self {
            client,
            url: url.into(),
        }
    }
}

#[async_trait]
impl Transport for HttpTransport {
    async fn send(&self, request: Request) -> Result<Response, TransportError> {
        let response = self.client
            .post(&self.url)
            .json(&request)
            .send()
            .await
            .map_err(|e| TransportError::Other(e.to_string()))?;

        let json_response: Response = response
            .json()
            .await
            .map_err(|e| TransportError::Other(e.to_string()))?;

        Ok(json_response)
    }

    async fn notify(&self, request: Request) -> Result<(), TransportError> {
        self.client
            .post(&self.url)
            .json(&request)
            .send()
            .await
            .map_err(|e| TransportError::Other(e.to_string()))?;

        Ok(())
    }

    async fn close(&self) -> Result<(), TransportError> {
        Ok(())
    }
}

/// Axum handler for HTTP transport
#[cfg(feature = "axum")]
pub mod axum_handler {
    use axum::{Json, response::IntoResponse};
    use super::*;

    pub async fn jsonrpc_handler<H, F>(
        handler: H,
        Json(request): Json<Request>,
    ) -> impl IntoResponse
    where
        H: Fn(Request) -> F + Send + Sync,
        F: std::future::Future<Output = Response> + Send,
    {
        let response = handler(request).await;
        Json(response)
    }
}
```

**Acceptance:**
- [ ] HTTP client transport
- [ ] Axum handler for server
- [ ] JSON body serialization

### 3.2.4 WebSocket Transport (3h)

**File:** `crates/dtx-protocol/src/transport/websocket.rs`

```rust
use async_trait::async_trait;
use tokio::sync::mpsc;
use futures::{SinkExt, StreamExt};

use crate::jsonrpc::{Request, Response};
use super::{Transport, TransportError};

/// WebSocket client transport
pub struct WebSocketTransport {
    sender: mpsc::Sender<String>,
    receiver: mpsc::Receiver<String>,
}

impl WebSocketTransport {
    pub async fn connect(url: &str) -> Result<Self, TransportError> {
        use tokio_tungstenite::connect_async;

        let (ws_stream, _) = connect_async(url)
            .await
            .map_err(|e| TransportError::Other(e.to_string()))?;

        let (mut write, mut read) = ws_stream.split();

        let (tx_out, mut rx_out) = mpsc::channel::<String>(32);
        let (tx_in, rx_in) = mpsc::channel::<String>(32);

        // Writer task
        tokio::spawn(async move {
            while let Some(msg) = rx_out.recv().await {
                if write.send(tokio_tungstenite::tungstenite::Message::Text(msg)).await.is_err() {
                    break;
                }
            }
        });

        // Reader task
        tokio::spawn(async move {
            while let Some(Ok(msg)) = read.next().await {
                if let tokio_tungstenite::tungstenite::Message::Text(text) = msg {
                    if tx_in.send(text).await.is_err() {
                        break;
                    }
                }
            }
        });

        Ok(Self {
            sender: tx_out,
            receiver: rx_in,
        })
    }

    pub async fn recv(&mut self) -> Result<Response, TransportError> {
        let text = self.receiver.recv().await
            .ok_or(TransportError::Closed)?;
        serde_json::from_str(&text).map_err(TransportError::from)
    }
}

#[async_trait]
impl Transport for WebSocketTransport {
    async fn send(&self, request: Request) -> Result<Response, TransportError> {
        let json = serde_json::to_string(&request)?;
        self.sender.send(json).await
            .map_err(|_| TransportError::Closed)?;

            .map_err(|_| TransportError::Closed)?;
        Ok(())
    }
```
