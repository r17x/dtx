### 5.2.1 Intent Parser (3h)

```rust
// crates/dtx-protocol/src/nl/parser.rs

pub struct IntentParser {
    provider: Arc<dyn AIProvider>,
}

#[derive(Debug, Serialize, Deserialize)]
pub struct ParsedIntent {
    pub operation: String,
    pub targets: Vec<String>,
    pub options: HashMap<String, String>,
    pub confidence: f32,
}

impl IntentParser {
    pub async fn parse(&self, input: &str) -> Result<ParsedIntent, Error> {
        let prompt = format!(
            r#"Parse this command into structured intent. Return JSON with:
- operation: start|stop|restart|status|logs
- targets: list of resource names (empty = all)
- options: any modifiers
- confidence: 0.0-1.0

Input: "{}"
JSON:"#,
            input
        );

        let response = self.provider.complete(&prompt).await?;
        serde_json::from_str(&response).map_err(|e| Error::Serialization(e.to_string()))
    }
}
```

### 5.2.2 Intent Executor (2h)

```rust
// crates/dtx-protocol/src/nl/executor.rs

pub struct IntentExecutor {
    orchestrator: Arc<Orchestrator>,
}

impl IntentExecutor {
    pub async fn execute(&self, intent: ParsedIntent, ctx: &Context) -> Result<Response> {
        match intent.operation.as_str() {
            "start" => {
                if intent.targets.is_empty() {
                    self.orchestrator.start_all(ctx).await
                } else {
                    for target in &intent.targets {
                        self.orchestrator.start(&ResourceId::new(target), ctx).await?;
                    }
                    Ok(Response::ok())
                }
            }
            "stop" => {
                if intent.targets.is_empty() {
                    self.orchestrator.stop_all(ctx).await
                } else {
                    for target in &intent.targets {
                        self.orchestrator.stop(&ResourceId::new(target), ctx).await?;
                    }
                    Ok(Response::ok())
                }
            }
            "status" => {
                // Return status for targets
                Ok(Response::data(self.get_status(&intent.targets).await))
            }
            _ => Err(Error::InvalidState { operation: intent.operation, .. })
        }
    }
}
```

### 5.2.3 CLI Integration (2h)

```rust
// crates/dtx/src/cmd/run.rs

pub async fn run_command(args: RunArgs) -> Result<()> {
    let parser = IntentParser::new(ai_provider);
    let executor = IntentExecutor::new(orchestrator);

    let input = args.command.join(" ");
    let intent = parser.parse(&input).await?;

    if intent.confidence < 0.7 {
        println!("I understood: {} {}", intent.operation, intent.targets.join(", "));
        print!("Is this correct? [Y/n] ");
        // Confirm with user
    }

    let ctx = Context::with_timeout(Duration::from_secs(60));
    executor.execute(intent, &ctx).await
}
```
