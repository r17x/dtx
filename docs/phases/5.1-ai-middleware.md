### 5.1.2 AI Middleware (3h)

```rust
// crates/dtx-middleware/src/ai/middleware.rs

pub struct AIMiddleware {
    provider: Arc<dyn AIProvider>,
    config: AIConfig,
}

pub struct AIConfig {
    pub suggest_on_error: bool,
    pub suggest_health_checks: bool,
    pub explain_failures: bool,
}

#[async_trait]
impl Middleware for AIMiddleware {
    fn name(&self) -> &'static str { "ai" }

    async fn handle(&self, op: Operation, ctx: Context, next: Next<'_>) -> Result<Response> {
        let result = next.run(op.clone(), ctx.clone()).await;

        if let Err(ref error) = result {
            if self.config.explain_failures {
                let explanation = self.explain_error(error).await;
                tracing::info!(explanation = %explanation, "AI explanation");
            }
        }

        result
    }
}

impl AIMiddleware {
    async fn explain_error(&self, error: &Error) -> String {
        let prompt = format!(
            "Explain this dtx orchestration error in simple terms and suggest a fix:\n{}",
            error
        );
        self.provider.complete(&prompt).await.unwrap_or_default()
    }
}
```

### 5.1.3 Config Suggestions (2h)

```rust
impl AIMiddleware {
    pub async fn suggest_health_check(&self, service: &Service) -> Option<ProbeConfig> {
        let prompt = format!(
            "Given a service running '{}' on port {:?}, suggest a health check configuration as JSON.",
            service.command,
            service.port
        );

        let response = self.provider.complete(&prompt).await.ok()?;
        serde_json::from_str(&response).ok()
    }

    pub async fn suggest_dependencies(&self, services: &[Service]) -> Vec<Dependency> {
        let prompt = format!(
            "Given these services: {:?}, suggest dependencies between them as JSON.",
            services.iter().map(|s| &s.name).collect::<Vec<_>>()
        );

        let response = self.provider.complete(&prompt).await.ok();
        response.and_then(|r| serde_json::from_str(&r).ok()).unwrap_or_default()
    }
}
```
